# 项目1: 构建有效前沿
import numpy as np
import pandas as pd
from scipy.optimize import minimize

def portfolio_variance(weights, cov_matrix):
    return weights @ cov_matrix @ weights.T

def portfolio_return(weights, returns):
    return weights @ returns

def efficient_frontier(returns, cov_matrix, n_points=100):
    # 生成有效前沿
    target_returns = np.linspace(returns.min(), returns.max(), n_points)
    frontier = []
    
    for target in target_returns:
        constraints = [
            {'type': 'eq', 'fun': lambda w: np.sum(w) - 1},
            {'type': 'eq', 'fun': lambda w: portfolio_return(w, returns) - target}
        ]
        bounds = tuple((0, 1) for _ in range(len(returns)))
        
        result = minimize(
            portfolio_variance,
            x0=np.ones(len(returns)) / len(returns),
            args=(cov_matrix,),
            method='SLSQP',
            bounds=bounds,
            constraints=constraints
        )
        
        frontier.append({
            'return': target,
            'risk': np.sqrt(result.fun),
            'weights': result.x
        })
    
    return frontier

# 项目2: 构建Fama-French三因子模型
import statsmodels.api as sm

def fama_french_regression(stock_returns, market_return, smb, hml):
    """
    Fama-French三因子回归
    R_it - R_ft = α + β(MKT) + s(SMB) + h(HML) + ε
    """
    X = pd.DataFrame({
        'MKT': market_return,
        'SMB': smb,
        'HML': hml
    })
    X = sm.add_constant(X)
    
    model = sm.OLS(stock_returns, X).fit()
    
    return {
        'alpha': model.params['const'],
        'beta': model.params['MKT'],
        'smb_loading': model.params['SMB'],
        'hml_loading': model.params['HML'],
        'r_squared': model.rsquared,
        'summary': model.summary()
    }
