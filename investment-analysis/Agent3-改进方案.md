**矛盾1：知识数量 vs 应用质量**

- 深度分析丰富，但代码化率<20%

- 理论知识丰富，但实战经验不足
- **结论**：目标过高，需降低预期或提升能力

# 优先级1：残差动量因子（提升IR 50%）
def residual_momentum(returns, market_returns, sector_returns):
    """
    剥离市场和行业因子后的纯动量
    知识来源：quantitative/01-动量因子改进方法.md
    """
    # 市场因子回归
    beta_mkt = np.cov(returns, market_returns)[0,1] / np.var(market_returns)
    residual = returns - beta_mkt * market_returns
    
    # 行业因子回归
    beta_sector = np.cov(residual, sector_returns)[0,1] / np.var(sector_returns)
    pure_momentum = residual - beta_sector * sector_returns
    
    return pure_momentum.rolling(20).sum()

def economic_cycle_position(pmi, inventory_growth, credit_growth):
    """
    判断当前经济周期位置
    知识来源：finance/deep-analysis/01-china-economic-cycle.md

# 优先级3：情绪指标（避免极端买入）
def market_sentiment(vix, put_call_ratio, margin_debt):
    """
    综合市场情绪指标
    需补充：情绪指标数据源
    """
    sentiment_score = (
        (vix - vix.median()) / vix.std() +
        (put_call_ratio - put_call_ratio.median()) / put_call_ratio.std() +
        (margin_debt - margin_debt.median()) / margin_debt.std()
    ) / 3
    return sentiment_score

# 优先级4：市场环境识别（自适应切换）
def market_regime(price, volume, ma60):
    """
    识别市场状态：牛市/震荡/熊市
    知识来源：Agent 2 建议
    """
    trend = price > ma60
    volume_surge = volume > volume.rolling(20).mean() * 1.5
    
    if trend and volume_surge:
        return "牛市"
    elif not trend and volume < volume.rolling(20).mean() * 0.7:
        return "熊市"
    else:
        return "震荡"

# 优先级5：止盈止损优化（降低回撤）
def dynamic_stop_loss(entry_price, current_price, atr, holding_days):
    """
    动态止盈止损
    知识来源：finance/deep-analysis/06-止盈止损执行纪律.md
    """
    profit_pct = (current_price - entry_price) / entry_price
    
    # 止损：初期-8%，后期动态调整
    if holding_days < 10:
        stop_loss = -0.08
    else:
        stop_loss = max(-0.15, profit_pct - 3 * atr / entry_price)
    
    # 止盈：盈利10%后启动移动止盈
    if profit_pct > 0.10:
        take_profit = max(0.05, profit_pct - 0.05)
    else:
        take_profit = None
    
    return stop_loss, take_profit
```

            '核心仓位': 0.60,

            '策略风格': '激进'
        }

            '策略风格': '稳健'
        }

            '核心仓位': 0.50,
            '卫星仓位': 0.10,
            '现金': 0.40,
            '策略风格': '防御'
        }
    
    # 估值调整
    if valuation_percentile < 0.3:
        allocation['卫星仓位'] += 0.10
        allocation['现金'] -= 0.10
    elif valuation_percentile > 0.7:
        allocation['现金'] += 0.10
        allocation['卫星仓位'] -= 0.10
    
    # 情绪调整
    if sentiment_score > 2:  # 极端乐观，减仓
        allocation['现金'] += 0.10
    elif sentiment_score < -2:  # 极端悲观，加仓
        allocation['卫星仓位'] += 0.10
    
    return allocation

class AutomatedInvestmentSystem:
    def __init__(self):
        self.market_detector = MarketRegimeDetector()
        self.valuation_checker = ValuationChecker()
        self.sentiment_analyzer = SentimentAnalyzer()
        self.risk_manager = RiskManager()
        self.executor = Executor()
        
    def make_decision(self):
        """自动化投资决策"""
        # 1. 市场环境判断
        market = self.market_detector.detect()
        
        # 2. 估值检查
        valuation = self.valuation_checker.check()
        
        # 3. 情绪分析
        sentiment = self.sentiment_analyzer.analyze()
        
        # 4. 基础仓位
        base_allocation = self.get_base_allocation(market)
        
        # 5. 估值调整
        valuation_adj = self.valuation_adjustment(valuation)
        
        # 6. 情绪调整
        sentiment_adj = self.sentiment_adjustment(sentiment)
        
        # 7. 最终仓位
        final_allocation = base_allocation + valuation_adj + sentiment_adj
        
        # 8. 风险检查
        if self.risk_manager.check(final_allocation):
            self.executor.execute(final_allocation)
        else:
            self.executor.execute(self.risk_manager.safe_allocation())
        
        return final_allocation
```

def parameter_robustness_test():
    """
    参数稳健性测试：确保参数在不同市场环境下都有效
    """
    # 测试参数范围
    param_ranges = {
        'ma_fast': [10, 15, 20, 25, 30],
        'ma_slow': [50, 60, 70, 80, 90],
        'stop_loss': [-0.05, -0.08, -0.10, -0.12],
    }
    
    # 滚动前向测试
    for train_start in range(0, len(data) - 252, 126):  # 每半年滚动
        train_end = train_start + 252 * 2  # 2年训练
        test_end = train_end + 126  # 半年测试
        
        # 在训练集优化
        best_params = optimize(data[train_start:train_end], param_ranges)
        
        # 在测试集验证
        test_result = backtest(data[train_end:test_end], best_params)
        
        # 记录结果
        results.append({
            'train_period': f'{train_start}-{train_end}',
            'test_period': f'{train_end}-{test_end}',
            'params': best_params,
            'test_return': test_result['return'],
            'test_drawdown': test_result['drawdown']
        })
    
    # 稳健性评估
    avg_test_return = np.mean([r['test_return'] for r in results])
    return_std = np.std([r['test_return'] for r in results])
    
    return avg_test_return, return_std
