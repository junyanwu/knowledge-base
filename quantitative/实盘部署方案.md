        获取实时数据
        
        Args:
            symbol: 标的代码
            
        Returns:
            市场数据
        """
        # 先从Redis缓存读取
        cache_key = f"realtime:{symbol}"
        cached_data = self.redis_client.get(cache_key)
        
        if cached_data:
            data = json.loads(cached_data)
            return MarketData(**data)
        
        # 从数据源获取
        try:
            if self.data_sources['akshare']:
                df = self.data_sources['akshare'].fund_etf_spot_em()

                    timestamp=datetime.now(),

                    close=float(etf_data['最新价']),
                    volume=float(etf_data['成交量'])
                )
                
                # 缓存数据（有效期5秒）
                self.redis_client.setex(
                    cache_key,
                    5,
                    json.dumps(market_data.__dict__, default=str)
                )
                
                return market_data
        except Exception as e:
            self.logger.error(f"获取实时数据失败 {symbol}: {e}")
        
        return None
    
    async def get_historical_data(self,
                                   symbol: str,
                                   start_date: str,
                                   end_date: str,
                                   freq: str = 'daily') -> pd.DataFrame:
        """
        获取历史数据
        
        Args:
            symbol: 标的代码
            start_date: 开始日期
            end_date: 结束日期
            freq: 频率
            
        Returns:
            历史数据DataFrame
        """
        # 先从MongoDB读取
        collection = self.db['historical_data']
        query = {
            'symbol': symbol,
            'date': {'$gte': start_date, '$lte': end_date}
        }
        
        cached_data = list(collection.find(query, {'_id': 0}))
        
        if len(cached_data) > 0:
            df = pd.DataFrame(cached_data)
            df['date'] = pd.to_datetime(df['date'])
            df = df.set_index('date')
            return df
        
        # 从数据源获取
        try:
            if self.data_sources['tushare']:
                df = self.data_sources['tushare'].pro_bar(
                    ts_code=symbol,
                    start_date=start_date,
                    end_date=end_date,
                    freq=freq
                )
                
                if len(df) > 0:
                    # 存储到MongoDB
                    df['symbol'] = symbol
                    records = df.to_dict('records')
                    collection.insert_many(records)
                    
                    return df
        except Exception as e:
            self.logger.error(f"获取历史数据失败 {symbol}: {e}")
        
        return pd.DataFrame()
    
    def update_data(self, symbol: str):
        """更新数据"""
        # 获取最新数据
        end_date = datetime.now().strftime('%Y%m%d')
        start_date = (datetime.now() - timedelta(days=365)).strftime('%Y%m%d')
        
        # 异步更新
        loop = asyncio.new_event_loop()
        asyncio.set_event_loop(loop)
        df = loop.run_until_complete(
            self.get_historical_data(symbol, start_date, end_date)
        )
        loop.close()
        
        return df


class StrategyLayer:
    """策略层"""
    
    def __init__(self, config: Dict):
        """
        初始化策略层
        
        Args:
            config: 配置字典
        """
        self.config = config
        self.logger = logging.getLogger('StrategyLayer')
        
        # 策略池
        self.strategies = {}
        
        # 信号队列
        self.signal_queue = asyncio.Queue()
        
        # 策略配置
        self.strategy_config = config.get('strategy_config', {})
        
    def register_strategy(self,
                           name: str,
                           strategy_func: Callable,
                           params: Dict):
        """
        注册策略
        
        Args:
            name: 策略名称
            strategy_func: 策略函数
            params: 策略参数
        """
        self.strategies[name] = {
            'func': strategy_func,
            'params': params,
            'enabled': True
        }
        self.logger.info(f"注册策略: {name}")
    
    async def generate_signals(self,
                                 market_data: Dict[str, MarketData],
                                 historical_data: Dict[str, pd.DataFrame]) -> List[Signal]:
        """
        生成交易信号
        
        Args:
            market_data: 实时市场数据
            historical_data: 历史数据
            
        Returns:
            信号列表
        """
        signals = []
        
        for strategy_name, strategy_info in self.strategies.items():
            if not strategy_info['enabled']:
                continue
            
            try:
                # 执行策略
                strategy_func = strategy_info['func']
                params = strategy_info['params']
                
                # 生成信号
                strategy_signals = await self._execute_strategy(
                    strategy_func,
                    market_data,
                    historical_data,
                    params
                )
                
                signals.extend(strategy_signals)
                
            except Exception as e:
                self.logger.error(f"策略 {strategy_name} 执行失败: {e}")
        
        # 信号合并与过滤
        filtered_signals = self._filter_signals(signals)
        
        return filtered_signals
    
    async def _execute_strategy(self,
                                  strategy_func: Callable,
                                  market_data: Dict[str, MarketData],
                                  historical_data: Dict[str, pd.DataFrame],
                                  params: Dict) -> List[Signal]:
        """执行策略"""
        signals = []
        
        for symbol, data in market_data.items():
            try:
                # 调用策略函数
                signal_value = strategy_func(
                    historical_data.get(symbol, pd.DataFrame()),
                    **params
                )
                
                if signal_value != 0:
                    signal = Signal(
                        symbol=symbol,
                        timestamp=datetime.now(),
                        signal_type=SignalType(signal_value),
                        strength=abs(signal_value),
                        strategy=strategy_func.__name__,
                        metadata={'params': params}
                    )
                    signals.append(signal)
                    
            except Exception as e:
                self.logger.error(f"策略执行失败 {symbol}: {e}")
        
        return signals
    
    def _filter_signals(self, signals: List[Signal]) -> List[Signal]:
        """过滤信号"""
        # 按标的分组
        signals_by_symbol = {}
        for signal in signals:
            if signal.symbol not in signals_by_symbol:
                signals_by_symbol[signal.symbol] = []
            signals_by_symbol[signal.symbol].append(signal)
        
        # 对每个标的选择最强信号
        filtered_signals = []
        for symbol, symbol_signals in signals_by_symbol.items():
            # 选择信号强度最大的
            strongest_signal = max(symbol_signals, key=lambda s: s.strength)
            filtered_signals.append(strongest_signal)
        
        return filtered_signals


class ExecutionLayer:
    """执行层"""
    
    def __init__(self, config: Dict):
        """
        初始化执行层
        
        Args:
            config: 配置字典
        """
        self.config = config
        self.logger = logging.getLogger('ExecutionLayer')
        
        # 订单管理
        self.orders = {}
        self.order_counter = 0
        
        # 持仓管理
        self.positions = {}
        
        # 执行配置
        self.execution_config = config.get('execution_config', {
            'max_slippage': 0.001,  # 最大滑点0.1%
            'commission_rate': 0.0003,  # 佣金率0.03%
            'max_order_size': 100000,  # 单笔订单最大金额
            'order_timeout': 60  # 订单超时时间（秒）
        })
        
        # 交易接口（模拟）
        self.broker = None
        
    async def execute_signal(self,
                               signal: Signal,
                               current_positions: Dict[str, Position],
                               cash: float) -> Optional[Order]:
        """
        执行信号
        
        Args:
            signal: 交易信号
            current_positions: 当前持仓
            cash: 现金
            
        Returns:
            订单
        """
        # 计算目标仓位
        target_position = self._calculate_target_position(
            signal,
            current_positions,
            cash
        )
        
        # 计算交易量
        trade_quantity = self._calculate_trade_quantity(
            signal.symbol,
            target_position,
            current_positions.get(signal.symbol)
        )
        
        if trade_quantity == 0:
            return None
        
        # 创建订单
        order = self._create_order(signal, trade_quantity)
        
        # 执行订单
        executed_order = await self._execute_order(order)
        
        return executed_order
    
    def _calculate_target_position(self,
                                     signal: Signal,
                                     current_positions: Dict[str, Position],
                                     cash: float) -> float:
        """计算目标仓位"""
        portfolio_value = cash + sum(
            pos.quantity * pos.current_price 
            for pos in current_positions.values()
        )
        
        # 根据信号强度计算权重
        base_weight = signal.strength * self.config.get('max_position_weight', 0.3)
        
        # 根据风险管理调整权重
        risk_adjusted_weight = self._adjust_for_risk(base_weight, signal)
        
        # 目标仓位金额
        target_value = portfolio_value * risk_adjusted_weight
        
        return target_value
    
    def _adjust_for_risk(self, weight: float, signal: Signal) -> float:
        """风险调整权重"""
        # 根据信号强度调整
        if signal.strength < 0.5:
            weight *= 0.5
        elif signal.strength < 0.8:
            weight *= 0.8
        
        # 限制最大权重
        weight = min(weight, self.config.get('max_position_weight', 0.3))
        
        return weight
    
    def _calculate_trade_quantity(self,
                                    symbol: str,
                                    target_value: float,
                                    current_position: Optional[Position]) -> float:
        """计算交易量"""
        # 获取当前价格（简化）
        current_price = 3.0  # 实际应从实时数据获取
        
        # 目标数量
        target_quantity = target_value / current_price
        
        # 当前数量
        current_quantity = current_position.quantity if current_position else 0
        
        # 交易数量
        trade_quantity = target_quantity - current_quantity
        
        # 向下取整到100股
        trade_quantity = int(trade_quantity / 100) * 100
        
        return trade_quantity
    
    def _create_order(self, signal: Signal, quantity: float) -> Order:
        """创建订单"""
        self.order_counter += 1
        order_id = f"ORD{datetime.now().strftime('%Y%m%d%H%M%S')}{self.order_counter:04d}"
        
        order = Order(
            order_id=order_id,
            symbol=signal.symbol,
            direction='buy' if quantity > 0 else 'sell',
            quantity=abs(quantity),
            price=None,  # 市价单
            order_type='market',
            status=OrderStatus.PENDING,
            filled_quantity=0,
            filled_price=0,
            created_at=datetime.now(),
            updated_at=datetime.now(),
            slippage=0,
            commission=0
        )
        
        return order
    
    async def _execute_order(self, order: Order) -> Order:
        """执行订单"""
        # 模拟执行（实际应调用券商接口）
        try:
            # 获取市场价格
            market_price = await self._get_market_price(order.symbol)
            
            # 计算滑点
            slippage = self._calculate_slippage(order.quantity, market_price)
            
            # 实际成交价
            filled_price = market_price * (1 + slippage if order.direction == 'buy' else 1 - slippage)
            
            # 更新订单状态
            order.status = OrderStatus.FILLED
            order.filled_quantity = order.quantity
            order.filled_price = filled_price
            order.slippage = slippage
            order.commission = filled_price * order.quantity * self.execution_config['commission_rate']
            order.updated_at = datetime.now()
            
            # 记录订单
            self.orders[order.order_id] = order
            
            self.logger.info(
                f"订单执行成功: {order.order_id} {order.symbol} "
                f"{order.direction} {order.filled_quantity}@{order.filled_price:.4f} "
                f"滑点={order.slippage:.4%} 佣金={order.commission:.2f}"
            )
            
            return order
            
        except Exception as e:
            order.status = OrderStatus.FAILED
            order.updated_at = datetime.now()
            self.logger.error(f"订单执行失败: {order.order_id} {e}")
            return order
    
    async def _get_market_price(self, symbol: str) -> float:
        """获取市场价格"""
        # 模拟价格
        return 3.0
    
    def _calculate_slippage(self, quantity: float, price: float) -> float:
        """计算滑点"""
        # 基于交易量的滑点模型
        trade_value = quantity * price
        
        # 线性滑点模型
        base_slippage = self.execution_config['max_slippage']
        size_factor = trade_value / self.execution_config['max_order_size']
        size_factor = min(size_factor, 1.0)  # 限制最大为1
        
        slippage = base_slippage * size_factor
        
        return slippage


class MonitoringLayer:
    """监控层"""
    
    def __init__(self, config: Dict):
        """
        初始化监控层

        """

        # 风险阈值

            'var_limit_95': 0.05
        })
        
        # 告警配置
        self.alert_config = config.get('alert_config', {
            'email': 'trader@example.com',
            'wechat': '',
            'dingtalk': ''
        })
        
        # Redis存储
        self.redis_client = redis.Redis(
            host=config.get('redis_host', 'localhost'),
            port=config.get('redis_port', 6379),
            db=1
        )
        
    def calculate_risk_metrics(self,
                                 positions: Dict[str, Position],
                                 cash: float,
                                 historical_returns: pd.DataFrame) -> RiskMetrics:
        """
        计算风险指标
        
        Args:
            positions: 持仓
            cash: 现金
            historical_returns: 历史收益率
            
        Returns:
            风险指标
        """
        # 组合价值
        portfolio_value = cash + sum(
            pos.quantity * pos.current_price 
            for pos in positions.values()
        )
        
        # 杠杆
        leverage = portfolio_value / cash if cash > 0 else 0
        
        # 日收益
        daily_pnl = sum(pos.pnl for pos in positions.values())
        daily_pnl_pct = daily_pnl / portfolio_value if portfolio_value > 0 else 0
        
        # 回撤
        drawdown = self._calculate_drawdown(positions, portfolio_value)
        
        # VaR
        var_95, var_99 = self._calculate_var(historical_returns, portfolio_value)
        
        # 持仓集中度
        position_concentration = self._calculate_concentration(positions, portfolio_value)
        
        # 行业暴露（简化）
        sector_exposure = self._calculate_sector_exposure(positions, portfolio_value)
        
        risk_metrics = RiskMetrics(
            timestamp=datetime.now(),
            portfolio_value=portfolio_value,
            cash=cash,
            leverage=leverage,
            daily_pnl=daily_pnl,
            daily_pnl_pct=daily_pnl_pct,
            drawdown=drawdown,
            var_95=var_95,
            var_99=var_99,
            position_concentration=position_concentration,
            sector_exposure=sector_exposure
        )
        
        return risk_metrics
    
    def _calculate_drawdown(self,
                              positions: Dict[str, Position],
                              portfolio_value: float) -> float:
        """计算回撤"""
        # 从Redis获取历史净值
        nav_key = "portfolio:nav_history"
        nav_history = self.redis_client.lrange(nav_key, 0, -1)
        
        if len(nav_history) == 0:
            return 0
        
        # 解析历史净值
        nav_values = [float(nav) for nav in nav_history]
        nav_values.append(portfolio_value)
        
        # 计算回撤
        cumulative_max = np.maximum.accumulate(nav_values)
        drawdowns = (nav_values - cumulative_max) / cumulative_max
        
        return drawdowns[-1]
    
    def _calculate_var(self,
                         returns: pd.DataFrame,
                         portfolio_value: float) -> Tuple[float, float]:
        """计算VaR"""
        if returns.empty:
            return 0, 0
        
        # 计算组合收益率
        portfolio_returns = returns.mean(axis=1)
        
        # VaR (95%和99%)
        var_95 = portfolio_value * np.percentile(portfolio_returns, 5)
        var_99 = portfolio_value * np.percentile(portfolio_returns, 1)
        
        return abs(var_95), abs(var_99)
    
    def _calculate_concentration(self,
                                   positions: Dict[str, Position],
                                   portfolio_value: float) -> float:
        """计算持仓集中度"""
        if len(positions) == 0 or portfolio_value == 0:
            return 0
        
        # HHI指数
        weights = [
            (pos.quantity * pos.current_price) / portfolio_value
            for pos in positions.values()
        ]
        
        hhi = sum(w ** 2 for w in weights)
        
        return hhi
    
    def _calculate_sector_exposure(self,
                                     positions: Dict[str, Position],
                                     portfolio_value: float) -> Dict[str, float]:
        """计算行业暴露"""
        # 简化：假设所有ETF都是股票类
        exposure = {'equity': 0, 'bond': 0, 'commodity': 0}
        
        for symbol, pos in positions.items():
            weight = (pos.quantity * pos.current_price) / portfolio_value
            
            # 根据代码判断类型（简化）
            if symbol.startswith('51'):
                exposure['equity'] += weight
            elif symbol.startswith('511'):
                exposure['bond'] += weight
            else:
                exposure['commodity'] += weight
        
        return exposure
    
    def check_risk_breaches(self, risk_metrics: RiskMetrics) -> List[Dict]:
        """
        检查风险突破
        
        Args:
            risk_metrics: 风险指标
            
        Returns:

        """
        breaches = []
        
        # 检查回撤
        if risk_metrics.drawdown < -self.risk_thresholds['max_drawdown']:
            breaches.append({
                'type': 'drawdown',
                'level': RiskLevel.HIGH,
                'value': risk_metrics.drawdown,
                'threshold': self.risk_thresholds['max_drawdown'],
                'message': f"回撤 {risk_metrics.drawdown:.2%} 超过阈值 {self.risk_thresholds['max_drawdown']:.2%}"
            })
        
        # 检查日损失
        if risk_metrics.daily_pnl_pct < -self.risk_thresholds['max_daily_loss']:
            breaches.append({
                'type': 'daily_loss',
                'level': RiskLevel.HIGH,
                'value': risk_metrics.daily_pnl_pct,
                'threshold': self.risk_thresholds['max_daily_loss'],
                'message': f"日损失 {risk_metrics.daily_pnl_pct:.2%} 超过阈值 {self.risk_thresholds['max_daily_loss']:.2%}"
            })
        
        # 检查杠杆
        if risk_metrics.leverage > self.risk_thresholds['max_leverage']:
            breaches.append({
                'type': 'leverage',
                'level': RiskLevel.MEDIUM,
                'value': risk_metrics.leverage,
                'threshold': self.risk_thresholds['max_leverage'],
                'message': f"杠杆 {risk_metrics.leverage:.2f} 超过阈值 {self.risk_thresholds['max_leverage']:.2f}"
            })
        
        # 检查VaR
        var_pct = risk_metrics.var_95 / risk_metrics.portfolio_value
        if var_pct > self.risk_thresholds['var_limit_95']:
            breaches.append({
                'type': 'var',
                'level': RiskLevel.MEDIUM,
                'value': var_pct,
                'threshold': self.risk_thresholds['var_limit_95'],
                'message': f"VaR {var_pct:.2%} 超过阈值 {self.risk_thresholds['var_limit_95']:.2%}"
            })
        
        return breaches

        self.positions = {}

    async def run(self):
        """运行交易系统"""
        self.running = True
        self.logger.info("交易系统启动")
        
        while self.running:
            try:
                # 1. 获取数据
                market_data = await self._get_market_data()
                historical_data = await self._get_historical_data()
                
                # 2. 生成信号
                signals = await self.strategy_layer.generate_signals(
                    market_data,
                    historical_data
                )
                
                # 3. 执行交易
                for signal in signals:
                    order = await self.execution_layer.execute_signal(
                        signal,
                        self.positions,
                        self.cash
                    )
                    
                    if order and order.status == OrderStatus.FILLED:
                        # 更新持仓和现金
                        self._update_positions(order)
                
                # 4. 风险监控
                risk_metrics = self.monitoring_layer.calculate_risk_metrics(
                    self.positions,
                    self.cash,
                    historical_data
                )
                
                breaches = self.monitoring_layer.check_risk_breaches(risk_metrics)
                
                for breach in breaches:
                    self.monitoring_layer.send_alert(breach)
                
                # 5. 休眠
                await asyncio.sleep(60)  # 每分钟执行一次
                
            except Exception as e:
                self.logger.error(f"交易系统运行错误: {e}")
                await asyncio.sleep(5)
    
    async def _get_market_data(self) -> Dict[str, MarketData]:
        """获取市场数据"""
        market_data = {}
        
        for symbol in self.config.get('etf_pool', []):
            data = await self.data_layer.get_realtime_data(symbol)
            if data:
                market_data[symbol] = data
        
        return market_data
    
    async def _get_historical_data(self) -> Dict[str, pd.DataFrame]:
        """获取历史数据"""
        historical_data = {}
        
        end_date = datetime.now().strftime('%Y%m%d')
        start_date = (datetime.now() - timedelta(days=365)).strftime('%Y%m%d')
        
        for symbol in self.config.get('etf_pool', []):
            df = await self.data_layer.get_historical_data(
                symbol,
                start_date,
                end_date
            )
            if not df.empty:
                historical_data[symbol] = df
        
        return historical_data
    
    def _update_positions(self, order: Order):
        """更新持仓"""
        symbol = order.symbol
        
        if symbol not in self.positions:
            self.positions[symbol] = Position(
                symbol=symbol,
                quantity=0,
                avg_price=0,
                current_price=order.filled_price,
                pnl=0,
                pnl_pct=0,
                weight=0
            )
        
        pos = self.positions[symbol]
        
        if order.direction == 'buy':
            # 买入
            new_quantity = pos.quantity + order.filled_quantity
            new_avg_price = (
                (pos.quantity * pos.avg_price + order.filled_quantity * order.filled_price) 
                / new_quantity if new_quantity > 0 else 0
            )
            pos.quantity = new_quantity
            pos.avg_price = new_avg_price
            self.cash -= order.filled_quantity * order.filled_price + order.commission
        else:
            # 卖出
            pos.quantity -= order.filled_quantity
            realized_pnl = (order.filled_price - pos.avg_price) * order.filled_quantity
            pos.pnl += realized_pnl
            self.cash += order.filled_quantity * order.filled_price - order.commission
    
    def stop(self):
        """停止交易系统"""
        self.running = False
        self.logger.info("交易系统停止")
