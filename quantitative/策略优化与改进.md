```python
from itertools import product

def grid_search(data, strategy_class, param_grid, metric='sharpe'):
    """
    网格搜索
    
    参数:
    data: 数据
    strategy_class: 策略类
    param_grid: 参数网格
    metric: 优化指标
    
    返回:
    最优参数
    """
    best_params = None
    best_score = -np.inf
    
    # 参数组合
    param_combinations = list(product(*param_grid.values()))
    
    for params in param_combinations:
        param_dict = dict(zip(param_grid.keys(), params))
        
        # 回测
        strategy = strategy_class(**param_dict)
        result = backtest(data, strategy)
        
        # 评估
        score = result['performance'][metric]
        
        if score > best_score:
            best_score = score
            best_params = param_dict
    
    return best_params, best_score
```

### 1.2 贝叶斯优化

```python
from skopt import gp_minimize

def bayesian_optimization(data, strategy_class, param_space, n_iter=50):
    """
    贝叶斯优化
    
    参数:
    data: 数据
    strategy_class: 策略类
    param_space: 参数空间
    n_iter: 迭代次数
    
    返回:
    最优参数
    """
    def objective(params):
        param_dict = {
            'short_window': int(params[0]),
            'long_window': int(params[1]),
            'stop_loss': params[2]
        }
        
        strategy = strategy_class(**param_dict)
        result = backtest(data, strategy)
        
        return -result['performance']['sharpe']  # 最小化负夏普
    
    # 参数空间
    space = [
        (5, 30),   # short_window
        (30, 120), # long_window
        (0.02, 0.15)  # stop_loss
    ]
    
    result = gp_minimize(objective, space, n_calls=n_iter, random_state=42)
    
    best_params = {
        'short_window': int(result.x[0]),
        'long_window': int(result.x[1]),
        'stop_loss': result.x[2]
    }
    
    return best_params, -result.fun
```

### 2.1 样本外测试

```python
def out_of_sample_test(data, strategy_class, params, train_ratio=0.7):
    """
    样本外测试
    
    参数:
    data: 数据
    strategy_class: 策略类
    params: 参数
    train_ratio: 训练集比例
    
    返回:
    样本外结果
    """
    # 数据划分
    train_size = int(len(data) * train_ratio)
    train_data = data[:train_size]
    test_data = data[train_size:]
    
    # 训练
    strategy = strategy_class(**params)
    train_result = backtest(train_data, strategy)
    
    # 测试
    test_result = backtest(test_data, strategy)
    
    # 比较
    comparison = {
        'train_sharpe': train_result['performance']['sharpe'],
        'test_sharpe': test_result['performance']['sharpe'],
        'sharpe_degradation': (train_result['performance']['sharpe'] - test_result['performance']['sharpe']) / train_result['performance']['sharpe']
    }
    
    return {
        'train_result': train_result,
        'test_result': test_result,
        'comparison': comparison
    }
```

### 2.2 交叉验证

```python
def time_series_cv(data, strategy_class, params, n_splits=5):
    """
    时间序列交叉验证
    
    参数:
    data: 数据
    strategy_class: 策略类
    params: 参数
    n_splits: 分割数
    
    返回:
    交叉验证结果
    """
    from sklearn.model_selection import TimeSeriesSplit
    
    tscv = TimeSeriesSplit(n_splits=n_splits)
    results = []
    
    for train_idx, test_idx in tscv.split(data):
        train_data = data.iloc[train_idx]
        test_data = data.iloc[test_idx]
        
        strategy = strategy_class(**params)
        train_result = backtest(train_data, strategy)
        test_result = backtest(test_data, strategy)
        
        results.append({
            'train_sharpe': train_result['performance']['sharpe'],
            'test_sharpe': test_result['performance']['sharpe']
        })
    
    # 平均性能
    avg_train_sharpe = np.mean([r['train_sharpe'] for r in results])
    avg_test_sharpe = np.mean([r['test_sharpe'] for r in results])
    
    return {
        'results': results,
        'avg_train_sharpe': avg_train_sharpe,
        'avg_test_sharpe': avg_test_sharpe
    }
```

### 2.3 Walk-Forward优化

```python
def walk_forward_optimization(data, strategy_class, param_grid, 
                             train_size=252, test_size=63):
    """
    Walk-Forward优化
    
    参数:
    data: 数据
    strategy_class: 策略类
    param_grid: 参数网格
    train_size: 训练窗口
    test_size: 测试窗口
    
    返回:
    优化结果
    """
    results = []
    params_history = []
    
    for i in range(train_size, len(data) - test_size, test_size):
        # 训练数据
        train_data = data.iloc[i-train_size:i]
        
        # 测试数据
        test_data = data.iloc[i:i+test_size]
        
        # 参数优化
        best_params, _ = grid_search(train_data, strategy_class, param_grid)
        
        # 测试
        strategy = strategy_class(**best_params)
        test_result = backtest(test_data, strategy)
        
        results.append(test_result)
        params_history.append(best_params)
    
    return {
        'results': results,
        'params_history': params_history,
        'avg_sharpe': np.mean([r['performance']['sharpe'] for r in results])
    }
```

### 3.1 信号过滤

```python
def signal_filtering(signals, filter_rules):
    """
    信号过滤
    
    参数:
    signals: 原始信号
    filter_rules: 过滤规则
    
    返回:
    过滤后信号
    """
    filtered = signals.copy()
    
    for rule in filter_rules:
        if rule == 'volatility':
            # 波动率过滤
            volatility = signals.rolling(20).std()
            high_vol = volatility > volatility.quantile(0.9)
            filtered[high_vol] = 0
        
        elif rule == 'trend':
            # 趋势过滤
            ma = signals.rolling(60).mean()
            downtrend = signals < ma
            filtered[downtrend] = 0
        
        elif rule == 'volume':
            # 成交量过滤
            volume = signals.rolling(20).mean()
            low_vol = volume < volume.quantile(0.1)
            filtered[low_vol] = 0
    
    return filtered
```

### 3.2 风险管理

```python
class RiskManager:
    """风险管理器"""
    
    def __init__(self, max_position=1.0, max_drawdown=0.15, max_loss_per_trade=0.05):
        self.max_position = max_position
        self.max_drawdown = max_drawdown
        self.max_loss_per_trade = max_loss_per_trade
    
    def manage_position(self, current_position, signal, current_value, peak_value):
        """
        管理仓位
        
        参数:
        current_position: 当前仓位
        signal: 信号
        current_value: 当前价值
        peak_value: 峰值价值
        
        返回:
        调整后仓位
        """
        # 回撤控制
        drawdown = (peak_value - current_value) / peak_value
        
        if drawdown > self.max_drawdown:
            return 0  # 清仓
        
        # 仓位限制
        if current_position >= self.max_position and signal > 0:
            return current_position  # 不再加仓
        
        return signal * self.max_position
    
    def manage_loss(self, entry_price, current_price):
        """
        管理止损
        
        参数:
        entry_price: 入场价格
        current_price: 当前价格
        
        返回:
        是否止损
        """
        loss = (current_price - entry_price) / entry_price
        
        if loss < -self.max_loss_per_trade:
            return True  # 触发止损
        
        return False
```

### 3.3 多策略组合

```python
class StrategyCombiner:
    """策略组合器"""
    
    def __init__(self, strategies, weights=None):
        """
        参数:
        strategies: 策略列表
        weights: 权重列表
        """
        self.strategies = strategies
        self.weights = weights or [1/len(strategies)] * len(strategies)
    
    def generate_signals(self, data):
        """生成组合信号"""
        signals_list = []
        
        for strategy in self.strategies:
            signal = strategy.generate_signal(data)
            signals_list.append(signal)
        
        # 加权组合
        combined = sum(s * w for s, w in zip(signals_list, self.weights))
        
        return combined
```

## 4. 性能提升

### 4.1 向量化加速

```python
import numpy as np

def vectorized_backtest(prices, signals, commission=0.0003):
    """
    向量化回测
    
    参数:
    prices: 价格数组
    signals: 信号数组
    commission: 手续费
    
    返回:
    收益数组
    """
    # 转换为numpy数组
    prices = np.array(prices)
    signals = np.array(signals)
    
    # 收益
    returns = np.diff(prices) / prices[:-1]
    
    # 信号收益
    strategy_returns = signals[:-1] * returns
    
    # 交易成本
    trades = np.abs(np.diff(signals))
    costs = trades * commission
    
    # 扣除成本
    strategy_returns = strategy_returns - costs
    
    return strategy_returns
```

### 4.2 并行计算

```python
from multiprocessing import Pool

def parallel_backtest(data, strategies):
    """
    并行回测
    
    参数:
    data: 数据
    strategies: 策略列表
    
    返回:
    回测结果列表
    """
    def backtest_strategy(strategy):
        return backtest(data, strategy)
    
    with Pool() as pool:
        results = pool.map(backtest_strategy, strategies)
    
    return results
```
