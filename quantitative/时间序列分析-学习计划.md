# 项目1: ARIMA模型预测
import pandas as pd
import numpy as np
from statsmodels.tsa.arima.model import ARIMA
from statsmodels.tsa.stattools import adfuller, acf, pacf
from statsmodels.graphics.tsaplots import plot_acf, plot_pacf
import matplotlib.pyplot as plt

class ARIMAModel:
    def __init__(self, series):
        self.series = series
        self.model = None
        self.results = None
    
    def check_stationarity(self):
        """检验平稳性"""
        result = adfuller(self.series, autolag='AIC')
        print(f'ADF Statistic: {result[0]:.4f}')
        print(f'p-value: {result[1]:.4f}')
        print('Critical Values:')
        for key, value in result[4].items():
            print(f'\t{key}: {value:.4f}')
        
        return result[1] < 0.05  # p < 0.05, 平稳
    
    def plot_acf_pacf(self, lags=40):
        """绘制ACF和PACF图"""
        fig, axes = plt.subplots(2, 1, figsize=(12, 8))
        
        plot_acf(self.series, lags=lags, ax=axes[0])
        axes[0].set_title('Autocorrelation Function')
        
        plot_pacf(self.series, lags=lags, ax=axes[1])
        axes[1].set_title('Partial Autocorrelation Function')
        
        plt.tight_layout()
        plt.savefig('acf_pacf.png', dpi=150)
        plt.show()
    
    def fit(self, order):
        """拟合ARIMA模型"""
        self.model = ARIMA(self.series, order=order)
        self.results = self.model.fit()
        print(self.results.summary())
        
        return self.results
    
    def diagnose(self):
        """模型诊断"""
        # 残差分析
        residuals = self.results.resid
        
        fig, axes = plt.subplots(2, 2, figsize=(12, 10))
        
        # 残差时序图
        axes[0, 0].plot(residuals)
        axes[0, 0].set_title('Residuals Time Series')
        
        # 残差直方图
        axes[0, 1].hist(residuals, bins=30, density=True, alpha=0.7)
        axes[0, 1].set_title('Residuals Histogram')
        
        # 残差ACF
        plot_acf(residuals, lags=40, ax=axes[1, 0])
        
        # 残差PACF
        plot_pacf(residuals, lags=40, ax=axes[1, 1])
        
        plt.tight_layout()
        plt.savefig('diagnostics.png', dpi=150)
        plt.show()
        
        # Ljung-Box检验
        from statsmodels.stats.diagnostic import acorr_ljungbox
        lb_test = acorr_ljungbox(residuals, lags=[10, 20, 30])
        print('\nLjung-Box Test:')
        print(lb_test)
    
    def forecast(self, steps=10):
        """预测"""
        forecast = self.results.forecast(steps=steps)
        forecast_ci = self.results.get_forecast(steps=steps).conf_int()
        
        # 绘制预测图
        fig, ax = plt.subplots(figsize=(12, 6))
        
        # 历史数据
        ax.plot(self.series.index, self.series, label='Historical')
        
        # 预测
        forecast_index = pd.date_range(
            start=self.series.index[-1],
            periods=steps+1,
            freq='D'
        )[1:]
        ax.plot(forecast_index, forecast, label='Forecast', color='red')
        
        # 置信区间
        ax.fill_between(
            forecast_index,
            forecast_ci.iloc[:, 0],
            forecast_ci.iloc[:, 1],
            color='red',
            alpha=0.2
        )
        
        ax.set_title('ARIMA Forecast')
        ax.legend()
        plt.tight_layout()
        plt.savefig('forecast.png', dpi=150)
        plt.show()
        
        return forecast, forecast_ci

from arch import arch_model

class GARCHModel:
    def __init__(self, returns):
        self.returns = returns
        self.model = None
        self.results = None
    
    def fit(self, p=1, q=1, model_type='GARCH'):
        """拟合GARCH模型"""
        self.model = arch_model(
            self.returns,
            vol=model_type,
            p=p,
            q=q,
            dist='normal'
        )
        self.results = self.model.fit(disp='off')
        print(self.results.summary())
        
        return self.results
    
    def forecast_volatility(self, horizon=10):
        """预测波动率"""
        forecast = self.results.forecast(horizon=horizon)
        volatility = np.sqrt(forecast.variance.iloc[-1])
        
        return volatility
    
    def plot_volatility(self):
        """绘制条件波动率"""
        conditional_vol = self.results.conditional_volatility
        
        fig, axes = plt.subplots(2, 1, figsize=(12, 10))
        
        # 收益
