# 绩效评估指标

## 1. 收益指标

### 1.1 绝对收益

```python
def calculate_returns(portfolio_values):
    """
    计算收益指标
    
    参数:
    portfolio_values: 组合价值序列
    
    返回:
    收益指标
    """
    # 总收益
    total_return = (portfolio_values.iloc[-1] / portfolio_values.iloc[0] - 1) * 100
    
    # 年化收益
    years = len(portfolio_values) / 252
    annual_return = ((portfolio_values.iloc[-1] / portfolio_values.iloc[0]) ** (1/years) - 1) * 100
    
    # 累计收益
    returns = portfolio_values.pct_change()
    cumulative_return = (1 + returns).prod() - 1
    
    return {
        'total_return': total_return,
        'annual_return': annual_return,
        'cumulative_return': cumulative_return * 100
    }
```

### 1.2 风险调整收益

```python
def risk_adjusted_returns(returns, risk_free_rate=0.03):
    """
    风险调整收益
    
    参数:
    returns: 收益率序列
    risk_free_rate: 无风险利率
    
    返回:
    风险调整指标
    """
    # 夏普比率
    excess_returns = returns - risk_free_rate / 252
    sharpe = excess_returns.mean() / returns.std() * np.sqrt(252)
    
    # 索提诺比率
    downside_returns = returns[returns < 0]
    downside_vol = downside_returns.std() * np.sqrt(252)
    sortino = excess_returns.mean() * 252 / downside_vol if downside_vol > 0 else 0
    
    # 卡玛比率
    cumulative = (1 + returns).cumprod()
    drawdown = (cumulative - cumulative.cummax()) / cumulative.cummax()
    max_drawdown = abs(drawdown.min())
    calmar = returns.mean() * 252 / max_drawdown if max_drawdown > 0 else 0
    
    return {
        'sharpe': sharpe,
        'sortino': sortino,
        'calmar': calmar
    }
```

## 2. 风险指标

### 2.1 波动率

```python
def volatility_metrics(returns):
    """
    波动率指标
    
    参数:
    returns: 收益率序列
    
    返回:
    波动率指标
    """
    # 年化波动率
    annual_vol = returns.std() * np.sqrt(252)
    
    # 下行波动率
    downside_returns = returns[returns < 0]
    downside_vol = downside_returns.std() * np.sqrt(252)
    
    # 上行波动率
    upside_returns = returns[returns > 0]
    upside_vol = upside_returns.std() * np.sqrt(252)
    
    return {
        'annual_vol': annual_vol,
        'downside_vol': downside_vol,
        'upside_vol': upside_vol
    }
```

### 2.2 回撤

```python
def drawdown_metrics(returns):
    """
    回撤指标
    
    参数:
    returns: 收益率序列
    
    返回:
    回撤指标
    """
    # 累计收益
    cumulative = (1 + returns).cumprod()
    
    # 回撤
    drawdown = (cumulative - cumulative.cummax()) / cumulative.cummax()
    
    # 最大回撤
    max_drawdown = drawdown.min()
    
    # 平均回撤
    avg_drawdown = drawdown.mean()
    
    # 回撤持续期
    drawdown_periods = (drawdown < 0).astype(int)
    drawdown_durations = []
    current_duration = 0
    for val in drawdown_periods:
        if val == 1:
            current_duration += 1
        else:
            if current_duration > 0:
                drawdown_durations.append(current_duration)
            current_duration = 0
    if current_duration > 0:
        drawdown_durations.append(current_duration)
    
    max_drawdown_duration = max(drawdown_durations) if drawdown_durations else 0
    avg_drawdown_duration = np.mean(drawdown_durations) if drawdown_durations else 0
    
    return {
        'max_drawdown': max_drawdown,
        'avg_drawdown': avg_drawdown,
        'max_drawdown_duration': max_drawdown_duration,
        'avg_drawdown_duration': avg_drawdown_duration
    }
```

### 2.3 VaR

```python
def var_metrics(returns):
    """
    VaR指标
    
    参数:
    returns: 收益率序列
    
    返回:
    VaR指标
    """
    # VaR
    var_95 = returns.quantile(0.05)
    var_99 = returns.quantile(0.01)
    
    # CVaR
    cvar_95 = returns[returns < var_95].mean()
    cvar_99 = returns[returns < var_99].mean()
    
    return {
        'var_95': var_95,
        'var_99': var_99,
        'cvar_95': cvar_95,
        'cvar_99': cvar_99
    }
```

## 3. 交易指标

### 3.1 胜率与盈亏比

```python
def trade_metrics(trades):
    """
    交易指标
    
    参数:
    trades: 交易列表
    
    返回:
    交易指标
    """
    # 胜率
    win_trades = [t for t in trades if t['pnl'] > 0]
    loss_trades = [t for t in trades if t['pnl'] < 0]
    win_rate = len(win_trades) / len(trades) if trades else 0
    
    # 盈亏比
    avg_win = np.mean([t['pnl'] for t in win_trades]) if win_trades else 0
    avg_loss = abs(np.mean([t['pnl'] for t in loss_trades])) if loss_trades else 0
    profit_loss_ratio = avg_win / avg_loss if avg_loss > 0 else 0
    
    # 交易频率
    num_trades = len(trades)
    
    return {
        'win_rate': win_rate,
        'profit_loss_ratio': profit_loss_ratio,
        'num_trades': num_trades,
        'avg_win': avg_win,
        'avg_loss': avg_loss
    }
```

### 3.2 换手率

```python
def turnover_metrics(trades, portfolio_value):
    """
    换手率指标
    
    参数:
    trades: 交易列表
    portfolio_value: 组合价值
    
    返回:
    换手率指标
    """
    # 单边换手率
    buy_volume = sum(t['quantity'] * t['price'] for t in trades if t['quantity'] > 0)
    sell_volume = sum(abs(t['quantity'] * t['price']) for t in trades if t['quantity'] < 0)
    
    # 年化换手率
    total_volume = buy_volume + sell_volume
    turnover = total_volume / portfolio_value / 2  # 双边换手
    
    return {
        'turnover': turnover,
        'buy_volume': buy_volume,
        'sell_volume': sell_volume
    }
```

## 4. 综合评估

```python
class PerformanceEvaluation:
    """绩效综合评估"""
    
    def __init__(self, returns, trades, benchmark_returns=None):
        self.returns = returns
        self.trades = trades
        self.benchmark_returns = benchmark_returns
    
    def evaluate(self):
        """综合评估"""
        # 收益指标
        returns_metrics = calculate_returns((1 + self.returns).cumprod() * 100000)
        
        # 风险指标
        risk_adj = risk_adjusted_returns(self.returns)
        vol = volatility_metrics(self.returns)
        dd = drawdown_metrics(self.returns)
        var = var_metrics(self.returns)
        
        # 交易指标
        trade = trade_metrics(self.trades)
        
        # 基准比较
        if self.benchmark_returns is not None:
            benchmark = self.compare_to_benchmark(self.returns, self.benchmark_returns)
        else:
            benchmark = {}
        
        return {
            'returns': returns_metrics,
            'risk_adjusted': risk_adj,
            'volatility': vol,
            'drawdown': dd,
            'var': var,
            'trade': trade,
            'benchmark': benchmark
        }
    
    def compare_to_benchmark(self, returns, benchmark_returns):
        """与基准比较"""
        # 超额收益
        excess_returns = returns - benchmark_returns
        
        # 跟踪误差
        tracking_error = excess_returns.std() * np.sqrt(252)
        
        # 信息比率
        information_ratio = excess_returns.mean() / excess_returns.std() * np.sqrt(252) if excess_returns.std() > 0 else 0
        
        # Alpha和Beta
        from scipy.stats import linregress
        slope, intercept, _, _, _ = linregress(benchmark_returns, returns)
        beta = slope
        alpha = intercept * 252
        
        return {
            'excess_return': excess_returns.mean() * 252,
            'tracking_error': tracking_error,
            'information_ratio': information_ratio,
            'alpha': alpha,
            'beta': beta
        }
```

## 5. 可视化

```python
import matplotlib.pyplot as plt

def plot_performance(results):
    """绘制绩效图表"""
    fig, axes = plt.subplots(2, 2, figsize=(15, 10))
    
    # 累计收益
    axes[0, 0].plot(results['cumulative'], label='Strategy')
    if 'benchmark' in results:
        axes[0, 0].plot(results['benchmark_cumulative'], label='Benchmark')
    axes[0, 0].set_title('Cumulative Returns')
    axes[0, 0].legend()
    
    # 回撤
    axes[0, 1].fill_between(results['drawdown'].index, results['drawdown'], 0, alpha=0.3, color='red')
    axes[0, 1].set_title('Drawdown')
    
    # 月度收益
    monthly_returns = results['returns'].resample('M').sum()
    axes[1, 0].bar(monthly_returns.index, monthly_returns)
    axes[1, 0].set_title('Monthly Returns')
    
    # 年度收益
    annual_returns = results['returns'].resample('Y').sum()
    axes[1, 1].bar(annual_returns.index.year, annual_returns)
    axes[1, 1].set_title('Annual Returns')
    
    plt.tight_layout()
    return fig
```
