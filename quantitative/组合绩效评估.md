### 2.1 收益指标

```python
import numpy as np
import pandas as pd
from typing import Dict, List, Tuple

class PerformanceMetrics:
    """绩效评估指标"""
    
    def __init__(self, 
                 returns: pd.Series,
                 benchmark_returns: pd.Series = None,
                 risk_free_rate: float = 0.03):
        """
        初始化
        
        Args:
            returns: 组合收益率序列
            benchmark_returns: 基准收益率序列
            risk_free_rate: 无风险利率
        """
        self.returns = returns
        self.benchmark_returns = benchmark_returns
        self.risk_free_rate = risk_free_rate
        self.n_periods = len(returns)
    
    # ==================== 收益指标 ====================
    
    def total_return(self) -> float:
        """累计收益率"""
        return (1 + self.returns).prod() - 1
    
    def annual_return(self) -> float:
        """年化收益率"""
        n_years = self.n_periods / 252
        return (1 + self.total_return()) ** (1 / n_years) - 1
    
    def arithmetic_mean_return(self) -> float:
        """算术平均收益率"""
        return self.returns.mean() * 252
    
    def geometric_mean_return(self) -> float:
        """几何平均收益率"""
        return (1 + self.returns).prod() ** (252 / self.n_periods) - 1
    
    def monthly_returns(self) -> pd.Series:
        """月度收益率"""
        return self.returns.resample('M').apply(lambda x: (1 + x).prod() - 1)
    
    def yearly_returns(self) -> pd.Series:
        """年度收益率"""
        return self.returns.resample('Y').apply(lambda x: (1 + x).prod() - 1)
    
    def rolling_return(self, window: int = 252) -> pd.Series:
        """滚动收益率"""
        return (1 + self.returns).rolling(window).apply(lambda x: x.prod()) - 1
    
    # ==================== 风险指标 ====================
    
    def volatility(self) -> float:
        """年化波动率"""
        return self.returns.std() * np.sqrt(252)
    
    def downside_volatility(self, 
                           target_return: float = 0.0) -> float:
        """下行波动率"""
        downside_returns = self.returns[self.returns < target_return]
        return downside_returns.std() * np.sqrt(252)
    
    def max_drawdown(self) -> float:
        """最大回撤"""
        cumulative = (1 + self.returns).cumprod()
        running_max = cumulative.cummax()
        drawdown = (cumulative - running_max) / running_max
        return drawdown.min()
    
    def max_drawdown_duration(self) -> int:
        """最大回撤持续时间"""
        cumulative = (1 + self.returns).cumprod()
        running_max = cumulative.cummax()
        drawdown = (cumulative - running_max) / running_max
        
        # 找到回撤期
        in_drawdown = drawdown < 0
        drawdown_periods = []
        current_period = 0
        
        for dd in in_drawdown:
            if dd:
                current_period += 1
            else:
                if current_period > 0:
                    drawdown_periods.append(current_period)
                current_period = 0
        
        if current_period > 0:
            drawdown_periods.append(current_period)
        
        return max(drawdown_periods) if drawdown_periods else 0
    
    def var(self, 
           confidence_level: float = 0.95) -> float:
        """在险价值（VaR）"""
        return np.percentile(self.returns, (1 - confidence_level) * 100)
    
    def cvar(self, 
            confidence_level: float = 0.95) -> float:
        """条件在险价值（CVaR/ES）"""
        var = self.var(confidence_level)
        return self.returns[self.returns <= var].mean()
    
    def var_historical(self, 
                      confidence_level: float = 0.95) -> float:
        """历史VaR"""
        return self.var(confidence_level)
    
    def var_parametric(self, 
                      confidence_level: float = 0.95) -> float:
        """参数VaR（假设正态分布）"""
        from scipy.stats import norm
        z_score = norm.ppf(1 - confidence_level)
        return self.returns.mean() + z_score * self.returns.std()
    
    # ==================== 风险调整收益 ====================
    
    def sharpe_ratio(self) -> float:
        """夏普比率"""
        excess_return = self.annual_return() - self.risk_free_rate
        return excess_return / self.volatility()
    
    def sortino_ratio(self, 
                     target_return: float = None) -> float:
        """索提诺比率"""
        if target_return is None:
            target_return = self.risk_free_rate / 252
        
        excess_return = self.annual_return() - self.risk_free_rate
        downside_vol = self.downside_volatility(target_return / 252)
        
        return excess_return / downside_vol if downside_vol > 0 else 0
    
    def calmar_ratio(self) -> float:
        """卡尔玛比率"""
        return self.annual_return() / abs(self.max_drawdown())
    
    def information_ratio(self) -> float:
        """信息比率"""
        if self.benchmark_returns is None:
            return np.nan
        
        excess_returns = self.returns - self.benchmark_returns
        tracking_error = excess_returns.std() * np.sqrt(252)
        active_return = excess_returns.mean() * 252
        
        return active_return / tracking_error if tracking_error > 0 else 0
    
    def treynor_ratio(self, 
                     beta: float) -> float:
        """特雷诺比率"""
        excess_return = self.annual_return() - self.risk_free_rate
        return excess_return / beta if beta > 0 else 0
    
    def omega_ratio(self, 
                   target_return: float = 0.0) -> float:
        """Omega比率"""
        gains = self.returns[self.returns > target_return] - target_return
        losses = target_return - self.returns[self.returns < target_return]
        
        return gains.sum() / losses.sum() if losses.sum() > 0 else np.inf
    
    # ==================== 尾部风险指标 ====================
    
    def tail_ratio(self) -> float:
        """尾部比率"""
        return np.percentile(self.returns, 95) / abs(np.percentile(self.returns, 5))
    
    def skewness(self) -> float:
        """偏度"""
        from scipy.stats import skew
        return skew(self.returns)
    
    def kurtosis(self) -> float:
        """峰度"""
        from scipy.stats import kurtosis
        return kurtosis(self.returns)
    
    def jarque_bera_test(self) -> Dict:
        """Jarque-Bera正态性检验"""
        from scipy.stats import jarque_bera
        jb_stat, p_value = jarque_bera(self.returns)
        
        return {
            'jb_statistic': jb_stat,
            'p_value': p_value,
            'is_normal': p_value > 0.05
        }
    
    # ==================== 综合绩效报告 ====================
    
    def performance_report(self) -> pd.DataFrame:
        """
        综合绩效报告
        
        Returns:
            绩效指标表
        """
        metrics = {
            # 收益指标
            '累计收益率': self.total_return(),
            '年化收益率': self.annual_return(),
            '算术平均收益率': self.arithmetic_mean_return(),
            '几何平均收益率': self.geometric_mean_return(),
            
            # 风险指标
            '年化波动率': self.volatility(),
            '下行波动率': self.downside_volatility(),
            '最大回撤': self.max_drawdown(),
            '最大回撤持续期': self.max_drawdown_duration(),
            'VaR(95%)': self.var(0.95),
            'CVaR(95%)': self.cvar(0.95),
            
            # 风险调整收益
            '夏普比率': self.sharpe_ratio(),
            '索提诺比率': self.sortino_ratio(),
            '卡尔玛比率': self.calmar_ratio(),
            '信息比率': self.information_ratio(),
            'Omega比率': self.omega_ratio(),
            
            # 尾部风险
            '尾部比率': self.tail_ratio(),
            '偏度': self.skewness(),
            '峰度': self.kurtosis()
        }
        
        return pd.DataFrame(metrics, index=['Value']).T
```

```python
class BenchmarkComparison:
    """基准对比"""
    
    def __init__(self, 
                 portfolio_returns: pd.Series,
                 benchmark_returns: pd.Series,
                 risk_free_rate: float = 0.03):
        """
        初始化
        
        Args:
            portfolio_returns: 组合收益率
            benchmark_returns: 基准收益率
            risk_free_rate: 无风险利率
        """
        self.portfolio_returns = portfolio_returns
        self.benchmark_returns = benchmark_returns
        self.risk_free_rate = risk_free_rate
        
        # 对齐数据
        common_index = portfolio_returns.index.intersection(benchmark_returns.index)
        self.portfolio_returns = portfolio_returns.loc[common_index]
        self.benchmark_returns = benchmark_returns.loc[common_index]
    
    def excess_return(self) -> float:
        """超额收益"""
        portfolio_annual = self.portfolio_returns.mean() * 252
        benchmark_annual = self.benchmark_returns.mean() * 252
        return portfolio_annual - benchmark_annual
    
    def tracking_error(self) -> float:
        """跟踪误差"""
        excess_returns = self.portfolio_returns - self.benchmark_returns
        return excess_returns.std() * np.sqrt(252)
    
    def beta(self) -> float:
        """Beta系数"""
        covariance = np.cov(self.portfolio_returns, self.benchmark_returns)[0, 1]
        variance = np.var(self.benchmark_returns)
        return covariance / variance if variance > 0 else 0
    
    def alpha(self) -> float:
        """Alpha"""
        portfolio_annual = self.portfolio_returns.mean() * 252
        benchmark_annual = self.benchmark_returns.mean() * 252
        beta = self.beta()
        
        return portfolio_annual - self.risk_free_rate - beta * (benchmark_annual - self.risk_free_rate)
    
    def up_capture(self) -> float:
        """上行捕捉比率"""
        up_market = self.benchmark_returns > 0
        if up_market.sum() == 0:
            return 0
        
        portfolio_up = self.portfolio_returns[up_market].mean()
        benchmark_up = self.benchmark_returns[up_market].mean()
        
        return portfolio_up / benchmark_up if benchmark_up > 0 else 0
    
    def down_capture(self) -> float:
        """下行捕捉比率"""
        down_market = self.benchmark_returns < 0
        if down_market.sum() == 0:
            return 0
        
        portfolio_down = self.portfolio_returns[down_market].mean()
        benchmark_down = self.benchmark_returns[down_market].mean()
        
        return portfolio_down / benchmark_down if benchmark_down < 0 else 0
    
    def comparison_report(self) -> pd.DataFrame:
        """
        对比报告
        
        Returns:
            对比结果表
        """
        portfolio_metrics = PerformanceMetrics(
            self.portfolio_returns, 
            self.benchmark_returns,
            self.risk_free_rate
        )
        benchmark_metrics = PerformanceMetrics(
            self.benchmark_returns,
            None,
            self.risk_free_rate
        )
        
        comparison = {
            '组合': [
                portfolio_metrics.annual_return(),
                portfolio_metrics.volatility(),
                portfolio_metrics.sharpe_ratio(),
                portfolio_metrics.max_drawdown(),
                portfolio_metrics.sortino_ratio(),
                portfolio_metrics.calmar_ratio(),
                '-',
                '-',
                '-',
                '-'
            ],
            '基准': [
                benchmark_metrics.annual_return(),
                benchmark_metrics.volatility(),
                benchmark_metrics.sharpe_ratio(),
                benchmark_metrics.max_drawdown(),
                benchmark_metrics.sortino_ratio(),
                benchmark_metrics.calmar_ratio(),
                '-',
                '-',
                '-',
                '-'
            ],
            '差异': [
                self.excess_return(),
                portfolio_metrics.volatility() - benchmark_metrics.volatility(),
                portfolio_metrics.sharpe_ratio() - benchmark_metrics.sharpe_ratio(),
                portfolio_metrics.max_drawdown() - benchmark_metrics.max_drawdown(),
                portfolio_metrics.sortino_ratio() - benchmark_metrics.sortino_ratio(),
                portfolio_metrics.calmar_ratio() - benchmark_metrics.calmar_ratio(),
                self.alpha(),
                self.beta(),
                self.up_capture(),
                self.down_capture()
            ]
        }
        
        index = [
            '年化收益率', '年化波动率', '夏普比率', '最大回撤',
            '索提诺比率', '卡尔玛比率', 'Alpha', 'Beta',
            '上行捕捉', '下行捕捉'
        ]
        
        return pd.DataFrame(comparison, index=index)
```

## 3. 绩效归因分析

### 3.1 Brinson归因模型

**Brinson归因模型**将组合收益分解为：
- 配置效应（Allocation Effect）
- 选择效应（Selection Effect）
- 交互效应（Interaction Effect）

```python
class BrinsonAttribution:
    """Brinson归因分析"""
    
    def __init__(self, 
                 portfolio_weights: pd.DataFrame,
                 portfolio_returns: pd.DataFrame,
                 benchmark_weights: pd.DataFrame,
                 benchmark_returns: pd.DataFrame):
        """
        初始化
        
        Args:
            portfolio_weights: 组合权重 (时间, 资产)
            portfolio_returns: 组合收益率 (时间, 资产)
            benchmark_weights: 基准权重 (时间, 资产)
            benchmark_returns: 基准收益率 (时间, 资产)
        """
        self.portfolio_weights = portfolio_weights
        self.portfolio_returns = portfolio_returns
        self.benchmark_weights = benchmark_weights
        self.benchmark_returns = benchmark_returns
        
        self.n_periods = len(portfolio_weights)
        self.n_assets = portfolio_weights.shape[1]
        self.asset_names = portfolio_weights.columns.tolist()
    
    def allocation_effect(self) -> pd.Series:
        """
        配置效应
        
        Returns:
            各资产配置效应
        """
        # 配置效应 = Σ(w_p - w_b) × R_b
        allocation = (
            (self.portfolio_weights - self.benchmark_weights) * 
            self.benchmark_returns
        ).sum()
        
        return allocation
    
    def selection_effect(self) -> pd.Series:
        """
        选择效应
        
        Returns:
            各资产选择效应
        """
        # 选择效应 = Σw_b × (R_p - R_b)
        selection = (
            self.benchmark_weights * 
            (self.portfolio_returns - self.benchmark_returns)
        ).sum()
        
        return selection
    
    def interaction_effect(self) -> pd.Series:
        """
        交互效应
        
        Returns:
            各资产交互效应
        """
        # 交互效应 = Σ(w_p - w_b) × (R_p - R_b)
        interaction = (
            (self.portfolio_weights - self.benchmark_weights) * 
            (self.portfolio_returns - self.benchmark_returns)
        ).sum()
        
        return interaction
    
    def total_active_return(self) -> float:
        """
        总主动收益
        
        Returns:
            总主动收益
        """
        portfolio_return = (self.portfolio_weights * self.portfolio_returns).sum().sum()
        benchmark_return = (self.benchmark_weights * self.benchmark_returns).sum().sum()
        
        return portfolio_return - benchmark_return
    
    def attribution_report(self) -> pd.DataFrame:
        """
        归因报告
        
        Returns:
            归因分析表
        """
        allocation = self.allocation_effect()
        selection = self.selection_effect()
        interaction = self.interaction_effect()
        
        report = pd.DataFrame({
            'Allocation Effect': allocation,
            'Selection Effect': selection,
            'Interaction Effect': interaction,
            'Total Effect': allocation + selection + interaction
        })
        
        # 添加合计行
        report.loc['Total'] = report.sum()
        
        return report
```

### 3.2 因子归因

**因子归因**将组合收益分解为因子收益

```python
class FactorAttribution:
    """因子归因分析"""
    
    def __init__(self, 
                 portfolio_returns: pd.Series,
                 factor_returns: pd.DataFrame,
                 factor_exposures: pd.Series):
        """
        初始化
        
        Args:
            portfolio_returns: 组合收益率
            factor_returns: 因子收益率 (时间, 因子)
            factor_exposures: 因子暴露 (因子)
        """
        self.portfolio_returns = portfolio_returns
        self.factor_returns = factor_returns
        self.factor_exposures = factor_exposures
        
        self.factor_names = factor_returns.columns.tolist()
        self.n_factors = len(self.factor_names)
    
    def factor_attribution(self) -> pd.DataFrame:
        """
        因子归因
        
        Returns:
            因子归因结果
        """
        # 因子收益贡献
        factor_contrib = self.factor_exposures * self.factor_returns.mean() * 252
        
        # 总收益
        total_return = self.portfolio_returns.mean() * 252
        
        # 特质收益
        factor_return = factor_contrib.sum()
        idiosyncratic_return = total_return - factor_return
        
        attribution = pd.DataFrame({
            'Factor': self.factor_names + ['Idiosyncratic', 'Total'],
            'Exposure': list(self.factor_exposures) + ['-', '-'],
            'Factor Return': list(self.factor_returns.mean() * 252) + ['-', '-'],
            'Contribution': list(factor_contrib) + [idiosyncratic_return, total_return]
        })
        
        attribution['Contribution %'] = attribution['Contribution'] / total_return
        
        return attribution
    
    def rolling_factor_attribution(self, 
                                  window: int = 60) -> pd.DataFrame:
        """
        滚动因子归因
        
        Args:
            window: 滚动窗口
            
        Returns:
            滚动因子归因结果
        """
        results = []
        
        for i in range(window, len(self.portfolio_returns)):
            portfolio_window = self.portfolio_returns.iloc[i-window:i]
            factor_window = self.factor_returns.iloc[i-window:i]
            
            # 因子收益贡献
            factor_contrib = self.factor_exposures * factor_window.mean() * 252
            
            # 总收益
            total_return = portfolio_window.mean() * 252
            
            # 特质收益
            idiosyncratic_return = total_return - factor_contrib.sum()
            
            result = {
                'Date': self.portfolio_returns.index[i],
                'Total Return': total_return,
                'Idiosyncratic Return': idiosyncratic_return
            }
            
            for j, factor in enumerate(self.factor_names):
                result[f'{factor} Contribution'] = factor_contrib.iloc[j]
            
            results.append(result)
        
        return pd.DataFrame(results).set_index('Date')
```

### 3.3 策略归因

**策略归因**将组合收益分解为各策略贡献

```python
class StrategyAttribution:
    """策略归因分析"""
    
    def __init__(self, 
                 strategy_returns: pd.DataFrame,
                 strategy_weights: pd.DataFrame):
        """
        初始化
        
        Args:
            strategy_returns: 策略收益率 (时间, 策略)
            strategy_weights: 策略权重 (时间, 策略)
        """
        self.strategy_returns = strategy_returns
        self.strategy_weights = strategy_weights
        
        self.strategy_names = strategy_returns.columns.tolist()
        self.n_strategies = len(self.strategy_names)
    
    def strategy_attribution(self) -> pd.DataFrame:
        """
        策略归因
        
        Returns:
            策略归因结果
        """
        # 策略收益贡献
        strategy_contrib = (self.strategy_weights * self.strategy_returns).sum()
        
        # 总收益
        total_return = strategy_contrib.sum()
        
        attribution = pd.DataFrame({
            'Strategy': self.strategy_names + ['Total'],
            'Weight': list(self.strategy_weights.mean()) + ['-'],
            'Return': list(self.strategy_returns.mean() * 252) + ['-'],
            'Contribution': list(strategy_contrib * 252) + [total_return * 252]
        })
        
        attribution['Contribution %'] = attribution['Contribution'] / (total_return * 252)
        
        return attribution
    
    def strategy_risk_attribution(self) -> pd.DataFrame:
        """
        策略风险归因
        
        Returns:
            策略风险归因结果
        """
        # 计算策略协方差
        cov_matrix = self.strategy_returns.cov() * 252
        
        # 平均权重
        avg_weights = self.strategy_weights.mean().values
        
        # 组合波动率
        portfolio_vol = np.sqrt(np.dot(avg_weights.T, np.dot(cov_matrix, avg_weights)))
        
        # 风险贡献
        marginal_risk = np.dot(cov_matrix, avg_weights) / portfolio_vol
        risk_contrib = avg_weights * marginal_risk
        risk_contrib_pct = risk_contrib / portfolio_vol
        
        attribution = pd.DataFrame({
            'Strategy': self.strategy_names + ['Total'],
            'Weight': list(avg_weights) + ['-'],
            'Marginal Risk': list(marginal_risk) + ['-'],
            'Risk Contribution': list(risk_contrib) + [portfolio_vol],
            'Risk Contribution %': list(risk_contrib_pct) + [1.0]
        })
        
        return attribution
```

```python
class MultiStrategyEvaluation:
    """多策略组合绩效评估"""
    
    def __init__(self, 
                 strategy_returns: pd.DataFrame,
                 portfolio_weights: np.ndarray = None):
        """
        初始化
        
        Args:
            strategy_returns: 策略收益率
            portfolio_weights: 组合权重
        """
        self.strategy_returns = strategy_returns
        self.strategy_names = strategy_returns.columns.tolist()
        self.n_strategies = len(self.strategy_names)
        
        if portfolio_weights is None:
            self.portfolio_weights = np.ones(self.n_strategies) / self.n_strategies
        else:
            self.portfolio_weights = portfolio_weights
    
    def evaluate_strategies(self) -> pd.DataFrame:
        """
        评估各策略
        
        Returns:
            策略评估结果
        """
        results = []
        
        for strategy in self.strategy_names:
            metrics = PerformanceMetrics(self.strategy_returns[strategy])
            
            results.append({
                'Strategy': strategy,
                'Annual Return': metrics.annual_return(),
                'Volatility': metrics.volatility(),
                'Sharpe Ratio': metrics.sharpe_ratio(),
                'Max Drawdown': metrics.max_drawdown(),
                'Sortino Ratio': metrics.sortino_ratio(),
                'Calmar Ratio': metrics.calmar_ratio()
            })
        
        # 组合表现
        portfolio_returns = (self.strategy_returns * self.portfolio_weights).sum(axis=1)
        portfolio_metrics = PerformanceMetrics(portfolio_returns)
        
        results.append({
            'Strategy': 'Portfolio',
            'Annual Return': portfolio_metrics.annual_return(),
            'Volatility': portfolio_metrics.volatility(),
            'Sharpe Ratio': portfolio_metrics.sharpe_ratio(),
            'Max Drawdown': portfolio_metrics.max_drawdown(),
            'Sortino Ratio': portfolio_metrics.sortino_ratio(),
            'Calmar Ratio': portfolio_metrics.calmar_ratio()
        })
        
        return pd.DataFrame(results)
    
    def correlation_analysis(self) -> pd.DataFrame:
        """
        策略相关性分析
        
        Returns:
            相关性矩阵
        """
        return self.strategy_returns.corr()
    
    def diversification_analysis(self) -> Dict:
        """
        分散化分析
        
        Returns:
            分散化指标
        """
        # 平均相关性
        corr_matrix = self.correlation_analysis()
        avg_correlation = (corr_matrix.values.sum() - self.n_strategies) / (
            self.n_strategies * (self.n_strategies - 1)
        )
        
        # 分散化比率
        portfolio_returns = (self.strategy_returns * self.portfolio_weights).sum(axis=1)
        portfolio_vol = portfolio_returns.std() * np.sqrt(252)
        
        weighted_vol = np.sum(
            self.portfolio_weights * self.strategy_returns.std() * np.sqrt(252)
        )
        
        diversification_ratio = weighted_vol / portfolio_vol
        
        return {
            'average_correlation': avg_correlation,
            'diversification_ratio': diversification_ratio,
            'portfolio_volatility': portfolio_vol,
            'weighted_volatility': weighted_vol
        }
```

```python
class RobustnessTest:
    """稳健性检验"""
    
    def __init__(self, 
                 returns: pd.Series,
                 benchmark_returns: pd.Series = None):
        """
        初始化
        
        Args:
            returns: 收益率序列
            benchmark_returns: 基准收益率
        """
        self.returns = returns
        self.benchmark_returns = benchmark_returns
    
    def sub_period_analysis(self, 
                           n_periods: int = 4) -> pd.DataFrame:
        """
        子期间分析
        
        Args:
            n_periods: 子期间数量
            
        Returns:
            子期间表现
        """
        period_length = len(self.returns) // n_periods
        
        results = []
        
        for i in range(n_periods):
            start = i * period_length
            end = (i + 1) * period_length if i < n_periods - 1 else len(self.returns)
            
            period_returns = self.returns.iloc[start:end]
            metrics = PerformanceMetrics(period_returns)
            
            results.append({
                'Period': f'Period {i+1}',
                'Start Date': period_returns.index[0],
                'End Date': period_returns.index[-1],
                'Annual Return': metrics.annual_return(),
                'Volatility': metrics.volatility(),
                'Sharpe Ratio': metrics.sharpe_ratio(),
                'Max Drawdown': metrics.max_drawdown()
            })
        
        return pd.DataFrame(results)
    
    def bootstrap_test(self, 
                      n_bootstrap: int = 1000,
                      metric: str = 'sharpe') -> Dict:
        """
        Bootstrap检验
        
        Args:
            n_bootstrap: Bootstrap次数
            metric: 检验指标
            
        Returns:
            Bootstrap检验结果
        """
        bootstrap_metrics = []
        
        for _ in range(n_bootstrap):
            # 重采样
            sample_idx = np.random.choice(len(self.returns), len(self.returns), replace=True)
            sample_returns = self.returns.iloc[sample_idx]
            
            metrics = PerformanceMetrics(sample_returns)
            
            if metric == 'sharpe':
                bootstrap_metrics.append(metrics.sharpe_ratio())
            elif metric == 'return':
                bootstrap_metrics.append(metrics.annual_return())
            elif metric == 'volatility':
                bootstrap_metrics.append(metrics.volatility())
        
        # 统计推断
        original_metrics = PerformanceMetrics(self.returns)
        
        if metric == 'sharpe':
            original_value = original_metrics.sharpe_ratio()
        elif metric == 'return':
            original_value = original_metrics.annual_return()
        elif metric == 'volatility':
            original_value = original_metrics.volatility()
        
        return {
            'original_value': original_value,
            'bootstrap_mean': np.mean(bootstrap_metrics),
            'bootstrap_std': np.std(bootstrap_metrics),
            'bootstrap_median': np.median(bootstrap_metrics),
            'confidence_interval_95': (
                np.percentile(bootstrap_metrics, 2.5),
                np.percentile(bootstrap_metrics, 97.5)
            ),
            'p_value': np.mean(np.array(bootstrap_metrics) <= 0) if metric in ['sharpe', 'return'] else None
        }
    
    def sensitivity_analysis(self, 
                            parameter: str = 'window',
                            values: List = None) -> pd.DataFrame:
        """
        敏感性分析
        
        Args:
            parameter: 分析参数
            values: 参数值列表
            
        Returns:
            敏感性分析结果
        """
        if values is None:
            values = [60, 120, 180, 252, 504]  # 默认窗口值
        
        results = []
        
        for value in values:
            if parameter == 'window':
                rolling_returns = self.returns.rolling(value).apply(lambda x: (1 + x).prod() - 1)
                metrics = PerformanceMetrics(rolling_returns.dropna())
                
                results.append({
                    'Window': value,
                    'Mean Rolling Return': rolling_returns.mean(),
                    'Std Rolling Return': rolling_returns.std(),
                    'Min Rolling Return': rolling_returns.min(),
                    'Max Rolling Return': rolling_returns.max()
                })
        
        return pd.DataFrame(results)
```

## 5. 核心结论

### 5.1 绩效评估指标体系

| 指标类型 | 关键指标 | 计算方法 | 应用场景 |
|---------|---------|---------|---------|
| 收益指标 | 年化收益率 | 几何平均年化 | 业绩评价 |
| 风险指标 | 最大回撤 | 峰值到谷值 | 风险控制 |
| 风险调整收益 | 夏普比率 | 超额收益/波动率 | 风险效率 |
| 尾部风险 | CVaR | 条件在险价值 | 极端风险 |

### 5.2 归因分析方法

| 归因类型 | 分解维度 | 应用场景 |
|---------|---------|---------|
| Brinson归因 | 配置、选择、交互 | 相对基准分析 |
| 因子归因 | 因子暴露、因子收益 | 因子投资分析 |
| 策略归因 | 策略贡献、策略风险 | 多策略组合 |

### 5.3 绩效评估实践

1. **全面评估**：收益、风险、风险调整收益综合考量
2. **基准对比**：与合适基准对比，评估相对表现
3. **归因分析**：识别收益来源，指导策略改进
4. **稳健性检验**：验证结果的稳定性和可靠性
