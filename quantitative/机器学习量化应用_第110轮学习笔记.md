```python
def handle_missing_values(features):
    """处理缺失值"""
    # 策略1：向前填充（时间序列）
    features = features.fillna(method='ffill')
    
    # 策略2：行业均值填充
    for col in features.columns:
        features[col] = features.groupby('industry')[col].transform(
            lambda x: x.fillna(x.mean())
        )
    
    # 策略3：极端值处理
    features = features.clip(lower=features.quantile(0.01), 
                            upper=features.quantile(0.99), 
                            axis=1)
    
    return features
```

## 二、模型选择

### 2.1 模型分类与适用场景

#### 传统机器学习模型

**随机森林（Random Forest）**
- **优势**：对异常值鲁棒、不易过拟合、特征重要性
- **劣势**：计算量大、难以捕捉线性关系
- **适用场景**：因子选股、风险评分

**XGBoost/LightGBM**
- **优势**：高效、准确、支持GPU加速
- **劣势**：需调参、可能过拟合
- **适用场景**：多因子选股、量化择时
- **特点**：适合结构化数据，特征重要性解释性强

**支持向量机（SVM）**
- **优势**：适合小样本、高维数据
- **劣势**：计算量大、参数敏感
- **适用场景**：趋势分类、异常检测

#### 深度学习模型

**LSTM（长短期记忆网络）**
- **核心原理**：记忆单元、门控机制
- **优势**：捕捉长序列依赖、解决梯度消失
- **适用场景**：股价预测、时间序列

**Transformer**
- **核心创新**：自注意力机制、并行计算
- **优势**：处理长序列、可解释性
- **适用场景**：多变量时间序列、股票关系建模

**Transformer变体（2026趋势）**
- **Temporal Fusion Transformer (TFT)**：多变量时间序列预测
- **Informer**：长序列预测，O(nlogn)复杂度
- **Autoformer**：分解时序趋势和季节性
- **PatchTST**：将时间序列分patch处理
- **TradeFM**：生成式Transformer，直接学习交易流微观结构

#### 深度学习混合模型（2026热点）

**LSTM+Transformer混合模型**

  - LSTM层：提取局部时序特征

## 三、过拟合防范

### 3.1 过拟合识别

#### 表现特征
- 回测表现完美，实盘亏损
- 参数高度敏感
- 样本内外表现差异巨大
- 训练损失持续下降，验证损失上升

#### 检测方法

**Walk-Forward分析**
```python
def walk_forward_analysis(data, model, n_splits=5):
    """滚动窗口验证"""
    results = []
    window_size = len(data) // (n_splits + 1)
    
    for i in range(n_splits):
        train_start = i * window_size
        train_end = train_start + window_size
        test_start = train_end
        test_end = test_start + window_size
        
        train_data = data[train_start:train_end]
        test_data = data[test_start:test_end]
        
        # 训练
        model.fit(train_data)
        
        # 测试
        train_score = model.evaluate(train_data)
        test_score = model.evaluate(test_data)
        
        results.append({
            'train_score': train_score,
            'test_score': test_score,
            'overfit_ratio': train_score / test_score
        })
    
    return results
```

```python
class TransactionCostModel:
    def __init__(self, commission=0.0003, slippage=0.0005):
        self.commission = commission  # 手续费率
        self.slippage = slippage      # 滑点率
    
    def calculate_cost(self, price, quantity):
        # 手续费
        commission_cost = price * quantity * self.commission
        # 滑点（假设价格不利变动）
        slippage_cost = price * quantity * self.slippage
        return commission_cost + slippage_cost
```

```python
class RollingWindowModel:
    """滚动窗口在线学习"""
    
    def __init__(self, model, window_size=252):
        self.model = model
        self.window_size = window_size
        self.data_buffer = []
    
    def update(self, new_data):
        """更新模型"""
        self.data_buffer.append(new_data)
        
        # 保持窗口大小
        if len(self.data_buffer) > self.window_size:
            self.data_buffer.pop(0)
        
        # 重新训练
        if len(self.data_buffer) == self.window_size:
            self.model.fit(self.data_buffer)
```

```python
class ExperienceReplay:
    """经验回放机制"""
    
    def __init__(self, capacity=10000):
        self.capacity = capacity
        self.buffer = []
    
    def push(self, experience):
        """添加经验"""
        if len(self.buffer) >= self.capacity:
            self.buffer.pop(0)
        self.buffer.append(experience)
    
    def sample(self, batch_size):
        """采样经验"""
        return random.sample(self.buffer, batch_size)
```

#### 知识蒸馏
```python
class KnowledgeDistillation:
    """知识蒸馏"""
    
    def __init__(self, teacher_model, student_model, temperature=3.0):
        self.teacher = teacher_model
        self.student = student_model
        self.temperature = temperature
    
    def distillation_loss(self, outputs, teacher_outputs):
        """蒸馏损失"""
        from torch.nn.functional import kl_div, log_softmax, softmax
        
        student_log_probs = log_softmax(outputs / self.temperature, dim=1)
        teacher_probs = softmax(teacher_outputs / self.temperature, dim=1)
        
        return kl_div(student_log_probs, teacher_probs, reduction='batchmean')
```

```python
    """混合模型"""
    
    def __init__(self, models):
        self.models = models
    
    def fit(self, X, y):
        for model in self.models:
            model.fit(X, y)
```

#### Transformer+LSTM融合
```python
import torch.nn as nn

class TransformerLSTMModel(nn.Module):
    """Transformer+LSTM混合模型"""
    
    def __init__(self, input_dim, hidden_dim=64, n_heads=8):
        super().__init__()
        
        # LSTM层：提取局部时序特征
        self.lstm = nn.LSTM(input_dim, hidden_dim, num_layers=2, batch_first=True)
        
        # Transformer层：捕捉全局依赖
        self.transformer = nn.TransformerEncoder(
            nn.TransformerEncoderLayer(
                d_model=hidden_dim,
                nhead=n_heads,
                batch_first=True
            ),
            num_layers=2
        )
        
        # 输出层
        self.fc = nn.Linear(hidden_dim, 1)
    
    def forward(self, x):
        # LSTM
        lstm_out, _ = self.lstm(x)
        
        # Transformer
        transformer_out = self.transformer(lstm_out)
        
        # 输出
        output = self.fc(transformer_out[:, -1, :])
        return output
```

```python
class AdaptiveLearningRate:
    """自适应学习率"""
    
    def __init__(self, initial_lr=0.01, decay_rate=0.95):
        self.lr = initial_lr
        self.decay_rate = decay_rate
        self.performance_history = []
    
    def update_lr(self, performance):
        """根据性能调整学习率"""
        self.performance_history.append(performance)
        
        # 如果性能下降，降低学习率
        if len(self.performance_history) > 1:
            if performance < self.performance_history[-2]:
                self.lr *= self.decay_rate
        
        return self.lr
```

#### 多臂老虎机探索
```python
from scipy.stats import beta

class MultiArmedBandit:
    """多臂老虎机探索"""
    
    def __init__(self, n_models):
        self.n_models = n_models
        self.alpha = np.ones(n_models)  # 成功次数
        self.beta = np.ones(n_models)   # 失败次数
    
    def select_model(self):
        """选择模型"""
        # Thompson Sampling
        samples = beta.rvs(self.alpha, self.beta)
        return np.argmax(samples)
    
    def update(self, model_idx, success):
        """更新模型表现"""
        if success:
            self.alpha[model_idx] += 1
        else:
            self.beta[model_idx] += 1
```
