## 1. 动态资产配置理论基础

### 1.1 战略资产配置 vs 战术资产配置

**战略资产配置（Strategic Asset Allocation, SAA）**：
- 时间框架：长期（3-5年）
- 目标：实现长期投资目标
- 方法：均值方差优化、风险平价、Black-Litterman
- 调整频率：年度或不定期
- 核心思想：基于长期风险收益特征确定目标配置

**战术资产配置（Tactical Asset Allocation, TAA）**：
- 时间框架：短期（月度/季度）
- 目标：捕捉市场短期机会
- 方法：市场时机、因子配置、信号驱动
- 调整频率：月度或更高频
- 核心思想：根据市场信号偏离战略配置

**动态资产配置（Dynamic Asset Allocation, DAA）**：
- 时间框架：灵活
- 目标：动态适应市场环境
- 方法：规则化触发机制
- 调整频率：根据信号触发
- 核心思想：系统化、规则化的配置调整

### 1.2 动态配置的数学框架

```
w(t) = w_saa + α(t) × Δw_taa

其中：
- w(t): t时刻的配置权重
- w_saa: 战略配置权重
- α(t): 战术调整系数（0-1）
- Δw_taa: 战术偏离幅度
```

```python
import numpy as np
import pandas as pd
from scipy.optimize import minimize
from typing import Dict, List, Tuple, Optional
from dataclasses import dataclass
from enum import Enum

class AllocationType(Enum):
    """配置类型"""
    STRATEGIC = "strategic"
    TACTICAL = "tactical"
    DYNAMIC = "dynamic"

class MarketRegime(Enum):
    """市场状态"""
    BULL = "bull"           # 牛市
    BEAR = "bear"           # 熊市
    SIDEWAYS = "sideways"   # 震荡市
    CRISIS = "crisis"       # 危机

@dataclass
class AllocationSignal:
    """配置信号"""
    signal_type: str
    signal_value: float  # -1到1
    confidence: float    # 0到1
    
    def to_adjustment(self) -> float:
        return self.signal_value * self.confidence

class DynamicAllocationFramework:
    """动态资产配置框架"""
    
    def __init__(self, assets: List[str], risk_free_rate: float = 0.02):
        self.assets = assets
        self.n_assets = len(assets)
        self.risk_free_rate = risk_free_rate
        self.strategic_weights = None
        self.current_weights = None
        
    def set_strategic_allocation(self, weights: np.ndarray):
        """设置战略配置"""
        assert len(weights) == self.n_assets
        self.strategic_weights = weights.copy()
        self.current_weights = weights.copy()
        
    def calculate_signals(self, returns: pd.DataFrame) -> List[AllocationSignal]:
        """计算配置信号"""
        signals = []
        
        # 估值信号
        cumulative = (1 + returns).prod() - 1
        mean_return = returns.mean() * 252
        valuation_dev = (cumulative - mean_return) / mean_return.std()
        signals.append(AllocationSignal('valuation', -np.mean(valuation_dev) / 2, 0.6))
        
        # 动量信号
        momentum = returns.rolling(60).sum().iloc[-1]
        momentum_zscore = (momentum - momentum.mean()) / momentum.std()
        signals.append(AllocationSignal('momentum', np.mean(momentum_zscore) / 2, 0.7))
        
        # 波动率信号
        vol = returns.rolling(60).std() * np.sqrt(252)
        vol_percentile = vol.rank(pct=True).iloc[-1].mean()
        signals.append(AllocationSignal('volatility', -(vol_percentile - 0.5) * 2, 0.5))
        
        return signals
```

## 2. 动态调整触发机制

### 2.1 估值触发机制

**原理**：基于资产估值水平调整配置，低估值超配，高估值低配。

```python
class ValuationTrigger:
    """估值触发机制"""
    
    def __init__(self, threshold_high: float = 1.2, threshold_low: float = 0.8):
        self.threshold_high = threshold_high
        self.threshold_low = threshold_low
    
    def calculate_zscore(self, valuation_series: pd.Series, window: int = 252) -> pd.Series:
        """计算估值Z-score"""
        rolling_mean = valuation_series.rolling(window).mean()
        rolling_std = valuation_series.rolling(window).std()
        return (valuation_series - rolling_mean) / rolling_std
    
    def generate_adjustment(self, current_val: float, historical_mean: float) -> float:
        """生成配置调整建议"""
        ratio = current_val / historical_mean
        if ratio < self.threshold_low:
            return min(0.10 * (1 - ratio / self.threshold_low), 0.20)
        elif ratio > self.threshold_high:
            return max(-0.10 * (ratio / self.threshold_high - 1), -0.20)
        return 0.0

# 示例
def example_valuation_trigger():
    np.random.seed(42)
    dates = pd.date_range('2015-01-01', '2025-12-31', freq='B')
    pe_data = 15 + 5 * np.sin(np.linspace(0, 4*np.pi, len(dates))) + \
              np.random.normal(0, 2, len(dates))
    
    trigger = ValuationTrigger()
    zscore = trigger.calculate_zscore(pd.Series(pe_data, index=dates))
    print(f"当前PE Z-score: {zscore.iloc[-1]:.2f}")
    return trigger
```

### 2.2 动量触发机制

**原理**：基于价格动量调整配置，动量为正时超配，为负时低配。

```python
class MomentumTrigger:
    """动量触发机制"""
    
    def __init__(self, lookback_periods: List[int] = [60, 120, 252]):
        self.lookback_periods = lookback_periods
    
    def time_series_momentum(self, returns: pd.DataFrame, lookback: int = 252) -> pd.Series:
        """时间序列动量"""
        momentum = (1 + returns).rolling(lookback).apply(lambda x: x.prod() - 1, raw=True)
        return np.sign(momentum.iloc[-1])
    
    def cross_sectional_momentum(self, returns: pd.DataFrame, lookback: int = 120) -> pd.Series:
        """横截面动量"""
        momentum = (1 + returns).rolling(lookback).apply(lambda x: x.prod() - 1, raw=True)
        rank = momentum.iloc[-1].rank(pct=True)
        signal = pd.Series(0.0, index=momentum.columns)
        signal[rank > 0.67] = 1.0
        signal[rank < 0.33] = -1.0
        return signal
    
    def momentum_weighting(self, returns: pd.DataFrame) -> pd.Series:
        """动量加权"""
        signals = []
        for period in self.lookback_periods:
            tsm = self.time_series_momentum(returns, period)
            signals.append(tsm)
        avg_signal = pd.concat(signals, axis=1).mean(axis=1)
        weights = avg_signal.copy()
        weights[weights < 0] = 0
        if weights.sum() > 0:
            weights = weights / weights.sum()
        else:
            weights = pd.Series(1/len(weights), index=weights.index)
        return weights

# 示例
def example_momentum_trigger():
    np.random.seed(42)
    dates = pd.date_range('2015-01-01', '2025-12-31', freq='B')
    returns = pd.DataFrame({
        '股票': np.random.normal(0.0008, 0.015, len(dates)),
        '债券': np.random.normal(0.0003, 0.005, len(dates)),
        '商品': np.random.normal(0.0004, 0.020, len(dates))
    }, index=dates)
    
    trigger = MomentumTrigger()
    weights = trigger.momentum_weighting(returns)
    print("动量加权权重:")
    for asset, weight in weights.items():
        print(f"  {asset}: {weight:.2%}")
    return trigger, weights
```

### 2.3 波动率触发机制

**原理**：基于市场波动率调整风险敞口，高波动降低仓位，低波动增加仓位。

```python
class VolatilityTrigger:
    """波动率触发机制"""
    
    def __init__(self, target_vol: float = 0.10, vol_window: int = 60,
                 max_leverage: float = 2.0, min_leverage: float = 0.5):
        self.target_vol = target_vol
        self.vol_window = vol_window
        self.max_leverage = max_leverage
        self.min_leverage = min_leverage
    
    def calculate_leverage(self, returns: pd.DataFrame) -> pd.Series:
        """计算目标杠杆"""
        realized_vol = returns.rolling(self.vol_window).std() * np.sqrt(252)
        leverage = self.target_vol / realized_vol
        return leverage.clip(self.min_leverage, self.max_leverage)
    
    def backtest_vol_targeting(self, returns: pd.DataFrame) -> Dict:
        """回测波动率目标策略"""
        portfolio_returns = returns.mean(axis=1)
        leverage = self.calculate_leverage(portfolio_returns.to_frame()).iloc[:, 0]
        adjusted_returns = portfolio_returns * leverage
        
        annual_return = adjusted_returns.mean() * 252
        annual_vol = adjusted_returns.std() * np.sqrt(252)
        sharpe = annual_return / annual_vol
        
        cumulative = (1 + adjusted_returns).cumprod()
        drawdown = (cumulative - cumulative.cummax()) / cumulative.cummax()
        
        return {
            'annual_return': annual_return,
            'annual_vol': annual_vol,
            'sharpe_ratio': sharpe,
            'max_drawdown': drawdown.min(),
            'target_vol': self.target_vol,
            'realized_vol': annual_vol
        }

# 示例
def example_volatility_trigger():
    np.random.seed(42)
    dates = pd.date_range('2015-01-01', '2025-12-31', freq='B')
    returns = pd.DataFrame({
        '股票': np.random.normal(0.0008, 0.015, len(dates)),
        '债券': np.random.normal(0.0003, 0.005, len(dates))
    }, index=dates)
    
    trigger = VolatilityTrigger(target_vol=0.10)
    result = trigger.backtest_vol_targeting(returns)
    print("波动率目标策略:")
    print(f"  年化收益: {result['annual_return']:.2%}")
    print(f"  夏普比率: {result['sharpe_ratio']:.4f}")
    return trigger, result
```

```python
class RebalanceOptimizer:
    """再平衡优化器"""
    
    def __init__(self, transaction_cost: float = 0.001):
        self.transaction_cost = transaction_cost
    
    def threshold_rebalance(self, current_weights: np.ndarray, 
                           target_weights: np.ndarray, 
                           threshold: float = 0.05) -> Tuple[bool, np.ndarray]:
        """阈值触发再平衡"""
        deviation = np.abs(current_weights - target_weights)
        need_rebalance = np.any(deviation > threshold)
        if need_rebalance:
            return True, target_weights.copy()
        return False, current_weights.copy()
    
    def time_rebalance(self, last_date: pd.Timestamp, 
                       current_date: pd.Timestamp, 
                       frequency: str = 'M') -> bool:
        """时间触发再平衡"""
        days_diff = (current_date - last_date).days
        freq_days = {'D': 1, 'W': 7, 'M': 30, 'Q': 90, 'A': 365}
        return days_diff >= freq_days.get(frequency, 30)
    
    def optimal_rebalance(self, returns: pd.DataFrame, 
                         target_weights: np.ndarray) -> Tuple[float, float]:
        """最优再平衡频率"""
        results = []
        for freq in ['W', 'M', 'Q']:
            # 模拟再平衡
            portfolio_value = 1.0
            for i in range(len(returns)):
                portfolio_return = (target_weights * returns.iloc[i]).sum()
                portfolio_value *= (1 + portfolio_return)
            
            annual_return = (portfolio_value ** (252/len(returns)) - 1)
            results.append((freq, annual_return))
        
        return max(results, key=lambda x: x[1])
```

```python
class MarketRegimeDetector:
    """市场状态识别器"""
    
    def __init__(self, lookback: int = 252):
        self.lookback = lookback
    
    def detect_by_ma(self, prices: pd.Series, short: int = 50, long: int = 200) -> str:
        """均线法识别"""
        short_ma = prices.rolling(short).mean().iloc[-1]
        long_ma = prices.rolling(long).mean().iloc[-1]
        if short_ma > long_ma * 1.05:
            return 'bull'
        elif short_ma < long_ma * 0.95:
            return 'bear'
        return 'sideways'
    
    def detect_by_drawdown(self, prices: pd.Series) -> str:
        """回撤法识别"""
        cumulative_max = prices.cummax()
        drawdown = (prices.iloc[-1] - cumulative_max.iloc[-1]) / cumulative_max.iloc[-1]
        if drawdown > -0.05:
            return 'bull'
        elif drawdown < -0.20:
            return 'bear'
        elif drawdown < -0.10:
            return 'correction'
        return 'sideways'
    
    def detect_by_volatility(self, returns: pd.Series, window: int = 60) -> str:
        """波动率法识别"""
        realized_vol = returns.rolling(window).std().iloc[-1] * np.sqrt(252)
        vol_percentile = (returns.rolling(window).std() < realized_vol / np.sqrt(252)).mean()
        if vol_percentile < 0.2:
            return 'calm'
        elif vol_percentile > 0.8:
            return 'crisis'
        return 'normal'
    
    def get_regime_allocation(self, regime: str) -> Dict[str, float]:
        """获取对应配置"""
        allocations = {
            'bull': {'股票': 0.60, '债券': 0.25, '商品': 0.10, '现金': 0.05},
            'bear': {'股票': 0.20, '债券': 0.50, '商品': 0.05, '现金': 0.25},
            'sideways': {'股票': 0.40, '债券': 0.40, '商品': 0.10, '现金': 0.10},
            'crisis': {'股票': 0.10, '债券': 0.60, '商品': 0.20, '现金': 0.10},
            'calm': {'股票': 0.55, '债券': 0.30, '商品': 0.10, '现金': 0.05}
        }
        return allocations.get(regime, allocations['sideways'])

# 示例
def example_regime_detection():
    np.random.seed(42)
    dates = pd.date_range('2015-01-01', '2025-12-31', freq='B')
    prices = pd.Series(100 * (1 + np.random.normal(0.0005, 0.015, len(dates))).cumprod(), index=dates)
    returns = prices.pct_change().dropna()
    
    detector = MarketRegimeDetector()
    regime_ma = detector.detect_by_ma(prices)
    regime_dd = detector.detect_by_drawdown(prices)
    regime_vol = detector.detect_by_volatility(returns)
    
    print("市场状态识别:")
    print(f"  均线法: {regime_ma}")
    print(f"  回撤法: {regime_dd}")
    print(f"  波动率法: {regime_vol}")
    
    allocation = detector.get_regime_allocation(regime_ma)
    print("\n对应配置:")
    for asset, weight in allocation.items():
        print(f"  {asset}: {weight:.0%}")
    
    return detector
```

```python
class MacroFactorAllocation:
    """宏观因子驱动配置"""
    
    def __init__(self):
        # 经济周期配置
        self.cycle_allocations = {
            'recovery': {'股票': 0.60, '债券': 0.25, '商品': 0.10, '现金': 0.05},
            'expansion': {'股票': 0.50, '债券': 0.30, '商品': 0.15, '现金': 0.05},
            'slowdown': {'股票': 0.35, '债券': 0.45, '商品': 0.05, '现金': 0.15},
            'recession': {'股票': 0.20, '债券': 0.55, '商品': 0.05, '现金': 0.20}
        }
    
    def identify_cycle(self, gdp_growth: float, inflation: float, 
                       interest_rate: float) -> str:
        """识别经济周期"""
        if gdp_growth > 0.03 and inflation < 0.03:
            return 'recovery'
        elif gdp_growth > 0.02 and inflation > 0.02:
            return 'expansion'
        elif gdp_growth < 0.02 and inflation > 0.03:
            return 'slowdown'
        else:
            return 'recession'
    
    def get_allocation(self, gdp_growth: float, inflation: float, 
                       interest_rate: float) -> Dict[str, float]:
        """获取配置"""
        cycle = self.identify_cycle(gdp_growth, inflation, interest_rate)
        return self.cycle_allocations[cycle]
```

### 3.3 风险预算动态调整

```python
class DynamicRiskBudgeting:
    """动态风险预算"""
    
    def __init__(self, base_risk_budget: Dict[str, float], 
                 target_vol: float = 0.10,
                 max_leverage: float = 2.0):
        self.base_risk_budget = base_risk_budget
        self.target_vol = target_vol
        self.max_leverage = max_leverage
    
    def adjust_budget_by_vol(self, returns: pd.DataFrame, window: int = 60) -> Dict[str, float]:
        """根据波动率调整风险预算"""
        # 计算滚动波动率
        vols = returns.rolling(window).std().iloc[-1] * np.sqrt(252)
        
        # 调整预算：高波动降低风险预算
        vol_ratio = vols / vols.mean()
        adjusted_budget = {}
        for asset, budget in self.base_risk_budget.items():
            adjusted_budget[asset] = budget / vol_ratio.get(asset, 1.0)
        
        # 归一化
        total = sum(adjusted_budget.values())
        return {k: v/total for k, v in adjusted_budget.items()}
    
    def adjust_budget_by_momentum(self, returns: pd.DataFrame, 
                                   lookback: int = 120) -> Dict[str, float]:
        """根据动量调整风险预算"""
        # 计算动量
        momentum = returns.rolling(lookback).sum().iloc[-1]
        
        # 调整预算：正动量增加风险预算
        adjusted_budget = {}
        for asset, budget in self.base_risk_budget.items():
            mom_adj = 1.0 + np.sign(momentum.get(asset, 0)) * 0.2
            adjusted_budget[asset] = budget * mom_adj
        
        # 归一化
        total = sum(adjusted_budget.values())
        return {k: v/total for k, v in adjusted_budget.items()}
    
    def calculate_leverage(self, cov_matrix: np.ndarray, 
                          weights: np.ndarray) -> float:
        """计算杠杆"""
        portfolio_vol = np.sqrt(weights @ cov_matrix @ weights)
        leverage = self.target_vol / portfolio_vol
        return min(leverage, self.max_leverage)

# 示例
def example_dynamic_risk_budget():
    np.random.seed(42)
    dates = pd.date_range('2015-01-01', '2025-12-31', freq='B')
    returns = pd.DataFrame({

    base_budget = {'股票': 0.5, '债券': 0.3, '商品': 0.2}
    drb = DynamicRiskBudgeting(base_budget)

    mom_adjusted = drb.adjust_budget_by_momentum(returns)
```

```python
class ETFDynamicAllocationSystem:
    """ETF动态配置系统"""
    
    def __init__(self, assets: List[str], 
                 strategic_weights: np.ndarray,
                 risk_free_rate: float = 0.02):
        self.assets = assets
        self.n_assets = len(assets)
        self.strategic_weights = strategic_weights
        self.risk_free_rate = risk_free_rate
        
        # 组件
        self.regime_detector = MarketRegimeDetector()
        self.valuation_trigger = ValuationTrigger()
        self.momentum_trigger = MomentumTrigger()
        self.volatility_trigger = VolatilityTrigger()
        self.rebalance_optimizer = RebalanceOptimizer()
        
        # 状态
        self.current_weights = strategic_weights.copy()
        self.weight_history = []
        self.signal_history = []
    
    def generate_signals(self, prices: pd.DataFrame, returns: pd.DataFrame) -> Dict:
        """生成综合信号"""
        # 市场状态
        regime = self.regime_detector.detect_by_ma(prices.mean(axis=1))
        regime_alloc = self.regime_detector.get_regime_allocation(regime)
        
        # 估值信号
        cumulative = (1 + returns).prod() - 1
        valuation_signal = -cumulative / cumulative.std()
        
        # 动量信号
        momentum_weights = self.momentum_trigger.momentum_weighting(returns)
        
        # 波动率杠杆
        vol_leverage = self.volatility_trigger.calculate_leverage(returns)
        
        return {
            'regime': regime,
            'regime_allocation': regime_alloc,
            'valuation_signal': valuation_signal,
            'momentum_weights': momentum_weights,
            'vol_leverage': vol_leverage.iloc[-1] if isinstance(vol_leverage, pd.DataFrame) else vol_leverage.iloc[-1, 0]
        }
    
    def calculate_target_weights(self, signals: Dict, 
                                 method: str = 'combined') -> np.ndarray:
        """计算目标权重"""
        if method == 'strategic':
            return self.strategic_weights
        
        elif method == 'regime':
            regime_weights = np.array([signals['regime_allocation'].get(a, 0) 
                                       for a in self.assets])
            return regime_weights / regime_weights.sum()
        
        elif method == 'momentum':
            return signals['momentum_weights'].values
        
        elif method == 'combined':
            # 组合权重
            regime_weights = np.array([signals['regime_allocation'].get(a, 0) 
                                       for a in self.assets])
            momentum_weights = signals['momentum_weights'].values
            
            # 混合（60% regime + 40% momentum）
            combined = 0.6 * regime_weights + 0.4 * momentum_weights
            combined = combined / combined.sum()
            
            # 波动率调整
            leverage = signals['vol_leverage']
            combined = combined * min(leverage, 1.5)  # 限制杠杆
            
            return combined / combined.sum()
        
        return self.strategic_weights
    
    def backtest(self, prices: pd.DataFrame, 
                 rebalance_freq: str = 'M',
                 method: str = 'combined') -> Dict:
        """回测"""
        returns = prices.pct_change().dropna()
        n_periods = len(returns)
        
        portfolio_values = [1.0]
        weights_history = [self.strategic_weights]
        
        for i in range(60, n_periods):  # 60天预热
            # 历史数据
            hist_prices = prices.iloc[:i]
            hist_returns = returns.iloc[:i]
            
            # 生成信号
            signals = self.generate_signals(hist_prices, hist_returns)
            
            # 计算目标权重
            target_weights = self.calculate_target_weights(signals, method)
            
            # 再平衡检查
            if i % 20 == 0:  # 月度再平衡
                self.current_weights = target_weights
            
            # 计算收益
            next_return = returns.iloc[i]
            portfolio_return = (self.current_weights * next_return).sum()
            portfolio_values.append(portfolio_values[-1] * (1 + portfolio_return))
            weights_history.append(self.current_weights.copy())
        
        # 绩效计算
        portfolio_series = pd.Series(portfolio_values, 
                                    index=[returns.index[0] - pd.Timedelta(days=1)] + 
                                          returns.index[60:].tolist())
        returns_series = portfolio_series.pct_change().dropna()
        
        annual_return = returns_series.mean() * 252
        annual_vol = returns_series.std() * np.sqrt(252)
        sharpe = (annual_return - self.risk_free_rate) / annual_vol
        
        cumulative = portfolio_series / portfolio_series.cummax()
        max_drawdown = (cumulative - 1).min()
        
        return {
            'portfolio_values': portfolio_series,
            'weights_history': weights_history,
            'annual_return': annual_return,
            'annual_volatility': annual_vol,
            'sharpe_ratio': sharpe,
            'max_drawdown': max_drawdown,
            'method': method
        }
```

### 5.1 交易成本模型

```python
class TransactionCostModel:
    """交易成本模型"""
    
    def __init__(self, fixed_cost: float = 0.0001,  # 固定成本（万分之1）
                 variable_cost: float = 0.001,      # 可变成本（千分之1）
                 market_impact: float = 0.0005):    # 市场冲击
        self.fixed_cost = fixed_cost
        self.variable_cost = variable_cost
        self.market_impact = market_impact
    
    def calculate_cost(self, old_weights: np.ndarray, 
                       new_weights: np.ndarray,
                       portfolio_value: float = 1.0) -> float:
        """计算交易成本"""
        # 换手率
        turnover = np.abs(new_weights - old_weights).sum() / 2
        
        # 固定成本
        fixed = self.fixed_cost * portfolio_value * (turnover > 0)
        
        # 可变成本
        variable = self.variable_cost * turnover * portfolio_value
        
        # 市场冲击（假设与换手率平方成正比）
        impact = self.market_impact * turnover ** 2 * portfolio_value
        
        return fixed + variable + impact
    
    def cost_aware_rebalance(self, old_weights: np.ndarray,
                            target_weights: np.ndarray,
                            expected_benefit: float,
                            portfolio_value: float = 1.0) -> Tuple[np.ndarray, float]:
        """成本感知再平衡"""
        # 计算交易成本
        cost = self.calculate_cost(old_weights, target_weights, portfolio_value)
        
        # 判断是否值得再平衡
        if expected_benefit > cost:
            return target_weights, cost
        else:
            return old_weights, 0.0

# 示例
def example_transaction_cost():

    old_weights = np.array([0.4, 0.4, 0.1, 0.1])
```

```python
def analyze_rebalance_frequency(returns: pd.DataFrame, 
                               strategic_weights: np.ndarray,
                               frequencies: List[str] = ['W', 'M', 'Q']) -> Dict:
    """分析最优再平衡频率"""
    results = {}
    
    for freq in frequencies:
        portfolio_value = 1.0
        current_weights = strategic_weights.copy()
        
        freq_days = {'W': 5, 'M': 20, 'Q': 60}
        
        for i in range(len(returns)):
            # 再平衡检查
            if i % freq_days.get(freq, 20) == 0:
                current_weights = strategic_weights.copy()
            
            # 计算收益
            portfolio_return = (current_weights * returns.iloc[i]).sum()
            portfolio_value *= (1 + portfolio_return)
            
            # 权重漂移
            current_weights = current_weights * (1 + returns.iloc[i])
            current_weights = current_weights / current_weights.sum()
        
        annual_return = portfolio_value ** (252/len(returns)) - 1
        results[freq] = {'final_value': portfolio_value, 'annual_return': annual_return}
    
    return results
```
