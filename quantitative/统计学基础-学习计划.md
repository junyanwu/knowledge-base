# 项目1: 因子回归分析
import pandas as pd
import numpy as np
import statsmodels.api as sm
from scipy import stats

def factor_regression(stock_returns, factors):
    """
    因子回归分析
    R_it = α + β1*F1 + β2*F2 + ... + ε
    """
    X = sm.add_constant(factors)
    model = sm.OLS(stock_returns, X).fit()
    
    return {
        'alpha': model.params['const'],
        'beta': model.params[1:],
        't_stat': model.tvalues,
        'p_value': model.pvalues,
        'r_squared': model.rsquared,
        'adj_r_squared': model.rsquared_adj,
        'summary': model.summary()
    }

def ic_significance_test(ic_series):
    """
    检验IC是否显著异于零
    H0: IC = 0
    H1: IC ≠ 0
    """
    t_stat, p_value = stats.ttest_1samp(ic_series, 0)
    
    return {
        'ic_mean': ic_series.mean(),
        'ic_std': ic_series.std(),
        't_stat': t_stat,
        'p_value': p_value,
        'significant': p_value < 0.05
    }

# 项目2: 策略收益检验
def strategy_return_test(strategy_returns, benchmark_returns=None):
    """
    检验策略收益是否显著为正
    """
    # 单样本t检验
    t_stat, p_value = stats.ttest_1samp(strategy_returns, 0)
    
    if benchmark_returns is not None:
        # 双样本t检验（配对）
        diff = strategy_returns - benchmark_returns
        t_stat_diff, p_value_diff = stats.ttest_1samp(diff, 0)
        
        return {
            'strategy_return': strategy_returns.mean(),
            'benchmark_return': benchmark_returns.mean(),
            'excess_return': diff.mean(),
            't_stat': t_stat_diff,
            'p_value': p_value_diff,
            'significant': p_value_diff < 0.05
        }
    
    return {
        'mean_return': strategy_returns.mean(),
        't_stat': t_stat,
        'p_value': p_value,
        'significant': p_value < 0.05
    }

# 项目3: 时间序列平稳性检验
from statsmodels.tsa.stattools import adfuller, kpss

def stationarity_test(series, test='adf'):
    """
    时间序列平稳性检验
    """
    if test == 'adf':
        # ADF检验
        # H0: 存在单位根（非平稳）
        # H1: 不存在单位根（平稳）
        result = adfuller(series, autolag='AIC')
        return {
            'test_statistic': result[0],
            'p_value': result[1],
            'critical_values': result[4],
            'stationary': result[1] < 0.05
        }
    elif test == 'kpss':
        # KPSS检验
        # H0: 平稳
        # H1: 非平稳
        result = kpss(series, regression='c')
        return {
            'test_statistic': result[0],
            'p_value': result[1],
            'critical_values': result[3],
            'stationary': result[1] > 0.05
        }

# 项目4: 正态性检验
def normality_test(returns):
    """
    检验收益率是否服从正态分布
    """
    # Jarque-Bera检验
    jb_stat, jb_pvalue = stats.jarque_bera(returns)
    
    # Shapiro-Wilk检验
    sw_stat, sw_pvalue = stats.shapiro(returns[:5000] if len(returns) > 5000 else returns)
    
    # 偏度和峰度
    skewness = stats.skew(returns)
    kurtosis = stats.kurtosis(returns)
    
    return {
        'skewness': skewness,
        'kurtosis': kurtosis,
        'jb_statistic': jb_stat,
        'jb_p_value': jb_pvalue,
        'sw_statistic': sw_stat,
        'sw_p_value': sw_pvalue,
        'normal': jb_pvalue > 0.05 and sw_pvalue > 0.05
    }
