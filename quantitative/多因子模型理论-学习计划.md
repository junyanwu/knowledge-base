# 项目1: 构建Fama-French三因子模型
import pandas as pd
import numpy as np
import statsmodels.api as sm
from statsmodels.regression.rolling import RollingOLS

class FamaFrenchModel:
    def __init__(self, stock_returns, market_return, smb, hml):
        """
        初始化Fama-French三因子模型
        
        参数:
        - stock_returns: 股票收益率
        - market_return: 市场收益率
        - smb: 规模因子
        - hml: 价值因子
        """
        self.stock_returns = stock_returns
        self.market_return = market_return
        self.smb = smb
        self.hml = hml
    
    def fit(self):
        """拟合三因子模型"""
        X = pd.DataFrame({
            'MKT': self.market_return,
            'SMB': self.smb,
            'HML': self.hml
        })
        X = sm.add_constant(X)
        
        model = sm.OLS(self.stock_returns, X).fit()
        
        return {
            'alpha': model.params['const'],
            'beta_mkt': model.params['MKT'],
            'beta_smb': model.params['SMB'],
            'beta_hml': model.params['HML'],
            'r_squared': model.rsquared,
            'adj_r_squared': model.rsquared_adj,
            'summary': model.summary()
        }
    
    def rolling_regression(self, window=60):
        """滚动回归估计动态Beta"""
        X = pd.DataFrame({
            'MKT': self.market_return,
            'SMB': self.smb,
            'HML': self.hml
        })
        
        model = RollingOLS(self.stock_returns, sm.add_constant(X), window=window).fit()
        
        return model.params

# 项目2: 因子有效性检验
class FactorTest:
    def __init__(self, factor_values, forward_returns):
        """
        因子有效性检验
        
        参数:
        - factor_values: 因子值
        - forward_returns: 未来收益率
        """
        self.factor_values = factor_values
        self.forward_returns = forward_returns
    
    def calculate_ic(self, method='spearman'):
        """计算IC"""
        if method == 'pearson':
            ic = self.factor_values.corr(self.forward_returns, method='pearson')
        elif method == 'spearman':
            ic = self.factor_values.corr(self.forward_returns, method='spearman')
        
        return ic
    
    def calculate_ir(self, window=12):
        """计算IR（IC的时间序列均值/标准差）"""
        ic_series = []
        
        for i in range(window, len(self.factor_values)):
            factor_window = self.factor_values.iloc[i-window:i]
            return_window = self.forward_returns.iloc[i-window:i]
            ic = factor_window.corr(return_window, method='spearman')
            ic_series.append(ic)
        
        ic_series = pd.Series(ic_series)
        ir = ic_series.mean() / ic_series.std()
        
        return ir, ic_series
    
    def layer_backtest(self, n_layers=5):
        """分层回测"""
        # 按因子值分组
        labels = [f'Layer{i+1}' for i in range(n_layers)]
        self.forward_returns.name = 'return'
        self.factor_values.name = 'factor'
        
        df = pd.concat([self.factor_values, self.forward_returns], axis=1)
        df['layer'] = pd.qcut(df['factor'], n_layers, labels=labels)
        
        # 计算各层收益
        layer_returns = df.groupby('layer')['return'].mean()
        
        # 多空收益
        long_short_return = layer_returns.iloc[-1] - layer_returns.iloc[0]
        
        return {
            'layer_returns': layer_returns,
            'long_short_return': long_short_return,
            'monotonicity': self._check_monotonicity(layer_returns)
        }
    
    def _check_monotonicity(self, layer_returns):
        """检查单调性"""
        diff = layer_returns.diff().dropna()
        monotonic = (diff > 0).all() or (diff < 0).all()
        return monotonic

from sklearn.linear_model import LinearRegression

class FactorOrthogonalization:
    def __init__(self, factors):
        """
        因子正交化
        
        参数:
        - factors: 因子DataFrame，每列一个因子
        """
        self.factors = factors
    
    def residualize(self, target_factor, base_factors):
        """残差化正交化"""
        # 将目标因子对基础因子回归
        X = self.factors[base_factors]
        y = self.factors[target_factor]
        
        model = LinearRegression()
        model.fit(X, y)
        
        # 残差作为正交化后的因子
        residual = y - model.predict(X)
        
        return residual
    
    def symmetric_orthogonalization(self):
        """对称正交化"""
        from scipy.linalg import sqrtm
        
        # 计算协方差矩阵
        cov = self.factors.cov()
        
        # 对称正交化矩阵
        cov_sqrt_inv = np.linalg.inv(sqrtm(cov))
        
        # 正交化
        factors_orth = self.factors.values @ cov_sqrt_inv
        
        return pd.DataFrame(factors_orth, index=self.factors.index, columns=self.factors.columns)

# 项目4: 多因子组合优化
from scipy.optimize import minimize

class PortfolioOptimizer:
    def __init__(self, expected_returns, factor_cov, idio_var, factor_loadings):
        """
        多因子组合优化
        
        参数:
        - expected_returns: 预期收益
        - factor_cov: 因子协方差矩阵
        - idio_var: 特质风险方差
        - factor_loadings: 因子载荷矩阵
        """
        self.expected_returns = expected_returns
        self.factor_cov = factor_cov
        self.idio_var = idio_var
        self.factor_loadings = factor_loadings
    
    def calculate_portfolio_risk(self, weights):
        """计算组合风险"""
        # 组合因子载荷
        portfolio_loadings = weights @ self.factor_loadings
        
        # 因子风险
        factor_risk = portfolio_loadings @ self.factor_cov @ portfolio_loadings.T
        
        # 特质风险
        idio_risk = (weights ** 2) @ self.idio_var
        
        # 总风险
        total_risk = np.sqrt(factor_risk + idio_risk)
        
        return total_risk
    
    def optimize(self, target_return=None, max_risk=None, 
                 max_weight=0.05, min_weight=0.0):
        """组合优化"""
        n_assets = len(self.expected_returns)
        
        def objective(weights):
            # 最小化风险或最大化收益
            if target_return is not None:
                return self.calculate_portfolio_risk(weights)
            else:
                return -weights @ self.expected_returns
        
        # 约束条件
        constraints = [
            {'type': 'eq', 'fun': lambda w: np.sum(w) - 1}  # 权重和为1
        ]
        
        if target_return is not None:
            constraints.append({
                'type': 'eq',
                'fun': lambda w: w @ self.expected_returns - target_return
            })
        
        if max_risk is not None:
            constraints.append({
                'type': 'ineq',
                'fun': lambda w: max_risk - self.calculate_portfolio_risk(w)
            })
        
        # 权重边界
        bounds = tuple((min_weight, max_weight) for _ in range(n_assets))
        
        # 初始权重
        initial_weights = np.ones(n_assets) / n_assets
        
        # 优化
        result = minimize(
            objective,
            initial_weights,
            method='SLSQP',
            bounds=bounds,
            constraints=constraints
        )
        
        return result.x
