# 稳健验证方法

## 概述

稳健验证是解决策略过拟合、提高样本外表现的关键。对于时序金融数据，传统的K折交叉验证不适用，需要专门的验证方法。

---

## 1. 分层验证（Stratified Validation）

### 核心思想
确保训练集和测试集的市场环境分布一致，避免环境差异导致的性能下降。

### 实现方法

#### 1.1 基于市场状态的分层
```python
def stratified_time_split(df, n_splits=5, stratify_column='regime'):
    """
    时序数据分层划分
    
    原理：
    - 按市场状态（牛市/熊市/震荡）分层
    - 每个fold中各状态的比例与整体一致
    """
    from sklearn.model_selection import StratifiedKFold
    
    # 创建分层标签
    labels = df[stratify_column].values
    
    # 使用StratifiedKFold（需要打乱）
    skf = StratifiedKFold(n_splits=n_splits, shuffle=True, random_state=42)
    
    splits = []
    for train_idx, test_idx in skf.split(df, labels):
        # 确保时序性：训练集索引都小于测试集
        train_idx = sorted(train_idx)
        test_idx = sorted(test_idx)
        
        # 只保留时序约束的样本
        if train_idx[-1] < test_idx[0]:
            splits.append((train_idx, test_idx))
    
    return splits
```

#### 1.2 基于收益分布的分层
```python
def stratified_by_returns(df, n_splits=5, n_bins=10):
    """
    按收益率分位数分层
    确保训练测试集的收益分布相似
    """
    # 将收益率分成n_bins个区间
    df['return_bin'] = pd.qcut(df['returns'], q=n_bins, labels=False, duplicates='drop')
    
    # 分层划分
    splits = []
    test_size = len(df) // n_splits
    
    for i in range(n_splits):
        # 测试集索引范围
        test_start = i * test_size
        test_end = (i + 1) * test_size
        test_idx = list(range(test_start, test_end))
        
        # 训练集：测试集之前的所有数据
        train_idx = list(range(0, test_start))
        
        # 检查分布一致性
        train_dist = df.iloc[train_idx]['return_bin'].value_counts(normalize=True)
        test_dist = df.iloc[test_idx]['return_bin'].value_counts(normalize=True)
        
        # 如果分布差异太大，调整测试集
        ks_stat, ks_pvalue = stats.ks_2samp(
            df.iloc[train_idx]['returns'],
            df.iloc[test_idx]['returns']
        )
        
        splits.append({
            'train_idx': train_idx,
            'test_idx': test_idx,
            'ks_stat': ks_stat,
            'ks_pvalue': ks_pvalue
        })
    
    return splits
```

#### 1.3 基于波动率状态的分层
```python
def stratified_by_volatility(df, n_splits=5):
    """
    按波动率状态分层
    确保训练测试集的波动率分布一致
    """
    # 计算滚动波动率
    df['volatility'] = df['returns'].rolling(63).std() * np.sqrt(252)
    
    # 分成低/中/高波动状态
    df['vol_state'] = pd.cut(
        df['volatility'],
        bins=[0, df['volatility'].quantile(0.33), df['volatility'].quantile(0.67), np.inf],
        labels=['low', 'medium', 'high']
    )
    
    # 分层验证
    splits = stratified_time_split(df, n_splits, 'vol_state')
    
    return splits
```

### 分层验证效果评估
```python
def evaluate_stratified_split(df, splits, regime_column='regime'):
    """
    评估分层验证的效果
    """
    results = []
    
    for i, split in enumerate(splits):
        train_idx, test_idx = split['train_idx'], split['test_idx']
        
        # 检查市场状态分布
        train_regime_dist = df.iloc[train_idx][regime_column].value_counts(normalize=True)
        test_regime_dist = df.iloc[test_idx][regime_column].value_counts(normalize=True)
        
        # 计算分布差异（KL散度）
        kl_div = 0
        for regime in train_regime_dist.index:
            if regime in test_regime_dist.index:
                p = train_regime_dist[regime]
                q = test_regime_dist[regime]
                kl_div += p * np.log(p / q)
        
        results.append({
            'fold': i,
            'train_size': len(train_idx),
            'test_size': len(test_idx),
            'ks_stat': split['ks_stat'],
            'ks_pvalue': split['ks_pvalue'],
            'kl_divergence': kl_div
        })
    
    return pd.DataFrame(results)
```

### 优缺点
**优点**：
- 确保训练测试集环境分布一致
- 减少环境差异导致的过拟合
- 更真实的样本外表现评估

**缺点**：
- 打乱时序可能引入前视偏差
- 实现较复杂
- 需要明确的环境分类标准

## 2. 嵌套交叉验证（Nested CV）

### 核心思想
外层循环评估策略，内层循环优化参数，避免参数选择的信息泄露。

### 实现方法
```python
def nested_cross_validation(df, outer_splits=5, inner_splits=3,
                           param_grid, strategy_func):
    """
    嵌套交叉验证
    
    外层：评估策略表现
    内层：优化策略参数
    
    流程：
    1. 外层将数据分成5份
    2. 对于每个外层fold：
       a. 内层在训练集上用3折CV优化参数
       b. 用最优参数在外层测试集上评估
    """
    from sklearn.model_selection import TimeSeriesSplit
    
    outer_cv = TimeSeriesSplit(n_splits=outer_splits)
    inner_cv = TimeSeriesSplit(n_splits=inner_splits)
    
    outer_results = []
    
    for outer_fold, (outer_train_idx, outer_test_idx) in enumerate(outer_cv.split(df)):
        print(f"Outer fold {outer_fold + 1}/{outer_splits}")
        
        outer_train = df.iloc[outer_train_idx]
        outer_test = df.iloc[outer_test_idx]
        
        # 内层优化
        best_params = None
        best_inner_score = -np.inf
        
        for params in ParameterGrid(param_grid):
            inner_scores = []
            
            for inner_train_idx, inner_val_idx in inner_cv.split(outer_train):
                inner_train = outer_train.iloc[inner_train_idx]
                inner_val = outer_train.iloc[inner_val_idx]
                
                # 在内层验证集上评估
                score = backtest(inner_val, strategy_func, params)
                inner_scores.append(score)
            
            # 内层平均得分
            avg_inner_score = np.mean(inner_scores)
            
            if avg_inner_score > best_inner_score:
                best_inner_score = avg_inner_score
                best_params = params
        
        # 用最优参数在外层测试集上评估
        outer_score = backtest(outer_test, strategy_func, best_params)
        
        outer_results.append({
            'outer_fold': outer_fold,
            'best_params': best_params,
            'inner_score': best_inner_score,
            'outer_score': outer_score,
            'train_size': len(outer_train),
            'test_size': len(outer_test)
        })
    
    return pd.DataFrame(outer_results)
```

## 3. 蒙特卡洛交叉验证

### 核心思想
随机抽样创建多个训练测试集组合，提高验证的稳定性。

### 实现方法
```python
def monte_carlo_cv(df, n_iterations=100, train_ratio=0.7, 
                   gap=63, random_state=42):
    """
    蒙特卡洛交叉验证
    
    参数：
    - n_iterations: 随机抽样次数
    - train_ratio: 训练集比例
    - gap: 训练集和测试集之间的间隔（避免数据泄露）
    """
    np.random.seed(random_state)
    
    results = []
    
    for i in range(n_iterations):
        # 随机选择训练集起始点
        max_start = len(df) - int(len(df) * train_ratio) - gap - int(len(df) * (1 - train_ratio))
        start = np.random.randint(0, max_start)
        
        train_size = int(len(df) * train_ratio)
        test_size = int(len(df) * (1 - train_ratio))
        
        # 划分
        train_idx = list(range(start, start + train_size))
        test_idx = list(range(start + train_size + gap, start + train_size + gap + test_size))
        
        results.append({
            'iteration': i,
            'train_idx': train_idx,
            'test_idx': test_idx
        })
    
    return results

def monte_carlo_cv_with_strategy(df, strategy_func, params, 
                                  n_iterations=100, train_ratio=0.7, gap=63):
    """
    带策略评估的蒙特卡洛CV
    """
    splits = monte_carlo_cv(df, n_iterations, train_ratio, gap)
    
    train_scores = []
    test_scores = []
    
    for split in splits:
        train_data = df.iloc[split['train_idx']]
        test_data = df.iloc[split['test_idx']]
        
        train_score = backtest(train_data, strategy_func, params)
        test_score = backtest(test_data, strategy_func, params)
        
        train_scores.append(train_score)
        test_scores.append(test_score)
    
    return {
        'train_scores': np.array(train_scores),
        'test_scores': np.array(test_scores),
        'mean_train': np.mean(train_scores),
        'mean_test': np.mean(test_scores),
        'std_test': np.std(test_scores),
        'overfit_gap': np.mean(train_scores) - np.mean(test_scores)
    }
```

### 增强版：带环境约束的MC-CV
```python
def monte_carlo_cv_regime_constrained(df, n_iterations=100, train_ratio=0.7,
                                       regime_column='regime', max_distribution_diff=0.1):
    """
    带环境约束的蒙特卡洛CV
    
    只接受训练测试集环境分布差异小于阈值的划分
    """
    valid_splits = []
    max_attempts = n_iterations * 10
    attempts = 0
    
    while len(valid_splits) < n_iterations and attempts < max_attempts:
        attempts += 1
        
        # 随机划分
        split = monte_carlo_cv(df, n_iterations=1, train_ratio=train_ratio)[0]
        
        train_data = df.iloc[split['train_idx']]
        test_data = df.iloc[split['test_idx']]
        
        # 检查环境分布差异
        train_dist = train_data[regime_column].value_counts(normalize=True)
        test_dist = test_data[regime_column].value_counts(normalize=True)
        
        # 计算总变差距离（Total Variation Distance）
        tvd = 0
        for regime in set(train_dist.index) | set(test_dist.index):
            p = train_dist.get(regime, 0)
            q = test_dist.get(regime, 0)
            tvd += abs(p - q) / 2
        
        # 如果差异小于阈值，接受这个划分
        if tvd <= max_distribution_diff:
            split['distribution_diff'] = tvd
            valid_splits.append(split)
    
    print(f"尝试 {attempts} 次，成功 {len(valid_splits)} 个有效划分")
    
    return valid_splits
```

def regime_weighted_validation(df, splits, strategy_func, params, 
                                regime_column='regime'):
    """
    按市场环境加权验证得分
    
    权重基于训练集中各环境的比例
    """
    # 计算整体环境分布（作为基准）
    baseline_dist = df[regime_column].value_counts(normalize=True)
    
    results = []
    
    for split in splits:
        train_data = df.iloc[split['train_idx']]
        test_data = df.iloc[split['test_idx']]
        
        # 分别计算各环境下的得分
        regime_scores = {}
        
        for regime in test_data[regime_column].unique():
            regime_test = test_data[test_data[regime_column] == regime]
            score = backtest(regime_test, strategy_func, params)
            regime_scores[regime] = score
        
        # 计算加权平均得分
        test_dist = test_data[regime_column].value_counts(normalize=True)
        
        weighted_score = 0
        for regime, score in regime_scores.items():
            # 使用基准分布作为权重
            weight = baseline_dist.get(regime, 0)
            weighted_score += weight * score
        
        results.append({
            'raw_score': np.mean(list(regime_scores.values())),
            'weighted_score': weighted_score,
            'regime_scores': regime_scores
        })
    
    return pd.DataFrame(results)
```

def regime_adjusted_returns(df, regime_column='regime'):
    """
    调整不同市场环境下的收益率，使其可比
    
    方法：在各环境内进行标准化
    """
    df = df.copy()
    
    # 各环境内的统计量
    regime_stats = {}
    for regime in df[regime_column].unique():
        regime_data = df[df[regime_column] == regime]['returns']
        regime_stats[regime] = {
            'mean': regime_data.mean(),
            'std': regime_data.std()
        }
    
    # 标准化
    df['adjusted_returns'] = 0.0
    
    for regime, stats in regime_stats.items():
        mask = df[regime_column] == regime
        df.loc[mask, 'adjusted_returns'] = (
            (df.loc[mask, 'returns'] - stats['mean']) / stats['std']
        )
    
    return df, regime_stats
```

def regime_aware_walk_forward(df, train_window=252, test_window=63,
                               strategy_func, param_grid, regime_column='regime'):
    """
    环境感知的Walk-Forward验证
    
    改进：
    1. 检查训练测试集的环境分布差异
    2. 如果差异过大，调整参数或跳过
    3. 根据训练集环境分布调整期望收益
    """
    results = []
    
    for i in range(train_window, len(df) - test_window, test_window):
        train_data = df.iloc[i-train_window:i]
        test_data = df.iloc[i:i+test_window]

                q = test_dist[regime]

#### 5.2 自适应窗口Walk-Forward
```python
def adaptive_window_walk_forward(df, min_train=126, max_train=504, 
                                  test_window=63, strategy_func, param_grid):
    """
    自适应窗口的Walk-Forward验证
    
    根据市场状态变化调整训练窗口大小
    """
    results = []
    
    for i in range(min_train, len(df) - test_window, test_window):
        # 计算市场稳定性
        recent_returns = df['returns'].iloc[i-63:i]
        volatility = recent_returns.std() * np.sqrt(252)
        
        # 自适应窗口大小
        if volatility > 0.3:  # 高波动
            train_window = min_train  # 使用较短窗口
        elif volatility < 0.15:  # 低波动
            train_window = max_train  # 使用较长窗口
        else:
            train_window = int((min_train + max_train) / 2)
        
        # 确保有足够数据
        if i - train_window < 0:
            train_window = i
        
        train_data = df.iloc[i-train_window:i]
        test_data = df.iloc[i:i+test_window]
        
        # 优化和测试
        best_params = optimize_params(train_data, param_grid, strategy_func)
        test_score = backtest(test_data, strategy_func, best_params)
        
        results.append({
            'date': df.index[i],
            'train_window': train_window,
            'volatility': volatility,
            'params': best_params,
            'test_score': test_score
        })
    
    return pd.DataFrame(results)
```

#### 5.3 多周期Walk-Forward
```python
def multi_horizon_walk_forward(df, train_window=252, test_windows=[21, 63, 126],
                                strategy_func, param_grid):
    """
    多预测周期的Walk-Forward验证
    
    在多个预测周期上验证策略稳健性
    """
    results = []
    
    for i in range(train_window, len(df) - max(test_windows), min(test_windows)):
        train_data = df.iloc[i-train_window:i]
        
        # 优化参数
        best_params = optimize_params(train_data, param_grid, strategy_func)
        
        # 在多个周期上测试
        horizon_scores = {}
        
        for horizon in test_windows:
            test_data = df.iloc[i:i+horizon]
            score = backtest(test_data, strategy_func, best_params)
            horizon_scores[f'score_{horizon}d'] = score
        
        results.append({
            'date': df.index[i],
            'params': best_params,
            **horizon_scores
        })
    
    return pd.DataFrame(results)
```

#### 5.4 集成Walk-Forward
```python
def ensemble_walk_forward(df, train_window=252, test_window=63,
                          strategy_func, param_grid, n_models=5):
    """
    集成多个Walk-Forward验证结果
    
    使用不同的训练窗口起点
    """
    all_results = []
    
    for offset in range(test_window):
        results = []
        
        for i in range(train_window + offset, len(df) - test_window, test_window):
            train_data = df.iloc[i-train_window:i]
            test_data = df.iloc[i:i+test_window]
            
            best_params = optimize_params(train_data, param_grid, strategy_func)
            test_score = backtest(test_data, strategy_func, best_params)
            
            results.append({
                'date': df.index[i],
                'params': best_params,
                'test_score': test_score
            })
        
        all_results.append(pd.DataFrame(results))
    
    # 合并所有结果
    combined = pd.concat(all_results, ignore_index=True)
    
    # 计算统计量
    summary = {
        'mean_score': combined['test_score'].mean(),
        'std_score': combined['test_score'].std(),
        'median_score': combined['test_score'].median(),
        'score_range': (combined['test_score'].min(), combined['test_score'].max()),
        'all_results': combined
    }
    
    return summary
```

## 应用到V4.2的建议

### 短期改进（优先级高）
1. **环境感知的Walk-Forward**
   - 在每个fold检查KL散度
   - 差异过大时发出警告

2. **自适应窗口**
   - 高波动时缩短训练窗口
   - 低波动时延长训练窗口

3. **分层验证**
   - 按市场状态分层
   - 确保训练测试集环境分布一致

4. **多周期验证**
   - 在21天、63天、126天三个周期验证
   - 检验策略稳健性

### 长期改进（优先级低）
5. **嵌套CV**
   - 外层评估策略，内层优化参数
   - 获得无偏估计
