# 自适应策略方法

## 概述

自适应策略是解决市场环境变化导致策略失效的核心方法。通过动态调整参数和策略组合，提高策略的稳健性和适应性。

---

## 1. 市场环境自适应参数调整

### 核心思想
根据市场环境动态调整策略参数，使策略能够适应不同的市场状态。

### 参数调整框架

#### 1.1 基于波动率的参数调整
```python
def adaptive_parameters_by_volatility(df, base_lookback=20, base_stop_loss=0.02):
    """
    根据波动率自适应调整参数
    
    原理：
    - 高波动 → 缩短回看期，放大止损
    - 低波动 → 延长回看期，收紧止损
    """
    # 计算波动率分位数
    df['volatility'] = df['returns'].rolling(60).std() * np.sqrt(252)
    df['vol_percentile'] = df['volatility'].rolling(252).rank(pct=True)
    
    vol_pct = df['vol_percentile'].iloc[-1]
    
    # 自适应参数
    if vol_pct < 0.25:  # 低波动
        lookback = int(base_lookback * 1.5)
        stop_loss = base_stop_loss * 0.7
        position_size = 1.2
    elif vol_pct > 0.75:  # 高波动
        lookback = int(base_lookback * 0.7)
        stop_loss = base_stop_loss * 1.5
        position_size = 0.7
    else:  # 正常波动
        lookback = base_lookback
        stop_loss = base_stop_loss
        position_size = 1.0
    
    return {
        'lookback': lookback,
        'stop_loss': stop_loss,
        'position_size': position_size,
        'vol_regime': 'low' if vol_pct < 0.25 else ('high' if vol_pct > 0.75 else 'normal')
    }
```

- 参数调整可能过于频繁

## 2. 滚动窗口优化

### 核心思想
使用滚动窗口动态优化参数，使参数能够适应最近的市場特征。

### 实现方法

#### 2.1 固定窗口滚动优化
```python
def rolling_optimization(df, train_window=252, test_window=63, 
                         param_grid, strategy_func):
    """
    滚动窗口优化
    
    流程：
    1. 在训练窗口上优化参数
    2. 在测试窗口上应用参数
    3. 滚动窗口，重复步骤1-2
    """
    results = []
    param_history = []
    
    for i in range(train_window, len(df) - test_window, test_window):
        # 训练数据
        train_data = df.iloc[i-train_window:i]
        
        # 测试数据
        test_data = df.iloc[i:i+test_window]
        
        # 网格搜索最优参数
        best_params = None
        best_score = -np.inf
        
        for params in ParameterGrid(param_grid):
            # 在训练集上评估
            train_score = backtest(train_data, strategy_func, params)
            
            if train_score > best_score:
                best_score = train_score
                best_params = params
        
        # 在测试集上应用最优参数
        test_score = backtest(test_data, strategy_func, best_params)
        
        results.append({
            'date': df.index[i],
            'params': best_params,
            'train_score': best_score,
            'test_score': test_score
        })
        param_history.append(best_params)
    
    return pd.DataFrame(results), param_history
```

#### 2.3 滚动窗口 + 参数平滑
```python
def rolling_opt_with_smoothing(df, train_window=252, test_window=21,
                                param_grid, strategy_func, smoothing_window=3):
    """
    滚动窗口优化 + 参数平滑
    
    解决参数跳变问题
    """
    # 先执行滚动优化
    results, param_history = rolling_optimization(
        df, train_window, test_window, param_grid, strategy_func
    )
    
    # 对连续数值参数进行平滑
    smoothed_params = []
    
    for i in range(len(param_history)):
        if i < smoothing_window:
            smoothed = param_history[i]
        else:
            # 使用最近smoothing_window个参数的移动平均
            recent_params = param_history[i-smoothing_window+1:i+1]
            smoothed = {
                key: np.mean([p[key] for p in recent_params if isinstance(p[key], (int, float))])
                for key in param_history[i].keys()
            }
        
        smoothed_params.append(smoothed)
    
    results['smoothed_params'] = smoothed_params
    
    return results
```

## 3. 在线学习算法

### 核心思想
算法在实时数据流中不断学习和更新，无需重新训练整个模型。

### 常用算法

#### 3.1 指数加权移动平均（EWMA）
```python
def online_ewma_update(old_value, new_value, alpha=0.1):
    """
    EWMA在线更新
    
    alpha: 衰减因子，越小越平滑
    """
    return alpha * new_value + (1 - alpha) * old_value

# 应用：动态调整策略参数
class OnlineStrategy:
    def __init__(self, initial_params, alpha=0.05):
        self.params = initial_params
        self.alpha = alpha
        self.performance_history = []
    
    def update(self, new_return, new_market_features):
        """
        在线更新参数
        """
        # 记录表现
        self.performance_history.append(new_return)
        
        # 根据最新表现调整参数
        if len(self.performance_history) > 20:
            recent_perf = np.mean(self.performance_history[-20:])
            
            if recent_perf < 0:
                # 表现不好，减小仓位
                self.params['position_size'] = online_ewma_update(
                    self.params['position_size'],
                    self.params['position_size'] * 0.9,
                    self.alpha
                )
            else:
                # 表现好，恢复正常仓位
                self.params['position_size'] = online_ewma_update(
                    self.params['position_size'],
                    1.0,
                    self.alpha
                )
        
        return self.params
```

#### 3.2 在线梯度下降
```python
class OnlineGradientDescent:
    def __init__(self, initial_weights, learning_rate=0.01, lambda_reg=0.01):
        self.weights = np.array(initial_weights)
        self.learning_rate = learning_rate
        self.lambda_reg = lambda_reg  # L2正则化
    
    def predict(self, features):
        """
        预测：w^T * x
        """
        return np.dot(self.weights, features)
    
    def update(self, features, target, prediction):
        """
        在线更新权重
        
        w = w - lr * (gradient + lambda * w)
        """
        error = prediction - target
        gradient = error * features + self.lambda_reg * self.weights
        
        self.weights -= self.learning_rate * gradient
        
        return self.weights
```

#### 3.3 在线模型选择（Bandit算法）
```python
import numpy as np

class UCB_Bandit:
    """
    Upper Confidence Bound算法
    用于在多个策略间动态选择
    """
    def __init__(self, n_strategies, c=1.0):
        self.n_strategies = n_strategies
        self.c = c  # 探索参数
        self.counts = np.zeros(n_strategies)  # 每个策略被选择的次数
        self.values = np.zeros(n_strategies)  # 每个策略的平均收益
    
    def select_strategy(self):
        """
        选择策略：平衡exploitation和exploration
        """
        # 对于未被选择的策略，先尝试
        for i in range(self.n_strategies):
            if self.counts[i] == 0:
                return i
        
        # UCB公式
        total_count = np.sum(self.counts)
        ucb_values = self.values + self.c * np.sqrt(np.log(total_count) / self.counts)
        
        return np.argmax(ucb_values)
    
    def update(self, strategy_idx, reward):
        """
        更新策略的收益估计
        """
        self.counts[strategy_idx] += 1
        n = self.counts[strategy_idx]
        value = self.values[strategy_idx]
        
        # 增量更新均值
        self.values[strategy_idx] = (value * (n - 1) + reward) / n

# 使用示例
bandit = UCB_Bandit(n_strategies=3, c=1.0)  # 3种策略

for day in trading_days:
    # 选择策略
    strategy_idx = bandit.select_strategy()
    
    # 执行策略，获取收益
    reward = execute_strategy(strategy_idx)
    
    # 更新bandit
    bandit.update(strategy_idx, reward)
```
