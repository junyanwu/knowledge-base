### 1.2 选股（Stock Selection）

**技术演进路径**：

#### 第一代：传统多因子模型（2010-2015）

**优点**：可解释性强、计算简单  
**缺点**：无法捕捉非线性、因子交互效应弱


**实战案例**：
- **国信证券多因子选股**：XGBoost+因子正交化，年化25%，夏普1.8
- **国盛金工机器学习系列**：随机森林因子重要性，自动筛选有效因子

#### 第三代：深度学习端到端（2020-2023）
```python
class StockSelectionModel(nn.Module):
    """端到端深度学习选股模型"""
    def __init__(self, input_dim, hidden_dim=64):
        super().__init__()
        self.lstm = nn.LSTM(input_dim, hidden_dim, num_layers=2, batch_first=True)
        self.attention = nn.MultiheadAttention(hidden_dim, num_heads=4)
        self.fc = nn.Linear(hidden_dim, 1)
    
    def forward(self, x):
        lstm_out, _ = self.lstm(x)
        attn_out, _ = self.attention(lstm_out, lstm_out, lstm_out)
        return self.fc(attn_out[:, -1, :])
```

**优势**：自动特征学习、捕捉长期依赖、端到端优化

#### 第四代：图神经网络+LLM（2023-2026）

```python
class NodeTransformerModel(nn.Module):
    """股票关系网络+Transformer"""
    def __init__(self):
        self.gnn = GNN(node_features=64, edge_features=16)
```

#### VaR实时估计（2026年最新）
```python
class RealTimeVaR:
    """分位数回归森林 + 共形校准"""
    def __init__(self):
        self.qrf = QuantileRegressionForest()
        self.calibrator = ConformalCalibration()
    
    def estimate(self, features, confidence=0.95):
        # 离线训练-在线估计框架（OSOA）
        quantile_pred = self.qrf.predict(features, quantile=1-confidence)
        # 共形校准确保统计可靠性
        calibrated_var = self.calibrator.adjust(quantile_pred)
        return calibrated_var
```

**优势**：
- 实时估计，延迟<100ms
- 统计可靠性保证（共形预测）
- 适应非平稳分布

```python
class CoVaRModel:
```

```python
class PortfolioAgent(nn.Module):
    """PPO算法动态资产配置"""
    def __init__(self, n_assets):
        self.actor = nn.Sequential(
            nn.Linear(state_dim, 256),
            nn.ReLU(),
            nn.Dropout(0.2),
            nn.Linear(256, 128),
            nn.ReLU(),
            nn.Linear(128, n_assets),
            nn.Softmax()  # 权重和为1
        )
    
    def act(self, market_state):
        weights = self.actor(market_state)
        return weights

# 训练环境
```

### 1.6 执行算法（Execution Algorithms）

**目标**：最小化市场冲击，优化执行成本

**强化学习应用**：
- **状态空间**：订单簿深度、成交历史、剩余订单量、市场微观结构
- **动作空间**：下单量、下单时机、下单方式（限价/市价）
- **奖励函数**：执行成本、市场冲击、时间风险

```python
class ExecutionEnv(gym.Env):
    """执行算法强化学习环境"""
    def __init__(self, total_shares, time_horizon):
        self.total_shares = total_shares
        self.remaining = total_shares
        self.time_horizon = time_horizon
    
    def step(self, action):
        # action: 下单量
        executed = min(action, self.remaining)
        # 市场冲击模型
        market_impact = self.calculate_impact(executed)
        # 时间风险
        time_penalty = self.time_penalty(self.time_remaining)
        reward = -market_impact - time_penalty
        
        self.remaining -= executed
        done = (self.remaining == 0)
        return self.get_state(), reward, done, {}
```

### 2.1 特征分类体系

特征工程是机器学习在量化中成功的关键。好的特征可以弥补模型的不足，但再好的模型也无法挽救糟糕的特征。

#### 第一类：价量特征（最基础、最可靠）

**价格特征**：
```python
def price_features(df):
    """价格相关特征"""
    features = pd.DataFrame()
    
    # 收益率系列
    features['return_1d'] = df['close'].pct_change(1)
    features['return_5d'] = df['close'].pct_change(5)
    features['return_20d'] = df['close'].pct_change(20)
    
    # 波动率
    features['volatility_5d'] = features['return_1d'].rolling(5).std()
    features['volatility_20d'] = features['return_1d'].rolling(20).std()
    features['volatility_ratio'] = features['volatility_5d'] / features['volatility_20d']
    
    # 价格位置
    features['price_pos_20d'] = (df['close'] - df['low'].rolling(20).min()) / \
                                 (df['high'].rolling(20).max() - df['low'].rolling(20).min())
    
    # 动量
    features['momentum_5d'] = df['close'] / df['close'].shift(5) - 1
    features['momentum_20d'] = df['close'] / df['close'].shift(20) - 1
    
    # 价格跳空
    features['gap'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # 影线
    features['upper_shadow'] = (df['high'] - df[['open', 'close']].max(axis=1)) / df['close']
    features['lower_shadow'] = (df[['open', 'close']].min(axis=1) - df['low']) / df['close']
    
    return features
```

**成交量特征**：
```python
def volume_features(df):
    """成交量特征"""
    features = pd.DataFrame()
    
    # 成交量变化
    features['volume_ma5'] = df['volume'].rolling(5).mean()
    features['volume_ma20'] = df['volume'].rolling(20).mean()
    features['volume_ratio'] = df['volume'] / features['volume_ma20']
    
    # 成交额
    features['amount_ma5'] = df['amount'].rolling(5).mean()
    features['turnover_rate'] = df['volume'] / df['circulating_shares']
    
    # OBV（能量潮）
    features['obv'] = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    features['obv_ma'] = features['obv'].rolling(20).mean()
    
    # VWAP
    features['vwap'] = (df['close'] * df['volume']).rolling(20).sum() / df['volume'].rolling(20).sum()
    features['vwap_deviation'] = (df['close'] - features['vwap']) / features['vwap']
    
    # 成交量波动
    features['volume_volatility'] = df['volume'].rolling(20).std() / features['volume_ma20']
    
    return features
```

#### 第二类：技术指标特征

```python
def technical_indicators(df):
    """技术指标特征"""
    features = pd.DataFrame()
    
    # 均线系统
    for period in [5, 10, 20, 60]:
        features[f'ma{period}'] = df['close'].rolling(period).mean() / df['close'] - 1
    
    # 均线交叉
    features['ma5_ma20_cross'] = (features['ma5'] > features['ma20']).astype(int)
    features['ma10_ma60_cross'] = (features['ma10'] > features['ma60']).astype(int)
    
    # RSI
    for period in [6, 14, 24]:
        delta = df['close'].diff()
        gain = delta.where(delta > 0, 0).rolling(period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(period).mean()
        features[f'rsi_{period}'] = 100 - (100 / (1 + gain / loss))
    
    # MACD
    ema12 = df['close'].ewm(span=12, adjust=False).mean()
    ema26 = df['close'].ewm(span=26, adjust=False).mean()
    features['macd'] = ema12 - ema26
    features['macd_signal'] = features['macd'].ewm(span=9, adjust=False).mean()
    features['macd_hist'] = features['macd'] - features['macd_signal']
    
    # 布林带
    features['boll_upper'] = df['close'].rolling(20).mean() + 2 * df['close'].rolling(20).std()
    features['boll_lower'] = df['close'].rolling(20).mean() - 2 * df['close'].rolling(20).std()
    features['boll_width'] = (features['boll_upper'] - features['boll_lower']) / df['close']
    features['boll_pos'] = (df['close'] - features['boll_lower']) / (features['boll_upper'] - features['boll_lower'])
    
    # KDJ
    low_min = df['low'].rolling(9).min()
    high_max = df['high'].rolling(9).max()
    features['kdj_k'] = (df['close'] - low_min) / (high_max - low_min) * 100
    features['kdj_d'] = features['kdj_k'].rolling(3).mean()
    features['kdj_j'] = 3 * features['kdj_k'] - 2 * features['kdj_d']
    
    # ATR（真实波动幅度）
    tr = pd.concat([
        df['high'] - df['low'],
        (df['high'] - df['close'].shift()).abs(),
        (df['low'] - df['close'].shift()).abs()
    ], axis=1).max(axis=1)
    features['atr_14'] = tr.rolling(14).mean()
    
    return features
```

#### 第三类：财务特征（基本面）

```python
def financial_features(df):
    """财务特征"""
    features = pd.DataFrame()
    
    # 估值指标
    features['pe_ratio'] = df['market_cap'] / df['net_profit']
    features['pb_ratio'] = df['market_cap'] / df['net_assets']
    features['ps_ratio'] = df['market_cap'] / df['revenue']
    features['peg_ratio'] = features['pe_ratio'] / df['profit_growth']
    
    # 盈利能力
    features['roe'] = df['net_profit'] / df['net_assets']
    features['roa'] = df['net_profit'] / df['total_assets']
    features['gross_margin'] = (df['revenue'] - df['cost']) / df['revenue']
    features['net_margin'] = df['net_profit'] / df['revenue']
    
    # 成长性
    features['revenue_growth'] = df['revenue'].pct_change(4)  # 同比增长
    features['profit_growth'] = df['net_profit'].pct_change(4)
    features['eps_growth'] = df['eps'].pct_change(4)
    
    # 财务健康度
    features['debt_ratio'] = df['total_liabilities'] / df['total_assets']
    features['current_ratio'] = df['current_assets'] / df['current_liabilities']
    features['quick_ratio'] = (df['current_assets'] - df['inventory']) / df['current_liabilities']
```

```python
def handle_missing_values(features):
    """处理缺失值"""
    # 策略1：向前填充（时间序列）
    features = features.fillna(method='ffill')
    
    # 策略2：行业均值填充
    for col in features.columns:
        features[col] = features.groupby('industry')[col].transform(
            lambda x: x.fillna(x.mean())
        )
    
    # 策略3：极端值处理
    features = features.clip(lower=features.quantile(0.01), 
                            upper=features.quantile(0.99), 
                            axis=1)
    
    return features
```

```python
def feature_selection(X, y, method='importance', top_k=50):
    """特征选择"""
    if method == 'importance':
        # 随机森林特征重要性
        rf = RandomForestRegressor(n_estimators=100, random_state=42, n_jobs=-1)
        rf.fit(X, y)
        importance = pd.Series(rf.feature_importances_, index=X.columns)
        selected = importance.nlargest(top_k).index.tolist()
    
    elif method == 'correlation':
        # 相关性筛选
        corr_with_target = X.corrwith(y).abs()
        selected = corr_with_target.nlargest(top_k).index.tolist()
    
    elif method == 'mutual_info':
        # 互信息
        from sklearn.feature_selection import mutual_info_regression
        mi = mutual_info_regression(X, y)
        mi_series = pd.Series(mi, index=X.columns)
        selected = mi_series.nlargest(top_k).index.tolist()
    
    elif method == 'shap':
        # SHAP值（最准确但计算量大）
        import shap
        model = XGBRegressor().fit(X, y)
        explainer = shap.TreeExplainer(model)
        shap_values = explainer.shap_values(X)
        shap_importance = pd.Series(np.abs(shap_values).mean(axis=0), index=X.columns)
        selected = shap_importance.nlargest(top_k).index.tolist()
    
    return X[selected], selected
```

```python
class LLMBasedFactorMining:
    """基于LLM的自动化因子挖掘"""
    def __init__(self, llm_model):
        self.llm = llm_model
        self.backtester = Backtester()
        self.code_generator = CodeGenerator()
    
    def generate_factor_hypothesis(self, domain_knowledge):
        """LLM生成因子假设"""
        prompt = f"""
        基于以下市场知识，生成新的因子假设：
        {domain_knowledge}
        
        要求：
        1. 因子逻辑清晰，有经济学意义
        2. 可量化实现，避免未来函数
        3. 与现有因子相关性低
        
        输出格式：
        - 因子名称
        - 因子逻辑（为什么有效）
        - 实现伪代码
        - 预期表现（IC、分组收益）
        """
        return self.llm.generate(prompt)
    
    def implement_factor(self, hypothesis):
        """代码生成Agent实现因子"""
        code = self.code_generator.generate(hypothesis)
        return eval(code)
    
    def backtest_and_feedback(self, factor_code):
        """回测反馈"""
        result = self.backtester.run(factor_code)
        if result['ic'] > 0.05 and result['sharpe'] > 1.5:
            return "Accept", result
        else:
            return "Reject", result
```
