## 2. 向量化回测

### 2.1 基础框架

```python
import pandas as pd
import numpy as np

class VectorBacktest:
    """向量化回测框架"""
    
    def __init__(self, initial_capital=100000, commission=0.0003):
        self.initial_capital = initial_capital
        self.commission = commission
    
    def run(self, signals, prices):
        """
        运行回测
        
        参数:
        signals: 交易信号 (1=买入, -1=卖出, 0=持有)
        prices: 价格序列
        
        返回:
        回测结果
        """
        # 计算收益
        returns = prices.pct_change()
        
        # 信号收益
        strategy_returns = signals.shift(1) * returns
        
        # 扣除交易成本
        trades = signals.diff().abs()
        transaction_costs = trades * self.commission
        strategy_returns = strategy_returns - transaction_costs
        
        # 累计收益
        cumulative = (1 + strategy_returns).cumprod()
        
        # 绩效指标
        performance = self.calculate_performance(strategy_returns)
        
        return {
            'returns': strategy_returns,
            'cumulative': cumulative,
            'performance': performance
        }
    
    def calculate_performance(self, returns):
        """计算绩效指标"""
        # 年化收益
        annual_return = returns.mean() * 252
        
        # 年化波动率
        annual_vol = returns.std() * np.sqrt(252)
        
        # 夏普比率
        sharpe = annual_return / annual_vol if annual_vol > 0 else 0
        
        # 最大回撤
        cumulative = (1 + returns).cumprod()
        drawdown = (cumulative - cumulative.cummax()) / cumulative.cummax()
        max_drawdown = drawdown.min()
        
        # 胜率
        win_rate = (returns > 0).mean()
        
        # 盈亏比
        profit_loss_ratio = returns[returns > 0].mean() / abs(returns[returns < 0].mean()) if (returns < 0).any() else 0
        
        return {
            'annual_return': annual_return,
            'annual_vol': annual_vol,
            'sharpe': sharpe,
            'max_drawdown': max_drawdown,
            'win_rate': win_rate,
            'profit_loss_ratio': profit_loss_ratio
        }
```

```python
class MultiStrategyBacktest:
    """多策略回测"""
    
    def __init__(self, initial_capital=100000):
        self.initial_capital = initial_capital
    
    def run(self, strategies_data):
        """
        运行多策略回测
        
        参数:
        strategies_data: {策略名: {'signals': signals, 'prices': prices}}
        
        返回:
        回测结果
        """
        results = {}
        
        for name, data in strategies_data.items():
            backtest = VectorBacktest(self.initial_capital)
            result = backtest.run(data['signals'], data['prices'])
            results[name] = result
        
        return results
    
    def compare(self, results):
        """比较策略表现"""
        comparison = pd.DataFrame({
            name: {
                '年化收益': result['performance']['annual_return'],
                '年化波动': result['performance']['annual_vol'],
                '夏普比率': result['performance']['sharpe'],
                '最大回撤': result['performance']['max_drawdown'],
                '胜率': result['performance']['win_rate']
            }
            for name, result in results.items()
        }).T
        
        return comparison
```

```python
class EventEngine:
    """事件引擎"""
    
    def __init__(self):
        self.events = []
    
    def put(self, event):
        """放入事件"""
        self.events.append(event)
    
    def get(self):
        """获取事件"""
        if self.events:
            return self.events.pop(0)
        return None
    
    def run(self):
        """运行事件循环"""
        while self.events:
            event = self.get()
            self.process(event)
    
    def process(self, event):
        """处理事件"""
        if isinstance(event, MarketEvent):
            self.on_market(event)
        elif isinstance(event, SignalEvent):
            self.on_signal(event)
        elif isinstance(event, OrderEvent):
```

### 3.3 完整回测框架

```python
class EventDrivenBacktest:
    """事件驱动回测"""
    
    def __init__(self, data, strategy, initial_capital=100000, commission=0.0003):
        self.data = data
        self.strategy = strategy
        self.initial_capital = initial_capital
        self.commission = commission
        
        # 状态
        self.cash = initial_capital
        self.positions = {}  # {symbol: quantity}
        self.portfolio_value = initial_capital
        
        # 记录
        self.trades = []
        self.daily_values = []
    
    def run(self):
        """运行回测"""
        for timestamp, row in self.data.iterrows():
            # 市场事件
            market_event = MarketEvent(timestamp, row)
            
            # 生成信号
            signal = self.strategy.generate_signal(row)
            
            if signal:
                # 创建订单
                order = self.create_order(signal, row)
                
                # 执行订单
                fill = self.execute_order(order, row)
                
                # 更新状态
                self.update_portfolio(fill)
            
            # 记录每日价值
            self.record_daily_value(timestamp, row)
        
        return self.get_results()
    
    def create_order(self, signal, data):
        """创建订单"""
        if signal == 'BUY':
            quantity = int(self.cash * 0.95 / data['close'])
            return OrderEvent('ETF', 'MARKET', quantity, data['close'])
        elif signal == 'SELL':
            quantity = self.positions.get('ETF', 0)
            return OrderEvent('ETF', 'MARKET', -quantity, data['close'])
    
    def execute_order(self, order, data):
        """执行订单"""
        price = data['close']
        commission = abs(order.quantity) * price * self.commission
        
        return FillEvent(
            order.symbol,
            data.name,
            order.quantity,
            price,
            commission
        )
    
    def update_portfolio(self, fill):
        """更新组合"""
        # 更新持仓
        self.positions[fill.symbol] = self.positions.get(fill.symbol, 0) + fill.quantity
        
        # 更新现金
        self.cash -= fill.quantity * fill.price + fill.commission
        
        # 记录交易
        self.trades.append({
            'timestamp': fill.timestamp,
            'symbol': fill.symbol,
            'quantity': fill.quantity,
            'price': fill.price,
            'commission': fill.commission
        })
    
    def record_daily_value(self, timestamp, data):
        """记录每日价值"""
        positions_value = sum(
            self.positions.get(symbol, 0) * data['close']
            for symbol in self.positions
        )
        total_value = self.cash + positions_value
        
        self.daily_values.append({
            'timestamp': timestamp,
            'cash': self.cash,
            'positions_value': positions_value,
            'total_value': total_value
        })
    
    def get_results(self):
        """获取回测结果"""
        daily_values = pd.DataFrame(self.daily_values).set_index('timestamp')
        
        # 计算收益
        returns = daily_values['total_value'].pct_change()
        
        # 绩效
        annual_return = returns.mean() * 252
        annual_vol = returns.std() * np.sqrt(252)
        sharpe = annual_return / annual_vol if annual_vol > 0 else 0
        
        drawdown = (daily_values['total_value'] - daily_values['total_value'].cummax()) / daily_values['total_value'].cummax()
        max_drawdown = drawdown.min()
        
        return {
            'daily_values': daily_values,
            'returns': returns,
            'trades': self.trades,
            'performance': {
                'annual_return': annual_return,
                'annual_vol': annual_vol,
                'sharpe': sharpe,
                'max_drawdown': max_drawdown
            }
        }
```

## 4. 回测陷阱

### 4.1 常见陷阱

**1. 未来函数**
```python
# 错误：使用未来数据
signal = data['close'] > data['close'].shift(-1)  # 使用了明天的价格

# 正确：使用历史数据
signal = data['close'] > data['close'].shift(1)  # 使用昨天的价格
```

**2. 幸存者偏差**
```python
# 错误：只使用当前存在的股票
data = get_current_stocks()  # 排除了退市股票

# 正确：使用历史成分股
data = get_historical_constituents(date)
```

**3. 过度拟合**
```python
# 错误：过度优化参数
best_params = optimize_all_parameters(data)  # 在同一数据上优化和测试

# 正确：样本外测试
train_data = data[:'2022']
test_data = data['2023':]
best_params = optimize(train_data)
test(test_data, best_params)
```

**4. 交易成本**
```python
# 错误：忽略交易成本
returns = signals.shift(1) * price_returns

# 正确：扣除交易成本
trades = signals.diff().abs()
costs = trades * commission
returns = signals.shift(1) * price_returns - costs
```

## 5. 实战示例

### 5.1 完整回测流程

```python
class CompleteBacktest:
    """完整回测流程"""
    
    def __init__(self, data, strategy_class, config):
        self.data = data
        self.strategy_class = strategy_class
        self.config = config
    
    def run(self):
        """运行完整回测"""
        # 1. 数据划分
        train_data = self.data[:'2022']
        test_data = self.data['2023':]
        
        # 2. 参数优化
        best_params = self.optimize_parameters(train_data)
        
        # 3. 样本外测试
        strategy = self.strategy_class(**best_params)
        backtest = EventDrivenBacktest(test_data, strategy, self.config)
        results = backtest.run()
        
        # 4. 绩效分析
        performance = self.analyze_performance(results)
        
        # 5. 风险分析
        risks = self.analyze_risks(results)
        
        return {
            'results': results,
            'performance': performance,
            'risks': risks,
            'best_params': best_params
        }
    
    def optimize_parameters(self, data):
        """参数优化"""
        from itertools import product
        
        # 参数网格
        param_grid = {
            'short_window': [5, 10, 20],
            'long_window': [40, 60, 80],
            'stop_loss': [0.05, 0.08, 0.10]
        }
        
        best_params = None
        best_sharpe = -np.inf
        
        for params in product(*param_grid.values()):
            param_dict = dict(zip(param_grid.keys(), params))
            
            strategy = self.strategy_class(**param_dict)
            backtest = EventDrivenBacktest(data, strategy, self.config)
            results = backtest.run()
            
            sharpe = results['performance']['sharpe']
            
            if sharpe > best_sharpe:
                best_sharpe = sharpe
                best_params = param_dict
        
        return best_params
    
    def analyze_performance(self, results):
        """绩效分析"""
        returns = results['returns']
        
        # 收益指标
        annual_return = returns.mean() * 252
        cumulative_return = (1 + returns).prod() - 1
        
        # 风险指标
        annual_vol = returns.std() * np.sqrt(252)
        sharpe = annual_return / annual_vol if annual_vol > 0 else 0
        
        # 回撤指标
        cumulative = (1 + returns).cumprod()
        drawdown = (cumulative - cumulative.cummax()) / cumulative.cummax()
        max_drawdown = drawdown.min()
        avg_drawdown = drawdown.mean()
        
        # 交易指标
        trades = results['trades']
        win_trades = [t for t in trades if t['quantity'] > 0]  # 简化
        win_rate = len(win_trades) / len(trades) if trades else 0
        
        return {
            'annual_return': annual_return,
            'cumulative_return': cumulative_return,
            'annual_vol': annual_vol,
            'sharpe': sharpe,
            'max_drawdown': max_drawdown,
            'avg_drawdown': avg_drawdown,
            'win_rate': win_rate,
            'num_trades': len(trades)
        }
    
    def analyze_risks(self, results):
        """风险分析"""
        returns = results['returns']
        
        # VaR
        var_95 = returns.quantile(0.05)
        var_99 = returns.quantile(0.01)
        
        # CVaR
        cvar_95 = returns[returns < var_95].mean()
        
        # 下行风险
        downside_returns = returns[returns < 0]
        downside_vol = downside_returns.std() * np.sqrt(252)
        sortino = returns.mean() * 252 / downside_vol if downside_vol > 0 else 0
        
        # 偏度、峰度
        skewness = returns.skew()
        kurtosis = returns.kurtosis()
        
        return {
            'var_95': var_95,
            'var_99': var_99,
            'cvar_95': cvar_95,
            'downside_vol': downside_vol,
            'sortino': sortino,
            'skewness': skewness,
            'kurtosis': kurtosis
        }
```

## 6. 总结

回测框架要点：

1. **向量化回测**：快速、简单、适合策略初筛
2. **事件驱动回测**：精确、复杂、适合实盘模拟
3. **常见陷阱**：未来函数、幸存者偏差、过拟合、交易成本
4. **检查清单**：确保回测可靠性
5. **完整流程**：数据划分→参数优化→样本外测试→绩效分析

**注意事项：**
- 始终使用样本外数据测试
- 考虑所有交易成本
- 检查未来函数
- 避免过度优化
