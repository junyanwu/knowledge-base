## 2. 权重优化方法

### 2.1 等权重组合

**定义**：每个资产分配相同权重

**公式**：
```
w_i = 1 / n  ∀i
```

**优势**：
- 简单易懂
- 无需参数估计
- 避免估计误差
- 稳健性强

**局限**：
- 忽略资产特征差异
- 风险不均衡
- 收益可能次优

**适用场景**：
- 资产特征相似
- 参数估计困难
- 不确定环境

```python
import numpy as np
import pandas as pd
from scipy.optimize import minimize
import matplotlib.pyplot as plt

class PortfolioWeightOptimizer:
    """组合权重优化器"""
    
    def __init__(self, 
                 returns: pd.DataFrame,
                 risk_free_rate: float = 0.03):
        """
        初始化
        
        Args:
            returns: 资产收益率DataFrame
            risk_free_rate: 无风险利率
        """
        self.returns = returns
        self.n_assets = returns.shape[1]
        self.asset_names = returns.columns.tolist()
        self.risk_free_rate = risk_free_rate
        
        # 计算统计量
        self.mean_returns = returns.mean() * 252
        self.cov_matrix = returns.cov() * 252
        self.vols = np.sqrt(np.diag(self.cov_matrix))
    
    def equal_weight(self) -> np.ndarray:
        """
        等权重组合
        
        Returns:
            权重向量
        """
        return np.ones(self.n_assets) / self.n_assets
    
    def inverse_volatility_weight(self) -> np.ndarray:
        """
        逆波动率权重
        
        Returns:
            权重向量
        """
        weights = 1 / self.vols
        return weights / np.sum(weights)
    
    def evaluate_portfolio(self, weights: np.ndarray) -> Dict:
        """
        评估组合表现
        
        Args:
            weights: 权重向量
            
        Returns:
            组合指标
        """
        # 组合收益
        portfolio_return = np.dot(weights, self.mean_returns)
        
        # 组合波动率
        portfolio_vol = np.sqrt(np.dot(weights.T, np.dot(self.cov_matrix, weights)))
        
        # 夏普比率
        sharpe_ratio = (portfolio_return - self.risk_free_rate) / portfolio_vol
        
        # 风险贡献
        marginal_risk = np.dot(self.cov_matrix, weights) / portfolio_vol
        risk_contrib = weights * marginal_risk
        risk_contrib_pct = risk_contrib / portfolio_vol
        
        return {
            'weights': weights,
            'expected_return': portfolio_return,
            'volatility': portfolio_vol,
            'sharpe_ratio': sharpe_ratio,
            'risk_contribution': risk_contrib_pct
        }
```

### 2.2 最小方差组合

**目标**：最小化组合方差

**优化问题**：
```
min w'Σw
s.t. Σ w_i = 1
     w_i ≥ 0
```

**闭式解（无约束）**：
```
w = Σ^(-1) 1 / (1' Σ^(-1) 1)
```

**特点**：
- 风险最小化
- 不需要预期收益估计
- 对协方差矩阵敏感
- 可能过度集中

```python
    def minimum_variance(self, 
                        allow_short: bool = False) -> np.ndarray:
        """
        最小方差组合
        
        Args:
            allow_short: 是否允许卖空
            
        Returns:
            权重向量
        """
        def portfolio_variance(weights):
            return np.dot(weights.T, np.dot(self.cov_matrix, weights))
        
        # 约束条件
        constraints = [{'type': 'eq', 'fun': lambda w: np.sum(w) - 1}]
        
        # 权重约束
        if allow_short:
            bounds = None
        else:
            bounds = tuple((0, 1) for _ in range(self.n_assets))
        
        # 初始值
        init_weights = np.ones(self.n_assets) / self.n_assets
        
        # 优化
        result = minimize(
            portfolio_variance,
            init_weights,
            method='SLSQP',
            bounds=bounds,
            constraints=constraints,
            options={'ftol': 1e-12, 'maxiter': 1000}
        )
        
        return result.x
```

### 2.3 风险平价权重

**目标**：每个资产风险贡献相等

**优化问题**：
```
min Σ (RC_i - σ_p/n)²
s.t. Σ w_i = 1
     w_i ≥ 0
```

```python
    def risk_parity(self, 
                   risk_budget: np.ndarray = None) -> np.ndarray:
        """
        风险平价权重
        
        Args:
            risk_budget: 风险预算（默认等风险贡献）
            
        Returns:
            权重向量
        """
        if risk_budget is None:
            risk_budget = np.ones(self.n_assets) / self.n_assets
        
        def risk_budget_objective(weights):
            # 组合波动率
            portfolio_vol = np.sqrt(
                np.dot(weights.T, np.dot(self.cov_matrix, weights))
            )
            
            # 风险贡献比例
            marginal_risk = np.dot(self.cov_matrix, weights) / portfolio_vol
            risk_contrib_pct = weights * marginal_risk / portfolio_vol
            
            # 目标函数
            return np.sum((risk_contrib_pct - risk_budget) ** 2)
        
        # 约束条件
        constraints = [{'type': 'eq', 'fun': lambda w: np.sum(w) - 1}]
        bounds = tuple((0.001, 1) for _ in range(self.n_assets))
        
        # 初始值
        init_weights = np.ones(self.n_assets) / self.n_assets
        
        # 优化
        result = minimize(
            risk_budget_objective,
            init_weights,
            method='SLSQP',
            bounds=bounds,
            constraints=constraints,
            options={'ftol': 1e-12, 'maxiter': 1000}
        )
        
        return result.x
```

### 2.4 最大夏普比率组合

**目标**：最大化风险调整收益

**优化问题**：
```
max (w'μ - r_f) / √(w'Σw)
s.t. Σ w_i = 1
     w_i ≥ 0
```

**闭式解（无约束）**：
```
w = Σ^(-1) (μ - r_f 1) / (1' Σ^(-1) (μ - r_f))
```

```python
    def max_sharpe_ratio(self, 
                        allow_short: bool = False) -> np.ndarray:
        """
        最大夏普比率组合
        
        Args:
            allow_short: 是否允许卖空
            
        Returns:
            权重向量
        """
        def neg_sharpe_ratio(weights):
            portfolio_return = np.dot(weights, self.mean_returns)
            portfolio_vol = np.sqrt(
                np.dot(weights.T, np.dot(self.cov_matrix, weights))
            )
            sharpe = (portfolio_return - self.risk_free_rate) / portfolio_vol
            return -sharpe  # 最小化负夏普比率
        
        # 约束条件
        constraints = [{'type': 'eq', 'fun': lambda w: np.sum(w) - 1}]
        
        # 权重约束
        if allow_short:
            bounds = None
        else:
            bounds = tuple((0, 1) for _ in range(self.n_assets))
        
        # 初始值
        init_weights = np.ones(self.n_assets) / self.n_assets
        
        # 优化
        result = minimize(
            neg_sharpe_ratio,
            init_weights,
            method='SLSQP',
            bounds=bounds,
            constraints=constraints,
            options={'ftol': 1e-12, 'maxiter': 1000}
        )
        
        return result.x
```

```python
    def efficient_frontier(self, 
                          n_points: int = 100,
                          allow_short: bool = False) -> pd.DataFrame:

        计算有效前沿

        Returns:
            有效前沿数据

        # 最小方差组合
        min_var_weights = self.minimum_variance(allow_short)
        min_var_return = np.dot(min_var_weights, self.mean_returns)
        
        # 最大收益组合
        max_return_idx = np.argmax(self.mean_returns)
        max_return = self.mean_returns[max_return_idx]
        
        # 目标收益范围
        target_returns = np.linspace(min_var_return, max_return, n_points)
        
        frontier = []
        
        for target_return in target_returns:
            def portfolio_variance(weights):
                return np.dot(weights.T, np.dot(self.cov_matrix, weights))
            
            # 约束条件
            constraints = [
                {'type': 'eq', 'fun': lambda w: np.sum(w) - 1},
                {'type': 'eq', 'fun': lambda w, r=target_return: np.dot(w, self.mean_returns) - r}
            ]

            if allow_short:
                bounds = None
            else:
                bounds = tuple((0, 1) for _ in range(self.n_assets))
            
            # 初始值
            init_weights = np.ones(self.n_assets) / self.n_assets
            
            # 优化
            result = minimize(
                portfolio_variance,
                init_weights,
                method='SLSQP',
                bounds=bounds,
                constraints=constraints,
                options={'ftol': 1e-12, 'maxiter': 1000}

                    'target_return': target_return,
                    'volatility': portfolio_vol,
                    'sharpe_ratio': sharpe,
```

### 3.1 权重上下限约束

```python
class ConstrainedOptimizer:
    """约束优化器"""
    
    def __init__(self, 
                 returns: pd.DataFrame,
                 risk_free_rate: float = 0.03):
        """
        初始化
        
        Args:
            returns: 资产收益率
            risk_free_rate: 无风险利率
        """
        self.returns = returns
        self.n_assets = returns.shape[1]
        self.asset_names = returns.columns.tolist()
        self.risk_free_rate = risk_free_rate
        
        self.mean_returns = returns.mean() * 252
        self.cov_matrix = returns.cov() * 252
    
    def weight_bounds(self, 
                     min_weight: float = 0.0,
                     max_weight: float = 1.0,
                     asset_specific_bounds: Dict[str, tuple] = None) -> List[tuple]:
        """
        权重上下限约束
        
        Args:
            min_weight: 最小权重
            max_weight: 最大权重
            asset_specific_bounds: 特定资产权重约束
            
        Returns:
            权重边界列表
        """
        bounds = []
        
        for asset in self.asset_names:
            if asset_specific_bounds and asset in asset_specific_bounds:
                bounds.append(asset_specific_bounds[asset])
            else:
                bounds.append((min_weight, max_weight))
        
        return bounds
    
    def min_weight_optimization(self, 
                               objective: str = 'sharpe',
                               min_weight: float = 0.05,
                               max_weight: float = 0.30) -> np.ndarray:
        """
        最小权重约束优化
        
        Args:
            objective: 优化目标
            min_weight: 最小权重
            max_weight: 最大权重
            
        Returns:
            权重向量
        """
        bounds = self.weight_bounds(min_weight, max_weight)
        
        if objective == 'sharpe':
            return self._optimize_sharpe(bounds)
        elif objective == 'min_var':
            return self._optimize_min_variance(bounds)
        elif objective == 'risk_parity':
            return self._optimize_risk_parity(bounds)
        else:
            raise ValueError(f"Unknown objective: {objective}")
    
    def _optimize_sharpe(self, bounds: List[tuple]) -> np.ndarray:
        """最大夏普比率优化"""
        def neg_sharpe(weights):
            portfolio_return = np.dot(weights, self.mean_returns)
            portfolio_vol = np.sqrt(
                np.dot(weights.T, np.dot(self.cov_matrix, weights))
            )
            return -(portfolio_return - self.risk_free_rate) / portfolio_vol
        
        constraints = [{'type': 'eq', 'fun': lambda w: np.sum(w) - 1}]
        init_weights = np.ones(self.n_assets) / self.n_assets
        
        result = minimize(
            neg_sharpe,
            init_weights,
            method='SLSQP',
            bounds=bounds,
            constraints=constraints
        )
        
        return result.x
```

### 3.2 行业/因子暴露约束

```python
    def factor_exposure_constraint(self, 
                                  factor_exposures: np.ndarray,
                                  target_exposure: np.ndarray,
                                  tolerance: float = 0.1) -> Dict:
        """
        因子暴露约束
        
        Args:
            factor_exposures: 因子暴露矩阵 (n_assets, n_factors)
            target_exposure: 目标因子暴露
            tolerance: 容忍度
            
        Returns:
            约束条件
        """
        constraints = []
        
        for i, target in enumerate(target_exposure):
            constraint = {
                'type': 'ineq',
                'fun': lambda w, i=i, target=target, tol=tolerance: 
                    tol - abs(np.dot(w, factor_exposures[:, i]) - target)
            }
            constraints.append(constraint)
        
        return constraints
    
    def sector_constraint(self, 
                         sector_mapping: Dict[str, str],
                         sector_limits: Dict[str, float]) -> Dict:
        """
        行业约束
        
        Args:
            sector_mapping: 资产-行业映射
            sector_limits: 行业权重上限
            
        Returns:
            约束条件
        """
        constraints = []
        
        for sector, limit in sector_limits.items():
            sector_assets = [
                i for i, asset in enumerate(self.asset_names) 
                if sector_mapping.get(asset) == sector
            ]
            
            if sector_assets:
                constraint = {
                    'type': 'ineq',
                    'fun': lambda w, assets=sector_assets, lim=limit: 
                        lim - np.sum(w[assets])
                }
                constraints.append(constraint)
        
        return constraints
    
    def constrained_max_sharpe(self, 
                              bounds: List[tuple],
                              extra_constraints: List[Dict] = None) -> np.ndarray:
        """
        约束优化-最大夏普比率
        
        Args:
            bounds: 权重边界
            extra_constraints: 额外约束
            
        Returns:
            权重向量
        """
        def neg_sharpe(weights):
            portfolio_return = np.dot(weights, self.mean_returns)
            portfolio_vol = np.sqrt(
                np.dot(weights.T, np.dot(self.cov_matrix, weights))
            )
            return -(portfolio_return - self.risk_free_rate) / portfolio_vol
        
        # 基础约束
        constraints = [{'type': 'eq', 'fun': lambda w: np.sum(w) - 1}]
        
        # 额外约束
        if extra_constraints:
            constraints.extend(extra_constraints)
        
        init_weights = np.ones(self.n_assets) / self.n_assets
        
        result = minimize(
            neg_sharpe,
            init_weights,
            method='SLSQP',
            bounds=bounds,
            constraints=constraints,
            options={'ftol': 1e-12, 'maxiter': 1000}
        )
        
        return result.x
```

### 3.3 换手率约束

```python
    def turnover_constraint(self, 
                           current_weights: np.ndarray,
                           max_turnover: float) -> Dict:
        """
        换手率约束
        
        Args:
            current_weights: 当前权重
            max_turnover: 最大换手率
            
        Returns:
            约束条件
        """
        constraint = {
            'type': 'ineq',
            'fun': lambda w: max_turnover - np.sum(np.abs(w - current_weights)) / 2
        }
        
        return constraint
    
    def transaction_cost_optimization(self, 
                                     current_weights: np.ndarray,
                                     transaction_cost: float = 0.001,
                                     max_turnover: float = 0.2) -> np.ndarray:
        """
        考虑交易成本的优化
        
        Args:
            current_weights: 当前权重
            transaction_cost: 交易成本
            max_turnover: 最大换手率
            
        Returns:
            权重向量
        """
        def objective_with_cost(weights):
            # 组合收益
            portfolio_return = np.dot(weights, self.mean_returns)
            
            # 组合风险
            portfolio_vol = np.sqrt(
                np.dot(weights.T, np.dot(self.cov_matrix, weights))
            )
            
            # 交易成本
            trades = np.abs(weights - current_weights)
            cost = np.sum(trades) * transaction_cost
            
            # 净夏普比率
            net_return = portfolio_return - cost
            sharpe = (net_return - self.risk_free_rate) / portfolio_vol
            
            return -sharpe
        
        # 约束条件
        constraints = [
            {'type': 'eq', 'fun': lambda w: np.sum(w) - 1},
            self.turnover_constraint(current_weights, max_turnover)
        ]
        
        bounds = tuple((0, 1) for _ in range(self.n_assets))
        init_weights = current_weights.copy()
        
        result = minimize(
            objective_with_cost,
            init_weights,
            method='SLSQP',
            bounds=bounds,
            constraints=constraints,
            options={'ftol': 1e-12, 'maxiter': 1000}
        )
        
        return result.x
```

```python
class MultiStrategyOptimizer:
    """多策略权重优化器"""
    
    def __init__(self, 
                 strategy_returns: pd.DataFrame,
                 strategy_categories: Dict[str, str] = None):
        """
        初始化
        
        Args:
            strategy_returns: 策略收益率
            strategy_categories: 策略分类
        """
        self.returns = strategy_returns
        self.n_strategies = strategy_returns.shape[1]
        self.strategy_names = strategy_returns.columns.tolist()
        self.categories = strategy_categories or {}
        
        self.mean_returns = strategy_returns.mean() * 252
        self.cov_matrix = strategy_returns.cov() * 252
        self.vols = np.sqrt(np.diag(self.cov_matrix))
    
    def optimize_by_sharpe(self, 
                          risk_free_rate: float = 0.03,
                          constraints: Dict = None) -> Dict:
        """
        最大夏普比率优化
        
        Args:
            risk_free_rate: 无风险利率
            constraints: 约束条件
            
        Returns:
            优化结果
        """
        def neg_sharpe(weights):
            portfolio_return = np.dot(weights, self.mean_returns)
            portfolio_vol = np.sqrt(
                np.dot(weights.T, np.dot(self.cov_matrix, weights))
            )
            return -(portfolio_return - risk_free_rate) / portfolio_vol
        
        # 基础约束
        constraint_list = [{'type': 'eq', 'fun': lambda w: np.sum(w) - 1}]
        
        # 额外约束
        if constraints:
            if 'max_weight' in constraints:
                bounds = tuple((0, constraints['max_weight']) for _ in range(self.n_strategies))
            else:
                bounds = tuple((0, 1) for _ in range(self.n_strategies))
            
            if 'category_limit' in constraints and self.categories:
                for category, limit in constraints['category_limit'].items():
                    cat_strategies = [
                        i for i, s in enumerate(self.strategy_names) 
                        if self.categories.get(s) == category
                    ]
                    if cat_strategies:
                        constraint_list.append({
                            'type': 'ineq',
                            'fun': lambda w, assets=cat_strategies, lim=limit: 
                                lim - np.sum(w[assets])
                        })
        else:
            bounds = tuple((0, 1) for _ in range(self.n_strategies))
        
        init_weights = np.ones(self.n_strategies) / self.n_strategies
        
        result = minimize(
            neg_sharpe,
            init_weights,
            method='SLSQP',
            bounds=bounds,
            constraints=constraint_list,
            options={'ftol': 1e-12, 'maxiter': 1000}
        )
        
        # 评估结果
        weights = result.x
        portfolio_return = np.dot(weights, self.mean_returns)
        portfolio_vol = np.sqrt(np.dot(weights.T, np.dot(self.cov_matrix, weights)))
        sharpe = (portfolio_return - risk_free_rate) / portfolio_vol
        
        return {
            'weights': dict(zip(self.strategy_names, weights)),
            'expected_return': portfolio_return,
            'volatility': portfolio_vol,
            'sharpe_ratio': sharpe,
            'optimization_success': result.success,
            'optimization_message': result.message
        }
    
    def optimize_by_risk_parity(self, 
                               risk_budget: Dict[str, float] = None) -> Dict:
        """
        风险平价优化
        
        Args:
            risk_budget: 风险预算
            
        Returns:
            优化结果
        """
        if risk_budget is None:
            budget = np.ones(self.n_strategies) / self.n_strategies
        else:
            budget = np.array([risk_budget.get(s, 1/self.n_strategies) 
                             for s in self.strategy_names])
            budget = budget / np.sum(budget)
        
        def risk_budget_objective(weights):
            portfolio_vol = np.sqrt(
                np.dot(weights.T, np.dot(self.cov_matrix, weights))
            )
            marginal_risk = np.dot(self.cov_matrix, weights) / portfolio_vol
            risk_contrib_pct = weights * marginal_risk / portfolio_vol
            return np.sum((risk_contrib_pct - budget) ** 2)
        
        constraints = [{'type': 'eq', 'fun': lambda w: np.sum(w) - 1}]
        bounds = tuple((0.001, 1) for _ in range(self.n_strategies))
        init_weights = np.ones(self.n_strategies) / self.n_strategies
        
        result = minimize(
            risk_budget_objective,
            init_weights,
            method='SLSQP',
            bounds=bounds,
            constraints=constraints,
            options={'ftol': 1e-12, 'maxiter': 1000}
        )
        
        weights = result.x
        portfolio_return = np.dot(weights, self.mean_returns)
        portfolio_vol = np.sqrt(np.dot(weights.T, np.dot(self.cov_matrix, weights)))
        
        # 风险贡献
        marginal_risk = np.dot(self.cov_matrix, weights) / portfolio_vol
        risk_contrib = weights * marginal_risk / portfolio_vol
        
        return {
            'weights': dict(zip(self.strategy_names, weights)),
            'expected_return': portfolio_return,
            'volatility': portfolio_vol,
            'risk_contribution': dict(zip(self.strategy_names, risk_contrib)),
            'optimization_success': result.success
        }
    
    def optimize_by_min_variance(self,

            return np.dot(weights.T, np.dot(self.cov_matrix, weights))

        weights = result.x
```

### 4.2 约束优化求解

```python
    def constrained_optimization(self, 
                                objective: str = 'sharpe',
                                constraints: Dict = None,
                                current_weights: np.ndarray = None) -> Dict:
        """
        约束优化
        
        Args:
            objective: 优化目标
            constraints: 约束条件字典
            current_weights: 当前权重（用于换手约束）
            
        Returns:
            优化结果
        """
        # 目标函数
        if objective == 'sharpe':
            def objective_func(weights):
                portfolio_return = np.dot(weights, self.mean_returns)
                portfolio_vol = np.sqrt(
                    np.dot(weights.T, np.dot(self.cov_matrix, weights))
                )
                return -portfolio_return / portfolio_vol
        
        elif objective == 'min_var':
            def objective_func(weights):
                return np.dot(weights.T, np.dot(self.cov_matrix, weights))
        
        elif objective == 'risk_parity':
            def objective_func(weights):
                portfolio_vol = np.sqrt(
                    np.dot(weights.T, np.dot(self.cov_matrix, weights))
                )
                marginal_risk = np.dot(self.cov_matrix, weights) / portfolio_vol
                risk_contrib_pct = weights * marginal_risk / portfolio_vol
                target = np.ones(self.n_strategies) / self.n_strategies
                return np.sum((risk_contrib_pct - target) ** 2)
        
        else:
            raise ValueError(f"Unknown objective: {objective}")
        
        # 构建约束
        constraint_list = [{'type': 'eq', 'fun': lambda w: np.sum(w) - 1}]
        
        bounds_list = [(0, 1) for _ in range(self.n_strategies)]
        
        if constraints:
            # 权重上下限
            if 'min_weight' in constraints or 'max_weight' in constraints:
                min_w = constraints.get('min_weight', 0)
                max_w = constraints.get('max_weight', 1)
                bounds_list = [(min_w, max_w) for _ in range(self.n_strategies)]
            
            # 特定资产权重约束
            if 'asset_weights' in constraints:
                for asset, (min_w, max_w) in constraints['asset_weights'].items():
                    if asset in self.strategy_names:
                        idx = self.strategy_names.index(asset)
                        bounds_list[idx] = (min_w, max_w)
            
            # 行业/类别约束
            if 'category_limit' in constraints and self.categories:
                for category, limit in constraints['category_limit'].items():
                    cat_strategies = [
                        i for i, s in enumerate(self.strategy_names) 
                        if self.categories.get(s) == category
                    ]
                    if cat_strategies:
                        constraint_list.append({
                            'type': 'ineq',
                            'fun': lambda w, assets=cat_strategies, lim=limit: 
                                lim - np.sum(w[assets])
                        })
            
            # 换手率约束
            if 'max_turnover' in constraints and current_weights is not None:
                max_turnover = constraints['max_turnover']
                constraint_list.append({
                    'type': 'ineq',
                    'fun': lambda w: max_turnover - np.sum(np.abs(w - current_weights)) / 2
                })
            
            # 目标收益约束
            if 'target_return' in constraints:
                target_return = constraints['target_return']
                constraint_list.append({
                    'type': 'eq',
                    'fun': lambda w, r=target_return: np.dot(w, self.mean_returns) - r
                })
            
            # 最大风险约束
            if 'max_volatility' in constraints:
                max_vol = constraints['max_volatility']
                constraint_list.append({
                    'type': 'ineq',
                    'fun': lambda w, v=max_vol: v - np.sqrt(np.dot(w.T, np.dot(self.cov_matrix, w)))
                })
        
        # 优化
        init_weights = current_weights if current_weights is not None else np.ones(self.n_strategies) / self.n_strategies
        
        result = minimize(
            objective_func,
            init_weights,
            method='SLSQP',
            bounds=bounds_list,
            constraints=constraint_list,
            options={'ftol': 1e-12, 'maxiter': 1000}
        )
        
        weights = result.x
        
        # 评估结果
        portfolio_return = np.dot(weights, self.mean_returns)
        portfolio_vol = np.sqrt(np.dot(weights.T, np.dot(self.cov_matrix, weights)))
        
        return {
            'weights': dict(zip(self.strategy_names, weights)),
            'expected_return': portfolio_return,
            'volatility': portfolio_vol,
            'sharpe_ratio': portfolio_return / portfolio_vol,
            'optimization_success': result.success,
            'optimization_message': result.message,
            'iterations': result.nit
        }
```

    def weight_stability_analysis(self, 
                                 n_bootstrap: int = 100,
                                 sample_ratio: float = 0.8) -> pd.DataFrame:
        """
        权重稳定性分析（Bootstrap方法）
        
        Args:
            n_bootstrap: Bootstrap次数
            sample_ratio: 样本比例
            
        Returns:
            权重稳定性统计
        """
        n_samples = int(len(self.returns) * sample_ratio)
        
        bootstrap_weights = {s: [] for s in self.strategy_names}
        
        for _ in range(n_bootstrap):
            # Bootstrap采样
            sample_idx = np.random.choice(
                len(self.returns), 
                n_samples, 
                replace=True
            )
            sample_returns = self.returns.iloc[sample_idx]
            
            # 计算统计量
            mean_returns = sample_returns.mean() * 252
            cov_matrix = sample_returns.cov() * 252
            
            # 优化
            def neg_sharpe(weights):
                portfolio_return = np.dot(weights, mean_returns)
                portfolio_vol = np.sqrt(np.dot(weights.T, np.dot(cov_matrix, weights)))
                return -portfolio_return / portfolio_vol
            
            constraints = [{'type': 'eq', 'fun': lambda w: np.sum(w) - 1}]
            bounds = tuple((0, 1) for _ in range(self.n_strategies))
            init_weights = np.ones(self.n_strategies) / self.n_strategies
            
            result = minimize(
                neg_sharpe,
                init_weights,
                method='SLSQP',
                bounds=bounds,
                constraints=constraints
            )
            
            if result.success:
                for i, strategy in enumerate(self.strategy_names):
                    bootstrap_weights[strategy].append(result.x[i])
        
        # 统计分析
        stability_stats = []
        for strategy in self.strategy_names:
            weights = bootstrap_weights[strategy]
            if weights:
                stability_stats.append({
                    'Strategy': strategy,
                    'Mean Weight': np.mean(weights),
                    'Std Weight': np.std(weights),
                    'Min Weight': np.min(weights),
                    'Max Weight': np.max(weights),
                    'Weight Range': np.max(weights) - np.min(weights),
                    'CV': np.std(weights) / np.mean(weights) if np.mean(weights) > 0 else np.inf,
                    'Stability': 'Stable' if np.std(weights) < 0.05 else 'Moderate' if np.std(weights) < 0.10 else 'Unstable'
                })
        
        return pd.DataFrame(stability_stats)
