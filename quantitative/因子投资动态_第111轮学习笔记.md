### 3.3 因子轮动策略实现

#### 动态因子权重模型

```python
class DynamicFactorRotation:
    """动态因子轮动策略"""
    
    def __init__(self, factors, lookback=60):
        self.factors = factors
        self.lookback = lookback
    
    def calculate_factor_momentum(self, factor_returns):
        """计算因子动量"""
        momentum = factor_returns.rolling(self.lookback).mean()
        return momentum
    
    def calculate_factor_volatility(self, factor_returns):
        """计算因子波动率"""
        volatility = factor_returns.rolling(self.lookback).std()
        return volatility
    
    def calculate_factor_crowding(self, factor_positions):
        """计算因子拥挤度"""
        # 基于持仓集中度、换手率、波动率综合计算
        crowding_score = factor_positions.rolling(20).std()
        return crowding_score
    
    def generate_weights(self, factor_returns, factor_positions):
        """生成动态权重"""
        # 动量信号
        momentum_signal = self.calculate_factor_momentum(factor_returns)
        
        # 波动率信号
        volatility_signal = 1 / self.calculate_factor_volatility(factor_returns)
        
        # 拥挤度信号
        crowding_signal = -self.calculate_factor_crowding(factor_positions)
        
        # 综合信号
        combined_signal = (momentum_signal + volatility_signal + crowding_signal) / 3
        
        # 归一化权重
        weights = combined_signal / combined_signal.sum()
        
        return weights
```

```python
def mean_variance_optimization(factor_returns, target_vol=0.15):
    """均值方差优化因子权重"""
    from scipy.optimize import minimize
    
    mu = factor_returns.mean()
    cov = factor_returns.cov()
    n = len(mu)
    
    def objective(w):
        portfolio_return = w @ mu
        portfolio_vol = np.sqrt(w @ cov @ w)
        # 最大化夏普比率
        return -portfolio_return / portfolio_vol
    
    constraints = [{'type': 'eq', 'fun': lambda w: np.sum(w) - 1}]
    bounds = [(0, 0.3) for _ in range(n)]  # 单因子权重上限30%
    
    result = minimize(objective, np.ones(n)/n, 
                     method='SLSQP', 
                     bounds=bounds, 
                     constraints=constraints)
    
    return result.x
```

**风险平价优化**：
```python
def risk_parity_optimization(factor_returns):
    """风险平价优化因子权重"""
    from scipy.optimize import minimize
    
    cov = factor_returns.cov()
    n = len(cov)
    
    def objective(w):
        # 计算每个因子的边际风险贡献
        portfolio_vol = np.sqrt(w @ cov @ w)
        marginal_contrib = cov @ w
        risk_contrib = w * marginal_contrib / portfolio_vol
        
        # 目标：使所有因子的风险贡献相等
        target_risk_contrib = portfolio_vol / n
        
        # 最小化风险贡献的方差
        return ((risk_contrib - target_risk_contrib) ** 2).sum()
    
    constraints = [{'type': 'eq', 'fun': lambda w: np.sum(w) - 1}]
    bounds = [(0, 0.5) for _ in range(n)]  # 单因子权重上限50%
    
    result = minimize(objective, np.ones(n)/n, 
                     method='SLSQP', 
                     bounds=bounds, 
                     constraints=constraints)
    
    return result.x
```

### 4.2 因子中性化处理

#### 行业中性化

```python
def industry_neutralize(factor_values, industry_dummies):
    """行业中性化：剔除因子中的行业暴露"""
    from sklearn.linear_model import LinearRegression
    
    # 用行业虚拟变量回归因子
    model = LinearRegression()
    model.fit(industry_dummies, factor_values)
    
    # 取残差作为中性化后的因子
    residuals = factor_values - model.predict(industry_dummies)
    
    return residuals
```

```python
def orthogonalize(target_factor, other_factors):
    """因子正交化：剔除目标因子中与其他因子的相关性"""
    from sklearn.linear_model import LinearRegression
    
    model = LinearRegression()
    model.fit(other_factors, target_factor)
    
    residuals = target_factor - model.predict(other_factors)
    
    return residuals
```
