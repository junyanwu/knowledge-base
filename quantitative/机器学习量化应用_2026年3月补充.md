## 二、特征工程最佳实践（补充）

### 2.1 特征工程流程标准化

```python
class FeatureEngineeringPipeline:
    """特征工程标准化流程"""
    
    def __init__(self):
        self.scalers = {}
        self.selected_features = None
    
    def fit_transform(self, df):
        """训练集特征工程"""
        # Step 1: 数据清洗
        df = self.clean_data(df)
        
        # Step 2: 特征构造
        features = self.construct_features(df)
        
        # Step 3: 特征选择
        features, selected = self.select_features(features, df['target'])
        
        # Step 4: 特征标准化
        features = self.standardize(features, fit=True)
        
        return features
    
    def transform(self, df):
        """测试集特征工程（避免数据泄露）"""
        df = self.clean_data(df)
        features = self.construct_features(df)
        features = features[self.selected_features]
        features = self.standardize(features, fit=False)
        return features
    
    def clean_data(self, df):
        """数据清洗"""
        # 缺失值处理
        df = df.fillna(method='ffill')
        
        # 极值处理
        df = df.clip(lower=df.quantile(0.01), upper=df.quantile(0.99), axis=1)
        
        return df
    
    def construct_features(self, df):
        """特征构造"""
        features = pd.DataFrame()
        
        # 价量特征
        price_feats = price_features(df)
        volume_feats = volume_features(df)
        
        # 技术指标
        tech_feats = technical_indicators(df)
        
        # 财务特征
        fin_feats = financial_features(df)
        
        # 合并
        features = pd.concat([price_feats, volume_feats, tech_feats, fin_feats], axis=1)
        
        return features
    
    def select_features(self, X, y, top_k=50):
        """特征选择"""
        # 随机森林特征重要性
        rf = RandomForestRegressor(n_estimators=100, random_state=42, n_jobs=-1)
        rf.fit(X, y)
        
        importance = pd.Series(rf.feature_importances_, index=X.columns)
        selected = importance.nlargest(top_k).index.tolist()
        
        self.selected_features = selected
        return X[selected], selected
    
    def standardize(self, X, fit=True):
        """特征标准化"""
        from sklearn.preprocessing import RobustScaler
        
        if fit:
            self.scaler = RobustScaler()
            return self.scaler.fit_transform(X)
        else:
            return self.scaler.transform(X)
```

### 2.2 避免常见陷阱

**错误示例**：


### 2.3 LLM驱动的自动化因子挖掘

**R&D-Agent框架（微软，NeurIPS 2025）**：

```python
class RD-Agent-Quant:
    """R&D-Agent: 自动化因子挖掘"""
    
    def __init__(self, llm_model):
        self.llm = llm_model  # GPT-4/Claude等
        self.backtester = Backtester()
        self.code_generator = CodeGenerator()
    
    def research_phase(self, domain_knowledge):
        """研究阶段：生成因子假设"""
        prompt = f"""
        你是一个量化研究员。基于以下市场知识和数据，生成新的因子假设：
        
        市场知识：
        {domain_knowledge}
        
        可用数据：
        - 价量数据：开盘价、收盘价、最高价、最低价、成交量
        - 财务数据：PE、PB、ROE、营收增长率等
        - 另类数据：新闻情感、分析师评级
        
        要求：
        1. 因子逻辑清晰，有经济学意义
        2. 可量化实现，避免未来函数
        3. 与常见因子相关性<0.3
        
        输出格式：
        - 因子名称
        - 因子逻辑（为什么有效）
        - 实现伪代码
        - 预期IC范围
        """
        
        hypotheses = self.llm.generate(prompt)
        return hypotheses
    
    def development_phase(self, hypothesis):
        """开发阶段：代码生成"""
        code = self.code_generator.generate(hypothesis)
        return code
    
    def feedback_phase(self, factor_code):
        """反馈阶段：回测评估"""
        result = self.backtester.run(factor_code)
        
        # 评估指标
        metrics = {
            'ic': result['ic'],
            'icir': result['icir'],
            'sharpe': result['sharpe'],
            'turnover': result['turnover']
        }
        
        # 多臂老虎机调度
        if metrics['ic'] > 0.05 and metrics['sharpe'] > 1.5:
            return "Accept", metrics
        else:
            return "Reject", metrics
    
    def iterative_optimization(self, domain_knowledge, n_iterations=100):
        """迭代优化"""
        best_factors = []
        
        for i in range(n_iterations):
            # 生成假设
            hypothesis = self.research_phase(domain_knowledge)
            
            # 实现代码
            code = self.development_phase(hypothesis)
            
            # 回测反馈
            decision, metrics = self.feedback_phase(code)
            
            if decision == "Accept":
                best_factors.append({
                    'hypothesis': hypothesis,
                    'code': code,
                    'metrics': metrics
                })
        
        return best_factors
```

```python
def strict_oos_test(X, y, n_splits=5):
    """严格的时间序列交叉验证"""
    from sklearn.model_selection import TimeSeriesSplit
    
    tscv = TimeSeriesSplit(n_splits=n_splits)
    
    train_scores = []
    test_scores = []
    
    for train_idx, test_idx in tscv.split(X):
        X_train, X_test = X.iloc[train_idx], X.iloc[test_idx]
        y_train, y_test = y.iloc[train_idx], y.iloc[test_idx]
        
        # 训练模型
        model = XGBRegressor().fit(X_train, y_train)
        
        # 评估
        train_score = model.score(X_train, y_train)
        test_score = model.score(X_test, y_test)
        
        train_scores.append(train_score)
        test_scores.append(test_score)
        
        print(f"Fold {len(train_scores)}: Train={train_score:.4f}, Test={test_score:.4f}")
    
    # 检查过拟合
    avg_train = np.mean(train_scores)
    avg_test = np.mean(test_scores)
    gap = avg_train - avg_test
    
    print(f"\n平均训练得分: {avg_train:.4f}")
    print(f"平均测试得分: {avg_test:.4f}")
    print(f"过拟合Gap: {gap:.4f}")
    
    if gap > 0.1:
        print("⚠️ 警告：可能存在过拟合！")
    
    return train_scores, test_scores
```

```python
def parameter_stability_test(X, y, param_ranges):
    """参数稳定性检验"""
    results = []
    
    for param_name, param_values in param_ranges.items():
        for value in param_values:
            # 不同时间段测试
            time_periods = [
                ('2010-2015', '2010-01-01', '2015-12-31'),
                ('2016-2020', '2016-01-01', '2020-12-31'),
                ('2021-2025', '2021-01-01', '2025-12-31')
            ]
            
            period_ics = []
            for period_name, start, end in time_periods:
                mask = (X.index >= start) & (X.index <= end)
                X_period = X[mask]
                y_period = y[mask]
                
                # 训练模型
                model = XGBRegressor(**{param_name: value})
                model.fit(X_period, y_period)
                
                # 预测IC
                pred = model.predict(X_period)
                ic = spearmanr(pred, y_period)[0]
                period_ics.append(ic)
            
            # 计算IC稳定性
            ic_mean = np.mean(period_ics)
            ic_std = np.std(period_ics)
            
            results.append({
                'param': param_name,
                'value': value,
                'ic_mean': ic_mean,
                'ic_std': ic_std,
                'stability': ic_mean / (ic_std + 1e-6)  # 夏普比类似
            })
    
    # 选择稳定性最好的参数
    results_df = pd.DataFrame(results)
    best_params = results_df.loc[results_df['stability'].idxmax()]
    
    return best_params
```

```python
def walk_forward_analysis(X, y, train_window=252, test_window=63):
    """Walk-Forward分析"""
    results = []
    
    for i in range(train_window, len(X) - test_window, test_window):
        # 训练窗口
        X_train = X.iloc[i-train_window:i]
        y_train = y.iloc[i-train_window:i]
        
        # 测试窗口
        X_test = X.iloc[i:i+test_window]
        y_test = y.iloc[i:i+test_window]
        
        # 训练模型
        model = XGBRegressor().fit(X_train, y_train)
        
        # 预测
        pred = model.predict(X_test)
        
        # 评估IC
        ic = spearmanr(pred, y_test)[0]
        results.append(ic)
    
    # 统计
    print(f"平均IC: {np.mean(results):.4f}")
    print(f"IC标准差: {np.std(results):.4f}")
    print(f"ICIR: {np.mean(results) / np.std(results):.4f}")
    print(f"IC胜率: {(np.array(results) > 0).mean():.2%}")
    
    return results
```

class FinRL_DeepSeek:
