class MonteCarloStressTester:
    """蒙特卡洛压力测试器"""
    
    def __init__(self,
                 returns: pd.DataFrame,
                 n_simulations: int = 10000,
                 time_horizon: int = 252):
        """
        初始化
        
        Args:
            returns: 历史收益率
            n_simulations: 模拟次数
            time_horizon: 时间跨度（天数）
        """
        self.returns = returns
        self.n_simulations = n_simulations
        self.time_horizon = time_horizon
        
    def fit_distribution(self,
                          distribution: str = 't') -> Dict:
        """
        拟合收益率分布
        
        Args:
            distribution: 分布类型 ('normal', 't', 'skew_t')
            
        Returns:
            分布参数
        """
        params = {}
        
        for col in self.returns.columns:
            if distribution == 'normal':
                mu, std = norm.fit(self.returns[col])
                params[col] = {'mu': mu, 'std': std, 'dist': 'normal'}
            
            elif distribution == 't':
                df, mu, std = t.fit(self.returns[col])
                params[col] = {'df': df, 'mu': mu, 'std': std, 'dist': 't'}
            
            elif distribution == 'skew_t':
                # 使用偏态t分布（需要scipy.stats.jf_skew_t或类似）
                # 简化处理：使用t分布
                df, mu, std = t.fit(self.returns[col])
                params[col] = {'df': df, 'mu': mu, 'std': std, 'dist': 't'}
        
        return params
    
    def simulate_paths(self,
                        params: Dict,
                        correlation_matrix: np.ndarray = None) -> np.ndarray:
        """
        模拟价格路径
        
        Args:
            params: 分布参数
            correlation_matrix: 相关性矩阵
            
        Returns:
            模拟路径 (n_simulations, time_horizon, n_assets)
        """
        n_assets = len(params)
        
        # 生成相关性随机数
        if correlation_matrix is None:
            correlation_matrix = self.returns.corr().values
        
        L = np.linalg.cholesky(correlation_matrix)
        
        # 模拟路径
        paths = np.zeros((self.n_simulations, self.time_horizon, n_assets))
        
        for i, (asset, param) in enumerate(params.items()):
            # 生成独立随机数
            if param['dist'] == 'normal':
                z = np.random.randn(self.n_simulations, self.time_horizon)
                returns_sim = param['mu'] + param['std'] * z
            
            elif param['dist'] == 't':
                z = np.random.randn(self.n_simulations, self.time_horizon)
                chi2 = np.random.chisquare(param['df'], (self.n_simulations, self.time_horizon))
                t_samples = z / np.sqrt(chi2 / param['df'])
                returns_sim = param['mu'] + param['std'] * t_samples
            
            paths[:, :, i] = returns_sim
        
        # 应用相关性
        for t in range(self.time_horizon):
            paths[:, t, :] = paths[:, t, :] @ L.T
        
        return paths
    
    def stress_test_portfolio(self,
                               weights: np.ndarray,
                               paths: np.ndarray,
                               var_levels: List[float] = [0.01, 0.05, 0.10]) -> Dict:
        """
        压力测试组合
        
        Args:
            weights: 组合权重
            paths: 模拟路径
            var_levels: VaR置信水平
            
        Returns:
            压力测试结果
        """
        # 计算组合收益
        portfolio_returns = np.sum(paths * weights, axis=2)
        
        # 计算累计收益
        cumulative_returns = np.prod(1 + portfolio_returns, axis=1) - 1
        
        # VaR
        var_results = {}
        for level in var_levels:
            var_results[f'VaR_{int(level*100)}'] = np.percentile(cumulative_returns, level * 100)
        
        # CVaR (Expected Shortfall)
        cvar_results = {}
        for level in var_levels:
            var = np.percentile(cumulative_returns, level * 100)
            cvar_results[f'CVaR_{int(level*100)}'] = cumulative_returns[cumulative_returns <= var].mean()
        
        # 最大回撤分布
        max_drawdowns = np.zeros(self.n_simulations)
        for i in range(self.n_simulations):
            cumulative = np.cumprod(1 + portfolio_returns[i, :])
            running_max = np.maximum.accumulate(cumulative)
            drawdown = (cumulative - running_max) / running_max
            max_drawdowns[i] = drawdown.min()
        
        # 极端损失概率
        extreme_loss_thresholds = [-0.30, -0.40, -0.50, -0.60]
        extreme_loss_probs = {}
        for threshold in extreme_loss_thresholds:
            prob = np.mean(cumulative_returns < threshold)
            extreme_loss_probs[f'P(Loss<{threshold:.0%})'] = prob
        
        # 恢复期分析
        recovery_periods = np.zeros(self.n_simulations)
        for i in range(self.n_simulations):
            cumulative = np.cumprod(1 + portfolio_returns[i, :])
            min_point = np.argmin(cumulative)
            if min_point < len(cumulative) - 1:
                # 找到恢复点
                recovery_idx = np.where(cumulative[min_point:] >= 1)[0]
                if len(recovery_idx) > 0:
                    recovery_periods[i] = recovery_idx[0]
                else:
                    recovery_periods[i] = self.time_horizon - min_point
            else:
                recovery_periods[i] = 0
        
        return {
            'cumulative_returns': cumulative_returns,
            'var': var_results,
            'cvar': cvar_results,
            'max_drawdown_distribution': max_drawdowns,
            'extreme_loss_probabilities': extreme_loss_probs,
            'recovery_periods': recovery_periods,
            'mean_cumulative_return': cumulative_returns.mean(),
            'std_cumulative_return': cumulative_returns.std()
        }
    
    def plot_stress_test_results(self,

class ReverseStressTester:
    """反向压力测试"""
    
    def __init__(self,
                 returns: pd.DataFrame,
                 portfolio_value: float = 1000000,
                 loss_threshold: float = 0.30):
        """
        初始化
        
        Args:
            returns: 历史收益率
            portfolio_value: 组合价值
            loss_threshold: 损失阈值
        """
        self.returns = returns
        self.portfolio_value = portfolio_value
        self.loss_threshold = loss_threshold
        
    def identify_failure_scenarios(self,
                                     n_scenarios: int = 100) -> List[Dict]:
        """
        识别导致超过损失阈值的场景
        
        Args:
            n_scenarios: 场景数量
            
        Returns:
            失败场景列表
        """
        failure_scenarios = []
        
        # 计算导致失败所需的收益冲击
        for _ in range(n_scenarios):
            # 随机生成场景
            scenario = self._generate_random_scenario()
            
            # 计算该场景下的损失
            loss = self._calculate_scenario_loss(scenario)
            
            if loss > self.loss_threshold:
                scenario['loss'] = loss
                failure_scenarios.append(scenario)
        
        # 按损失排序
        failure_scenarios.sort(key=lambda x: x['loss'], reverse=True)
        
        return failure_scenarios
    
    def _generate_random_scenario(self) -> Dict:
        """生成随机场景"""
        n_assets = len(self.returns.columns)
        
        scenario = {
            'shock_type': np.random.choice(['market', 'sector', 'idiosyncratic']),
            'shock_magnitude': np.random.uniform(-0.50, 0.10, n_assets),
            'vol_spike': np.random.uniform(1.0, 5.0),
            'correlation_spike': np.random.uniform(0, 0.3),
            'duration': np.random.randint(5, 90)
        }
        
        return scenario
    
    def _calculate_scenario_loss(self, scenario: Dict) -> float:
        """计算场景损失"""
        # 简化：直接使用冲击幅度
        loss = -scenario['shock_magnitude'].mean()
        
        # 考虑持续时间和波动率飙升的影响
        duration_factor = 1 + (scenario['duration'] / 90) * 0.2
        vol_factor = 1 + (scenario['vol_spike'] - 1) * 0.1
        
        adjusted_loss = loss * duration_factor * vol_factor
        
        return adjusted_loss
    
    def analyze_failure_drivers(self,
                                  failure_scenarios: List[Dict]) -> Dict:
        """
        分析失败驱动因素
        
        Args:
            failure_scenarios: 失败场景
            
        Returns:
            驱动因素分析
        """
        # 统计场景类型分布
        shock_types = [s['shock_type'] for s in failure_scenarios]
        shock_type_dist = pd.Series(shock_types).value_counts(normalize=True)
        
        # 统计冲击幅度
        shock_magnitudes = np.array([s['shock_magnitude'] for s in failure_scenarios])
        mean_shock = shock_magnitudes.mean(axis=0)
        
        # 统计持续时间
        durations = [s['duration'] for s in failure_scenarios]
        mean_duration = np.mean(durations)
        
        # 统计波动率飙升
        vol_spikes = [s['vol_spike'] for s in failure_scenarios]
        mean_vol_spike = np.mean(vol_spikes)
        
        return {
            'shock_type_distribution': shock_type_dist.to_dict(),
            'mean_shock_by_asset': dict(zip(self.returns.columns, mean_shock)),
            'mean_duration': mean_duration,
            'mean_volatility_spike': mean_vol_spike,
            'most_common_failure_type': shock_type_dist.index[0] if len(shock_type_dist) > 0 else None
        }

class DrawdownCapacityAnalyzer:
    """回撤承受能力分析器"""
    
    def __init__(self,
                 initial_capital: float = 1000000,
                 risk_tolerance: str = 'moderate'):
        """
        初始化
        
        Args:
            initial_capital: 初始资金
            risk_tolerance: 风险承受能力 ('conservative', 'moderate', 'aggressive')
        """
        self.initial_capital = initial_capital
        self.risk_tolerance = risk_tolerance
        
        # 风险承受能力参数
        self.risk_params = {
            'conservative': {
                'max_drawdown': 0.15,
                'stop_loss': 0.10,
                'position_limit': 0.60
            },
            'moderate': {
                'max_drawdown': 0.25,
                'stop_loss': 0.15,
                'position_limit': 0.80
            },
            'aggressive': {
                'max_drawdown': 0.35,
                'stop_loss': 0.25,
                'position_limit': 1.00
            }
        }
        
    def calculate_capacity_metrics(self,
                                    portfolio_returns: pd.Series) -> Dict:
        """
        计算回撤承受能力指标
        
        Args:
            portfolio_returns: 组合收益率
            
        Returns:
            承受能力指标
        """
        # 计算净值曲线
        nav = (1 + portfolio_returns).cumprod()
        
        # 计算回撤序列
        running_max = nav.cummax()
        drawdown = (nav - running_max) / running_max
        
        # 最大回撤
        max_dd = drawdown.min()
        
        # 回撤持续时间
        dd_duration = self._calculate_drawdown_duration(drawdown)
        
        # 回撤恢复时间
        recovery_time = self._calculate_recovery_time(nav, drawdown)
        
        # 回撤频率
        dd_frequency = self._calculate_drawdown_frequency(drawdown)
        
        # 回撤深度分布
        dd_depth_distribution = self._calculate_depth_distribution(drawdown)
        
        # 风险承受能力评估
        capacity_score = self._assess_capacity(max_dd, dd_duration, dd_frequency)
        
        return {
            'max_drawdown': max_dd,
            'avg_drawdown_duration': dd_duration['mean'],
            'max_drawdown_duration': dd_duration['max'],
            'avg_recovery_time': recovery_time['mean'],
            'max_recovery_time': recovery_time['max'],
            'drawdown_frequency': dd_frequency,
            'dd_depth_distribution': dd_depth_distribution,
            'capacity_score': capacity_score,
            'risk_params': self.risk_params[self.risk_tolerance]
        }
    
    def _calculate_drawdown_duration(self, drawdown: pd.Series) -> Dict:
        """计算回撤持续时间"""
        # 找到回撤期
        in_drawdown = drawdown < 0
        
        # 计算每个回撤期的持续时间
        durations = []
        current_duration = 0
        
        for is_dd in in_drawdown:
            if is_dd:
                current_duration += 1
            else:
                if current_duration > 0:
                    durations.append(current_duration)
                    current_duration = 0
        
        if len(durations) == 0:
            durations = [0]
        
        return {
            'mean': np.mean(durations),
            'max': np.max(durations),
            'median': np.median(durations)
        }
    
    def _calculate_recovery_time(self, nav: pd.Series, drawdown: pd.Series) -> Dict:
        """计算恢复时间"""
        recovery_times = []
        
        # 找到回撤谷底
        dd_peaks = drawdown[drawdown == drawdown.min()].index
        
        for peak_idx in dd_peaks:
            # 找到下一个恢复点
            peak_loc = nav.index.get_loc(peak_idx)
            peak_value = nav.iloc[peak_loc]
            
            # 向后查找恢复
            recovered = False
            for i in range(peak_loc + 1, len(nav)):
                if nav.iloc[i] >= peak_value:
                    recovery_times.append(i - peak_loc)
                    recovered = True
                    break
            
            if not recovered:
                recovery_times.append(len(nav) - peak_loc)
        
        if len(recovery_times) == 0:
            recovery_times = [0]
        
        return {
            'mean': np.mean(recovery_times),
            'max': np.max(recovery_times),
            'median': np.median(recovery_times)
        }
    
    def _calculate_drawdown_frequency(self, drawdown: pd.Series, threshold: float = -0.10) -> int:
        """计算回撤频率（超过阈值的次数）"""
        in_drawdown = drawdown < threshold
        
        # 计算回撤次数
        dd_count = 0
        prev_dd = False
        
        for is_dd in in_drawdown:
            if is_dd and not prev_dd:
                dd_count += 1
            prev_dd = is_dd
        
        return dd_count
    
    def _calculate_depth_distribution(self, drawdown: pd.Series) -> Dict:
        """计算回撤深度分布"""
        # 仅考虑回撤期
        dd_periods = drawdown[drawdown < 0]
        
        if len(dd_periods) == 0:
            return {
                'mean': 0,
                'median': 0,
                'std': 0,
                'q25': 0,
                'q75': 0
            }
        
        return {
            'mean': dd_periods.mean(),
            'median': dd_periods.median(),
            'std': dd_periods.std(),
            'q25': dd_periods.quantile(0.25),
            'q75': dd_periods.quantile(0.75)
        }
    
    def _assess_capacity(self,
                          max_dd: float,
                          dd_duration: Dict,
                          dd_frequency: int) -> Dict:
        """评估风险承受能力"""
        params = self.risk_params[self.risk_tolerance]
        
        # 评估指标
        dd_score = 1 - (abs(max_dd) / params['max_drawdown'])
        dd_score = max(0, min(1, dd_score))
        
        duration_score = 1 - (dd_duration['mean'] / 60)  # 假设60天为阈值
        duration_score = max(0, min(1, duration_score))
        
        frequency_score = 1 - (dd_frequency / 5)  # 假设5次为阈值
        frequency_score = max(0, min(1, frequency_score))
        
        # 综合评分
        overall_score = 0.5 * dd_score + 0.3 * duration_score + 0.2 * frequency_score
        
        # 能力等级
        if overall_score >= 0.8:
            capacity_level = '优秀'
        elif overall_score >= 0.6:
            capacity_level = '良好'
        elif overall_score >= 0.4:
            capacity_level = '一般'
        else:
            capacity_level = '不足'
        
        return {
            'overall_score': overall_score,
            'dd_score': dd_score,
            'duration_score': duration_score,
            'frequency_score': frequency_score,
            'capacity_level': capacity_level,
            'recommendation': self._get_recommendation(overall_score)
        }
    
    def _get_recommendation(self, score: float) -> str:
        """获取建议"""
        if score >= 0.8:
            return "风险承受能力优秀，可考虑适当提高风险敞口"
        elif score >= 0.6:
            return "风险承受能力良好，维持当前配置"
        elif score >= 0.4:
            return "风险承受能力一般，建议降低风险敞口或增加对冲"
        else:
            return "风险承受能力不足，需大幅降低风险敞口"
