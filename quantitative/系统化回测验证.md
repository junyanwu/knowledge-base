```python
class SystematicBacktestValidator:
    """系统化回测验证器"""
    
    def __init__(self,

    def classify_market_regime(self, 
                               prices: pd.DataFrame,
                               window: int = 120) -> Dict[str, pd.Series]:
        """
        分类市场环境
        
        Args:
            prices: 价格数据
            window: 观察窗口
            
        Returns:
            市场环境分类
        """
        # 使用沪深300作为基准
        benchmark = prices['510300']
        
        # 计算收益率
        returns = benchmark.pct_change(window)
        
        # 计算波动率
        volatility = benchmark.pct_change().rolling(window).std() * np.sqrt(252)
        
        # ADX趋势强度
        adx = self._calculate_adx(benchmark, window)
        
        # 分类规则
        regimes = {}
        
        # 牛市：收益率>10%且ADX>25
        bull_mask = (returns > 0.10) & (adx > 25)
        regimes['bull'] = bull_mask
        
        # 熊市：收益率<-10%且ADX>25
        bear_mask = (returns < -0.10) & (adx > 25)
        regimes['bear'] = bear_mask
        
        # 震荡市：其他情况
        sideways_mask = ~(bull_mask | bear_mask)
        regimes['sideways'] = sideways_mask
        
        self.market_regimes = regimes
        
        return regimes
    
    def _calculate_adx(self, 
                       prices: pd.Series,
                       window: int = 14) -> pd.Series:
        """计算ADX"""
        high = prices.rolling(window).max()
        low = prices.rolling(window).min()
        close = prices
        
        # +DM和-DM
        up_move = high - high.shift(1)
        down_move = low.shift(1) - low
        
        plus_dm = np.where((up_move > down_move) & (up_move > 0), up_move, 0)
        minus_dm = np.where((down_move > up_move) & (down_move > 0), down_move, 0)
        
        # TR
        tr1 = high - low
        tr2 = abs(high - close.shift(1))
        tr3 = abs(low - close.shift(1))
        tr = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
        
        # 平滑
        atr = tr.rolling(window).mean()
        plus_di = 100 * pd.Series(plus_dm).rolling(window).mean() / atr
        minus_di = 100 * pd.Series(minus_dm).rolling(window).mean() / atr
        
        # DX和ADX
        dx = 100 * abs(plus_di - minus_di) / (plus_di + minus_di)
        adx = dx.rolling(window).mean()
        
        return adx
    
    def run_full_sample_backtest(self,
                                  prices: pd.DataFrame,
                                  strategy_name: str,
                                  **kwargs) -> Dict:
        """
        运行全样本回测
        
        Args:
            prices: 价格数据
            strategy_name: 策略名称
            **kwargs: 策略参数
            
        Returns:
            回测结果
        """
        strategy_func = self.strategies[strategy_name]
        
        # 执行策略
        signals = strategy_func(prices, **kwargs)
        
        # 计算仓位
        positions = signals.shift(1).fillna(0)
        
        # 计算收益
        returns = prices.pct_change()
        strategy_returns = (positions * returns).sum(axis=1)
        
        # 计算净值
        nav = (1 + strategy_returns).cumprod()
        
        # 性能指标
        performance = self._calculate_performance(strategy_returns)
        
        # 分市场环境表现
        regime_performance = self._calculate_regime_performance(
            strategy_returns, 
            self.market_regimes
        )
        
        result = {
            'nav': nav,
            'returns': strategy_returns,
            'positions': positions,
            'performance': performance,
            'regime_performance': regime_performance,
            'signals': signals
        }
        
        self.backtest_results[strategy_name] = result
        
        return result
    
    def _cta_trend_strategy(self,
                            prices: pd.DataFrame,
                            short_window: int = 20,
                            long_window: int = 50,
                            adx_threshold: float = 25) -> pd.DataFrame:
        """CTA趋势跟踪策略"""
        signals = pd.DataFrame(index=prices.index, columns=prices.columns, data=0.0)
        
        for col in prices.columns:
            # EMA趋势
            ema_short = prices[col].ewm(span=short_window).mean()
            ema_long = prices[col].ewm(span=long_window).mean()
            
            # ADX强度
            adx = self._calculate_adx(prices[col])
            
            # 信号
            trend_signal = np.where(ema_short > ema_long, 1, -1)
            strength_filter = (adx > adx_threshold).astype(int)
            
            signals[col] = trend_signal * strength_filter
        
        # 标准化
        signals = signals.div(signals.abs().sum(axis=1), axis=0)
        signals = signals.fillna(0)
        
        return signals
    
    def _mean_reversion_strategy(self,
                                  prices: pd.DataFrame,
                                  window: int = 20,
                                  entry_z: float = 2.0,
                                  exit_z: float = 0.5) -> pd.DataFrame:
        """均值回归策略"""
        signals = pd.DataFrame(index=prices.index, columns=prices.columns, data=0.0)
        
        for col in prices.columns:
            # 计算z-score
            mean = prices[col].rolling(window).mean()
            std = prices[col].rolling(window).std()
            z_score = (prices[col] - mean) / std
            
            # 生成信号
            signals[col] = np.where(z_score < -entry_z, 1,
                                   np.where(z_score > entry_z, -1,
                                           np.where(abs(z_score) < exit_z, 0, np.nan)))
        
        signals = signals.ffill().fillna(0)
        signals = signals.div(signals.abs().sum(axis=1).replace(0, 1), axis=0)
        
        return signals
    
    def _grid_trading_strategy(self,
                                prices: pd.DataFrame,
                                grid_num: int = 10,
                                grid_spacing: float = 0.02) -> pd.DataFrame:
        """网格交易策略"""
        signals = pd.DataFrame(index=prices.index, columns=prices.columns, data=0.0)
        
        for col in prices.columns:
            # 建立网格
            base_price = prices[col].iloc[0]
            grid_levels = np.linspace(
                base_price * (1 - grid_num * grid_spacing),
                base_price * (1 + grid_num * grid_spacing),
                2 * grid_num + 1
            )
            
            # 根据价格位置生成仓位
            position = 0
            positions_list = []
            
            for price in prices[col]:
                # 网格交易逻辑
                if len(positions_list) > 0:
                    prev_position = positions_list[-1]
                else:
                    prev_position = 0
                
                # 计算当前网格位置
                grid_idx = np.searchsorted(grid_levels, price)
                target_position = (grid_idx - grid_num) / grid_num
                
                # 平滑仓位变化
                position = 0.7 * prev_position + 0.3 * target_position
                positions_list.append(position)
            
            signals[col] = positions_list
        
        signals = signals.div(signals.abs().sum(axis=1).replace(0, 1), axis=0)
        
        return signals
    
    def _volatility_strategy(self,
                              prices: pd.DataFrame,
                              vol_window: int = 20,
                              target_vol: float = 0.15) -> pd.DataFrame:
        """波动率策略"""
        returns = prices.pct_change()
        
        # 计算波动率
        rolling_vol = returns.rolling(vol_window).std() * np.sqrt(252)
        
        # 波动率倒数加权
        inv_vol = 1 / rolling_vol
        weights = inv_vol.div(inv_vol.sum(axis=1), axis=0)
        
        # 动态调整仓位以维持目标波动率
        current_vol = (returns * weights).sum(axis=1).rolling(vol_window).std() * np.sqrt(252)
        leverage = target_vol / current_vol
        leverage = leverage.clip(0.5, 2.0)  # 限制杠杆范围
        
        signals = weights.multiply(leverage, axis=0)
        
        return signals
    
    def _pairs_trading_strategy(self,
                                 prices: pd.DataFrame,
                                 window: int = 60,
                                 entry_threshold: float = 2.0) -> pd.DataFrame:
        """配对交易策略"""
        signals = pd.DataFrame(index=prices.index, columns=prices.columns, data=0.0)
        
        # 寻找配对
        returns = prices.pct_change()
        corr_matrix = returns.corr()
        
        # 选择高相关性配对
        pairs = []
        for i in range(len(prices.columns)):
            for j in range(i+1, len(prices.columns)):
                if corr_matrix.iloc[i, j] > 0.7:
                    pairs.append((prices.columns[i], prices.columns[j]))
        
        # 对每个配对生成信号
        for pair in pairs:
            asset1, asset2 = pair
            
            # 计算价差
            spread = np.log(prices[asset1]) - np.log(prices[asset2])
            
            # 标准化
            mean = spread.rolling(window).mean()
            std = spread.rolling(window).std()
            z_score = (spread - mean) / std
            
            # 生成信号
            signals[asset1] += np.where(z_score > entry_threshold, -1,
                                       np.where(z_score < -entry_threshold, 1, 0))
            signals[asset2] += np.where(z_score > entry_threshold, 1,
                                       np.where(z_score < -entry_threshold, -1, 0))
        
        # 标准化
        signals = signals.div(signals.abs().sum(axis=1).replace(0, 1), axis=0)
        
        return signals
    
    def _options_enhanced_strategy(self,
                                    prices: pd.DataFrame,
                                    call_strike_pct: float = 1.02,
                                    put_strike_pct: float = 0.98) -> pd.DataFrame:
        """期权增强策略（备兑看涨）"""
        signals = pd.DataFrame(index=prices.index, columns=prices.columns, data=0.0)
        
        returns = prices.pct_change()
        
        # 基础持仓
        base_position = 0.6
        
        # 波动率估计
        vol = returns.rolling(20).std() * np.sqrt(252)
        
        for col in prices.columns:
            # 根据波动率调整持仓
            vol_adj_position = base_position * (0.20 / vol[col])
            vol_adj_position = vol_adj_position.clip(0.3, 0.8)
            
            # 趋势判断
            trend = prices[col].pct_change(60)
            
            # 高波动且上涨时减仓（模拟备兑看涨）
            signals[col] = np.where(
                (vol[col] > 0.20) & (trend > 0),
                vol_adj_position * 0.8,
                vol_adj_position
            )
        
        # 标准化
        signals = signals.div(signals.abs().sum(axis=1).replace(0, 1), axis=0)
        
        return signals
    
    def _calculate_performance(self, 
                               returns: pd.Series) -> Dict:
        """计算性能指标"""
        # 基本统计
        total_return = (1 + returns).prod() - 1
        annual_return = (1 + returns).prod() ** (252 / len(returns)) - 1
        daily_return_mean = returns.mean()
        daily_return_std = returns.std()
        
        # 风险指标
        annual_vol = returns.std() * np.sqrt(252)
        downside_returns = returns[returns < 0]
        downside_std = downside_returns.std() * np.sqrt(252)
        
        # 风险调整收益
        sharpe_ratio = (annual_return - 0.03) / annual_vol if annual_vol > 0 else 0
        sortino_ratio = (annual_return - 0.03) / downside_std if downside_std > 0 else 0
        
        # 回撤分析
        cumulative = (1 + returns).cumprod()
        running_max = cumulative.cummax()
        drawdown = (cumulative - running_max) / running_max
        max_drawdown = drawdown.min()
        
        # 卡玛比率
        calmar_ratio = annual_return / abs(max_drawdown) if max_drawdown != 0 else 0
        
        # 胜率
        win_rate = (returns > 0).sum() / len(returns)
        
        # 盈亏比
        avg_win = returns[returns > 0].mean()
        avg_loss = abs(returns[returns < 0].mean())
        profit_loss_ratio = avg_win / avg_loss if avg_loss > 0 else 0
        
        return {
            'total_return': total_return,
            'annual_return': annual_return,
            'annual_volatility': annual_vol,
            'sharpe_ratio': sharpe_ratio,
            'sortino_ratio': sortino_ratio,
            'max_drawdown': max_drawdown,
            'calmar_ratio': calmar_ratio,
            'win_rate': win_rate,
            'profit_loss_ratio': profit_loss_ratio,
            'daily_return_mean': daily_return_mean,
            'daily_return_std': daily_return_std
        }
    
    def _calculate_regime_performance(self,
                                       returns: pd.Series,
                                       regimes: Dict[str, pd.Series]) -> Dict:
        """计算分市场环境表现"""
        regime_perf = {}
        
        for regime_name, regime_mask in regimes.items():
            regime_returns = returns[regime_mask]
            
            if len(regime_returns) > 0:
                regime_perf[regime_name] = {
                    'total_return': (1 + regime_returns).prod() - 1,
                    'annual_return': regime_returns.mean() * 252,
                    'annual_vol': regime_returns.std() * np.sqrt(252),
                    'sharpe': regime_returns.mean() / regime_returns.std() * np.sqrt(252) if regime_returns.std() > 0 else 0,
                    'win_rate': (regime_returns > 0).sum() / len(regime_returns),
                    'count': len(regime_returns),
                    'pct_time': len(regime_returns) / len(returns) * 100
                }
        
        return regime_perf
```

```python
class MarketRegimeAnalyzer:
    """市场环境分析器"""
    
    def __init__(self, validator: SystematicBacktestValidator):
        self.validator = validator
        
    def analyze_regime_performance(self,
                                    strategy_results: Dict) -> pd.DataFrame:
        """
        分析各市场环境下的策略表现
        
        Args:
            strategy_results: 策略回测结果
            
        Returns:
            市场环境表现矩阵
        """
        regime_summary = []
        
        for strategy_name, result in strategy_results.items():
            for regime, perf in result['regime_performance'].items():
                regime_summary.append({
                    '策略': strategy_name,
                    '市场环境': regime,
                    '年化收益': perf['annual_return'],
                    '夏普比率': perf['sharpe'],
                    '胜率': perf['win_rate'],
                    '时间占比': perf['pct_time']
                })
        
        df = pd.DataFrame(regime_summary)
        
        return df
    
    def plot_regime_comparison(self,
                                regime_df: pd.DataFrame):
        """绘制市场环境对比图"""
        fig, axes = plt.subplots(2, 2, figsize=(15, 10))
        
        # 年化收益对比
        pivot_return = regime_df.pivot(index='策略', columns='市场环境', values='年化收益')
        pivot_return.plot(kind='bar', ax=axes[0, 0], title='各市场环境年化收益')
        axes[0, 0].set_ylabel('年化收益率')
        axes[0, 0].axhline(y=0, color='r', linestyle='--', alpha=0.3)
        
        # 夏普比率对比
        pivot_sharpe = regime_df.pivot(index='策略', columns='市场环境', values='夏普比率')
        pivot_sharpe.plot(kind='bar', ax=axes[0, 1], title='各市场环境夏普比率')
        axes[0, 1].set_ylabel('夏普比率')
        axes[0, 1].axhline(y=1, color='r', linestyle='--', alpha=0.3)
        
        # 胜率对比
        pivot_winrate = regime_df.pivot(index='策略', columns='市场环境', values='胜率')
        pivot_winrate.plot(kind='bar', ax=axes[1, 0], title='各市场环境胜率')
        axes[1, 0].set_ylabel('胜率')
        axes[1, 0].axhline(y=0.5, color='r', linestyle='--', alpha=0.3)
        
        # 时间占比
        pivot_time = regime_df.pivot(index='策略', columns='市场环境', values='时间占比')
        pivot_time.plot(kind='bar', stacked=True, ax=axes[1, 1], title='市场环境时间占比')
        axes[1, 1].set_ylabel('时间占比 (%)')
        
        plt.tight_layout()
        plt.savefig('/Users/sigma/.openclaw/workspace/quantitative/results/regime_comparison.png', dpi=300)
        plt.close()
        
    def generate_regime_report(self,
                                regime_df: pd.DataFrame) -> str:
        """生成市场环境分析报告"""
        report = []
        report.append("# 市场环境分析报告\n")
        report.append("## 1. 总体表现\n")
        
        # 各策略在牛市的表现
        bull_perf = regime_df[regime_df['市场环境'] == 'bull']
        report.append("### 牛市表现 TOP3:")
        bull_top = bull_perf.nlargest(3, '年化收益')
        for idx, row in bull_top.iterrows():
            report.append(f"- {row['策略']}: 年化收益 {row['年化收益']:.2%}, 夏普 {row['夏普比率']:.2f}")
        
        # 各策略在熊市的表现
        bear_perf = regime_df[regime_df['市场环境'] == 'bear']
        report.append("\n### 熊市表现 TOP3:")
        bear_top = bear_perf.nlargest(3, '年化收益')
        for idx, row in bear_top.iterrows():
            report.append(f"- {row['策略']}: 年化收益 {row['年化收益']:.2%}, 夏普 {row['夏普比率']:.2f}")
        
        # 各策略在震荡市的表现
        sideways_perf = regime_df[regime_df['市场环境'] == 'sideways']
        report.append("\n### 震荡市表现 TOP3:")
        sideways_top = sideways_perf.nlargest(3, '年化收益')
        for idx, row in sideways_top.iterrows():
            report.append(f"- {row['策略']}: 年化收益 {row['年化收益']:.2%}, 夏普 {row['夏普比率']:.2f}")
        
        return "\n".join(report)
```

```python
class WalkForwardValidator:
    """Walk-Forward验证器"""
    
    def __init__(self,
                 prices: pd.DataFrame,
                 strategy_func: callable,
                 in_sample_pct: float = 0.7,
                 n_splits: int = 5):
        """
        初始化
        
        Args:
            prices: 价格数据
            strategy_func: 策略函数
            in_sample_pct: 样本内占比
            n_splits: 分割数量
        """
        self.prices = prices
        self.strategy_func = strategy_func
        self.in_sample_pct = in_sample_pct
        self.n_splits = n_splits
        
    def walk_forward_analysis(self,
                               **strategy_params) -> Dict:
        """
        Walk-Forward分析
        
        Args:
            **strategy_params: 策略参数
            
        Returns:
            验证结果
        """
        n_periods = len(self.prices)
        period_length = n_periods // self.n_splits
        
        results = {
            'in_sample': [],
            'out_sample': [],
            'params': []
        }
        
        for i in range(self.n_splits):
            # 样本内期间
            is_start = i * period_length
            is_end = is_start + int(period_length * self.in_sample_pct)
            
            # 样本外期间
            oos_start = is_end
            oos_end = (i + 1) * period_length if i < self.n_splits - 1 else n_periods
            
            # 样本内数据
            is_data = self.prices.iloc[is_start:is_end]
            
            # 样本外数据
            oos_data = self.prices.iloc[oos_start:oos_end]
            
            # 在样本内优化参数
            best_params = self._optimize_params(is_data, strategy_params)
            
            # 在样本外测试
            is_signals = self.strategy_func(is_data, **best_params)
            oos_signals = self.strategy_func(oos_data, **best_params)
            
            # 计算收益
            is_returns = (is_signals.shift(1) * is_data.pct_change()).sum(axis=1)
            oos_returns = (oos_signals.shift(1) * oos_data.pct_change()).sum(axis=1)
            
            # 存储结果
            results['in_sample'].append(self._calc_metrics(is_returns))
            results['out_sample'].append(self._calc_metrics(oos_returns))
            results['params'].append(best_params)
        
        return results
    
    def _optimize_params(self,
                          data: pd.DataFrame,
                          param_ranges: Dict) -> Dict:
        """
        参数优化
        
        Args:
            data: 样本内数据
            param_ranges: 参数范围
            
        Returns:
            最优参数
        """
        # 简化：使用网格搜索
        # 实际应用中可以使用更高级的优化方法
        best_params = {}
        best_sharpe = -np.inf
        
        # 这里简化处理，返回默认参数
        # 实际应该遍历参数空间
        for param, value in param_ranges.items():
            if isinstance(value, list):
                best_params[param] = value[len(value)//2]  # 取中值
            else:
                best_params[param] = value
        
        return best_params
    
    def _calc_metrics(self, returns: pd.Series) -> Dict:
        """计算性能指标"""
        return {
            'total_return': (1 + returns).prod() - 1,
            'sharpe': returns.mean() / returns.std() * np.sqrt(252) if returns.std() > 0 else 0,
            'max_dd': self._calculate_max_drawdown(returns)
        }
    
    def _calculate_max_drawdown(self, returns: pd.Series) -> float:
        """计算最大回撤"""
        cumulative = (1 + returns).cumprod()
        running_max = cumulative.cummax()
        drawdown = (cumulative - running_max) / running_max
        return drawdown.min()
    
    def analyze_overfitting(self,
                             results: Dict) -> Dict:
        """
        分析过拟合
        
        Args:
            results: Walk-Forward结果
            
        Returns:
            过拟合分析
        """
        is_sharpes = [r['sharpe'] for r in results['in_sample']]
        oos_sharpes = [r['sharpe'] for r in results['out_sample']]
        
        # 计算过拟合指标
        avg_is_sharpe = np.mean(is_sharpes)
        avg_oos_sharpe = np.mean(oos_sharpes)
        
        # 过拟合比率
        overfit_ratio = avg_is_sharpe / avg_oos_sharpe if avg_oos_sharpe > 0 else np.inf
        
        # 参数稳定性
        params_df = pd.DataFrame(results['params'])
        param_stability = params_df.std() / params_df.mean()
        
        return {
            'avg_is_sharpe': avg_is_sharpe,
            'avg_oos_sharpe': avg_oos_sharpe,
            'overfit_ratio': overfit_ratio,
            'param_stability': param_stability.to_dict(),
            'is_overfitted': overfit_ratio > 1.5  # 样本内夏普比样本外高50%以上
        }
```

```python
class PortfolioOptimizer:
    """组合权重优化器"""
    
    def __init__(self,
                 strategy_returns: pd.DataFrame,
                 target_return: float = 0.15,
                 max_weight: float = 0.3):
        """
        初始化
        
        Args:
            strategy_returns: 策略收益DataFrame
            target_return: 目标收益
            max_weight: 单策略最大权重
        """
        self.returns = strategy_returns
        self.target_return = target_return
        self.max_weight = max_weight
        
    def optimize_weights(self,
                          method: str = 'max_sharpe') -> np.ndarray:
        """
        优化权重
        
        Args:
            method: 优化方法 ('max_sharpe', 'min_vol', 'risk_parity')
            
        Returns:
            最优权重
        """
        n_strategies = self.returns.shape[1]
        
        if method == 'max_sharpe':
            # 最大夏普比率
            return self._max_sharpe()
        elif method == 'min_vol':
            # 最小波动率
            return self._min_volatility()
        elif method == 'risk_parity':
            # 风险平价
            return self._risk_parity()
        else:
            # 等权重
            return np.ones(n_strategies) / n_strategies
    
    def _max_sharpe(self) -> np.ndarray:
        """最大夏普比率优化"""
        from scipy.optimize import minimize
        
        n = self.returns.shape[1]
        mu = self.returns.mean() * 252
        cov = self.returns.cov() * 252
        
        def neg_sharpe(weights):
            ret = np.dot(weights, mu)
            vol = np.sqrt(np.dot(weights.T, np.dot(cov, weights)))
            return -(ret - 0.03) / vol
        
        constraints = [
            {'type': 'eq', 'fun': lambda w: np.sum(w) - 1},
            {'type': 'ineq', 'fun': lambda w: w},
            {'type': 'ineq', 'fun': lambda w: self.max_weight - w}
        ]
        
        result = minimize(
            neg_sharpe,
            np.ones(n) / n,
            method='SLSQP',
            constraints=constraints
        )
        
        return result.x
    
    def _min_volatility(self) -> np.ndarray:
        """最小波动率优化"""
        from scipy.optimize import minimize
        
        n = self.returns.shape[1]
        cov = self.returns.cov() * 252
        
        def portfolio_vol(weights):
            return np.sqrt(np.dot(weights.T, np.dot(cov, weights)))
        
        constraints = [
            {'type': 'eq', 'fun': lambda w: np.sum(w) - 1},
            {'type': 'ineq', 'fun': lambda w: w},
            {'type': 'ineq', 'fun': lambda w: self.max_weight - w}
        ]
        
        result = minimize(
            portfolio_vol,
            np.ones(n) / n,
            method='SLSQP',
            constraints=constraints
        )
        
        return result.x
    
    def _risk_parity(self) -> np.ndarray:
        """风险平价优化"""
        n = self.returns.shape[1]
        vol = self.returns.std() * np.sqrt(252)
        
        # 风险平价权重 = 1/vol_i / sum(1/vol_j)
        inv_vol = 1 / vol
        weights = inv_vol / inv_vol.sum()
        
        # 限制最大权重
        weights = np.minimum(weights, self.max_weight)
        weights = weights / weights.sum()
        
        return weights
    
    def backtest_portfolio(self,
                            weights: np.ndarray,
                            rebalance_freq: str = 'Q') -> Dict:
        """
        回测组合
        
        Args:
            weights: 权重
            rebalance_freq: 再平衡频率
            
        Returns:
            回测结果
        """
        # 计算组合收益
        portfolio_returns = (self.returns * weights).sum(axis=1)
        
        # 计算净值
        nav = (1 + portfolio_returns).cumprod()
        
        # 性能指标
        performance = {
            'total_return': nav.iloc[-1] - 1,
            'annual_return': portfolio_returns.mean() * 252,
            'annual_vol': portfolio_returns.std() * np.sqrt(252),
            'sharpe': portfolio_returns.mean() / portfolio_returns.std() * np.sqrt(252),
            'max_dd': self._calculate_max_drawdown(portfolio_returns)
        }
        
        return {
            'nav': nav,
            'returns': portfolio_returns,
            'weights': weights,
            'performance': performance
        }
    
    def _calculate_max_drawdown(self, returns: pd.Series) -> float:
        """计算最大回撤"""
        cumulative = (1 + returns).cumprod()
        running_max = cumulative.cummax()
        drawdown = (cumulative - running_max) / running_max
        return drawdown.min()
```
