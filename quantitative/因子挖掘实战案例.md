### 1.1 加载数据

```python
import pandas as pd
import numpy as np

# 加载股票数据
def load_stock_data(tickers, start_date, end_date):
    """
    加载股票数据
    
    参数:
    tickers: 股票列表
    start_date: 开始日期
    end_date: 结束日期
    
    返回:
    数据字典
    """
    data = {}
    for ticker in tickers:
        # 这里使用示例数据
        df = pd.read_csv(f'data/{ticker}.csv', index_col=0, parse_dates=True)
        data[ticker] = df
    
    return data

# 计算目标变量（未来收益率）
def calculate_target(df, forward_period=5):
    """
    计算目标变量
    
    参数:
    df: 股票数据
    forward_period: 前瞻期
    
    返回:
    目标变量
    """
    target = df['close'].pct_change(forward_period).shift(-forward_period)
    return target
```

## 2. 因子构造

### 2.1 传统因子

```python
def construct_traditional_factors(df):
    """
    构造传统因子
    
    参数:
    df: 股票数据
    
    返回:
    因子DataFrame
    """
    factors = pd.DataFrame(index=df.index)
    
    # 价值因子
    if 'book_value' in df.columns and 'market_cap' in df.columns:
        factors['BP'] = df['book_value'] / df['market_cap']
    
    # 动量因子
    factors['momentum_1m'] = df['close'].pct_change(20)
    factors['momentum_3m'] = df['close'].pct_change(60)
    factors['momentum_6m'] = df['close'].pct_change(120)
    
    # 波动率因子
    factors['volatility_1m'] = df['close'].pct_change().rolling(20).std() * np.sqrt(252)
    factors['volatility_3m'] = df['close'].pct_change().rolling(60).std() * np.sqrt(252)
    
    # 成交量因子
    factors['volume_ratio'] = df['volume'] / df['volume'].rolling(20).mean()
    
    # 技术指标
    # RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(14).mean()
    factors['rsi'] = 100 - (100 / (1 + gain / loss))
    
    # MACD
    ema12 = df['close'].ewm(span=12, adjust=False).mean()
    ema26 = df['close'].ewm(span=26, adjust=False).mean()
    factors['macd'] = ema12 - ema26
    
    return factors
```

### 2.2 机器学习因子

```python
from sklearn.ensemble import RandomForestRegressor
import xgboost as xgb

def extract_ml_factors(X_train, y_train, X_test, model_type='xgboost'):
    """
    提取机器学习因子
    
    参数:
    X_train: 训练特征
    y_train: 训练目标
    X_test: 测试特征
    model_type: 模型类型
    
    返回:
    预测因子
    """
    if model_type == 'xgboost':
        model = xgb.XGBRegressor(
            n_estimators=200,
            max_depth=5,
            learning_rate=0.05,
            random_state=42,
            n_jobs=-1
        )
    elif model_type == 'random_forest':
        model = RandomForestRegressor(
            n_estimators=200,
            max_depth=10,
            random_state=42,
            n_jobs=-1
        )
    
    # 训练
    model.fit(X_train, y_train)
    
    # 预测
    predictions = model.predict(X_test)
    
    return predictions, model
```

## 3. 因子检验

### 3.1 IC检验

```python
from scipy.stats import spearmanr

def ic_test(factor_values, returns):
    """
    IC检验
    
    参数:
    factor_values: 因子值
    returns: 收益率
    
    返回:
    IC统计
    """
    # IC
    ic, p_value = spearmanr(factor_values, returns)
    
    # Rank IC
    factor_rank = pd.Series(factor_values).rank()
    ret_rank = pd.Series(returns).rank()
    rank_ic, _ = spearmanr(factor_rank, ret_rank)
    
    return {
        'IC': ic,
        'Rank IC': rank_ic,
        'p_value': p_value
    }

def ic_series(factor_df, return_df):
    """
    IC时间序列
    
    参数:
    factor_df: 因子DataFrame (date, stock)
    return_df: 收益率DataFrame (date, stock)
    
    返回:
    IC序列
    """
    ic_series = []
    
    for date in factor_df.index:
        factor_values = factor_df.loc[date]
        returns = return_df.loc[date]
        
        # 去除NaN
        mask = ~(factor_values.isna() | returns.isna())
        factor_values = factor_values[mask]
        returns = returns[mask]
        
        if len(factor_values) > 10:
            ic, _ = spearmanr(factor_values, returns)
            ic_series.append({'date': date, 'IC': ic})
    
    return pd.DataFrame(ic_series)
```

### 3.2 分组回测

```python
def group_backtest(factor_values, returns, n_groups=5):
    """
    分组回测
    
    参数:
    factor_values: 因子值
    returns: 收益率
    n_groups: 分组数
    
    返回:
    各组收益
    """
    # 分组
    groups = pd.qcut(factor_values, n_groups, labels=False, duplicates='drop')
    
    # 各组平均收益
    group_returns = []
    for i in range(n_groups):
        group_mask = groups == i
        group_ret = returns[group_mask].mean()
        group_returns.append(group_ret)
    
    # 多空收益
    long_short = group_returns[-1] - group_returns[0]
    
    return group_returns, long_short
```

## 4. 实战案例

### 4.1 完整流程

```python
class FactorMiningPipeline:
    """因子挖掘流水线"""
    
    def __init__(self, data):
        self.data = data
        self.factors = None
        self.target = None
    
    def construct_factors(self):
        """构造因子"""
        factors_list = []
        
        for ticker, df in self.data.items():
            # 传统因子
            trad_factors = construct_traditional_factors(df)
            
            # 添加股票标识
            trad_factors['ticker'] = ticker
            factors_list.append(trad_factors)
        
        self.factors = pd.concat(factors_list)
        return self
    
    def calculate_target(self, forward_period=5):
        """计算目标"""
        target_list = []
        
        for ticker, df in self.data.items():
            target = calculate_target(df, forward_period)
            target.name = 'target'
            target_list.append(target)
        
        self.target = pd.concat(target_list)
        return self
    
    def train_ml_factor(self, train_start, train_end, test_start, test_end):
        """训练ML因子"""
        # 数据划分
        train_mask = (self.factors.index >= train_start) & (self.factors.index <= train_end)
        test_mask = (self.factors.index >= test_start) & (self.factors.index <= test_end)
        
        X_train = self.factors[train_mask].drop('ticker', axis=1)
        y_train = self.target[train_mask]
        
        X_test = self.factors[test_mask].drop('ticker', axis=1)
        y_test = self.target[test_mask]
        
        # 训练
        predictions, model = extract_ml_factors(X_train, y_train, X_test)
        
        # IC检验
        ic_stats = ic_test(predictions, y_test)
        
        return predictions, model, ic_stats
```

### 4.2 多因子组合

```python
def combine_factors(factors_dict, method='ic_weighted'):
    """
    组合因子
    
    参数:
    factors_dict: 因子字典 {factor_name: factor_values}
    method: 组合方法
    
    返回:
    组合因子
    """
    if method == 'equal_weighted':
        # 等权
        combined = pd.DataFrame(factors_dict).mean(axis=1)
    
    elif method == 'ic_weighted':
        # IC加权
        weights = []
        for name, factor in factors_dict.items():
            # 这里需要IC值
            weights.append(1)  # 简化处理
        
        weights = np.array(weights) / sum(weights)
        combined = pd.DataFrame(factors_dict) @ weights
    
    return combined
```
