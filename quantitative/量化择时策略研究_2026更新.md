## 二、最新择时策略详解

### 2.1 RSRS择时策略（阻力支撑相对强度）

#### 2.1.1 策略原理

RSRS（Resistance Support Relative Strength）是光大证券开发的经典择时指标，通过量化阻力位和支撑位的相对强度来判断市场趋势。

**核心思想**：
- 支撑位和阻力位不是固定值，而是动态变化的
- 通过线性回归获取斜率和R²值
- 斜率反映趋势强度，R²反映拟合质量

#### 2.1.2 实现代码

```python
import numpy as np
import pandas as pd
from sklearn.linear_model import LinearRegression

class RSRS:
    """阻力支撑相对强度择时策略"""
    
    def __init__(self, window=18):
        self.window = window
        
    def calculate_rsrs(self, high, low):
        """
        计算RSRS指标
        
        参数:
        - high: 最高价序列
        - low: 最低价序列
        
        返回:
        - beta: 斜率
        - r_squared: R²值
        """
        betas = []
        r_squareds = []
        
        for i in range(self.window, len(high)):
            # 构建回归数据
            y = high.iloc[i-self.window:i].values.reshape(-1, 1)
            x = low.iloc[i-self.window:i].values.reshape(-1, 1)
            
            # 线性回归
            model = LinearRegression()
            model.fit(x, y)
            
            # 斜率
            beta = model.coef_[0][0]
            
            # R²
            r_squared = model.score(x, y)
            
            betas.append(beta)
            r_squareds.append(r_squared)
        
        return np.array(betas), np.array(r_squareds)
    
    def generate_signal(self, beta, r_squared, threshold=0.8):
        """
        生成交易信号
        
        策略逻辑:
        - beta > 1 且 R² > threshold: 看多（买入信号）
        - beta < 1 且 R² > threshold: 看空（卖出信号）
        - R² < threshold: 震荡（不操作）
        """
        signals = np.zeros(len(beta))
        
        for i in range(len(beta)):
            if r_squared[i] > threshold:
                if beta[i] > 1:
                    signals[i] = 1  # 买入
                elif beta[i] < 1:
                    signals[i] = -1  # 卖出
        
        return signals

# 使用示例
rsrs = RSRS(window=18)
beta, r_squared = rsrs.calculate_rsrs(data['high'], data['low'])
signals = rsrs.generate_signal(beta, r_squared)
```

```python
class EnhancedRSRS:
    """增强版RSRS策略"""
    
    def __init__(self, window=18, ma_window=600):
        self.window = window
        self.ma_window = ma_window
        
    def calculate_standardized_beta(self, high, low):
        """计算标准化斜率"""
        betas, r_squareds = [], []
        
        for i in range(self.window, len(high)):
            y = high.iloc[i-self.window:i].values.reshape(-1, 1)
            x = low.iloc[i-self.window:i].values.reshape(-1, 1)
            
            model = LinearRegression()
            model.fit(x, y)
            
            betas.append(model.coef_[0][0])
            r_squareds.append(model.score(x, y))
        
        betas = np.array(betas)
        
        # 标准化
        beta_mean = pd.Series(betas).rolling(self.ma_window).mean()
        beta_std = pd.Series(betas).rolling(self.ma_window).std()
        standardized_beta = (betas - beta_mean) / beta_std
        
        return standardized_beta, np.array(r_squareds)
    
    def generate_enhanced_signal(self, beta, r_squared, momentum, 
                                  beta_threshold=1.0, r2_threshold=0.8):
        """
        生成增强信号
        
        结合动量因子和RSRS
        """
        signals = np.zeros(len(beta))
        
        for i in range(len(beta)):
            # RSRS信号
            if r_squared[i] > r2_threshold:
                if beta[i] > beta_threshold:
                    rsrs_signal = 1
                elif beta[i] < -beta_threshold:
                    rsrs_signal = -1
                else:
                    rsrs_signal = 0
            else:
                rsrs_signal = 0
            
            # 动量信号
            mom_signal = 1 if momentum[i] > 0 else -1 if momentum[i] < 0 else 0
            
            # 综合信号
            signals[i] = 0.7 * rsrs_signal + 0.3 * mom_signal
        
        # 信号平滑
        signals = pd.Series(signals).rolling(5).mean()
        
        return np.where(signals > 0.5, 1, np.where(signals < -0.5, -1, 0))
```

### 2.2 HHT模型择时（希尔伯特-黄变换）

#### 2.2.1 策略原理

HHT（Hilbert-Huang Transform）是一种自适应的时频分析方法，特别适合非平稳、非线性的金融时间序列。

**核心步骤**：
1. **EMD（经验模态分解）**：将价格序列分解为多个IMF（本征模态函数）
2. **Hilbert变换**：对每个IMF进行Hilbert变换，获取瞬时频率和振幅
3. **信号重构**：选择关键IMF分量重构信号

#### 2.2.2 实现代码

```python
from scipy.signal import hilbert
from scipy.interpolate import interp1d

class HHTTiming:
    """基于HHT的择时策略"""
    
    def __init__(self, n_imf=6):
        self.n_imf = n_imf
        
    def emd_decomposition(self, signal):
        """
        经验模态分解
        """
        imfs = []
        residue = signal.copy()
        
        for _ in range(self.n_imf):
            # 提取IMF
            imf = self._extract_imf(residue)
            imfs.append(imf)
            residue = residue - imf
        
        return imfs, residue
    
    def _extract_imf(self, signal):
        """提取单个IMF"""
        imf = signal.copy()
        
        for _ in range(100):  # 最大迭代次数
            # 找极值点
            maxima = self._find_peaks(imf)
            minima = self._find_peaks(-imf)
            
            # 插值上下包络
            upper_envelope = self._interpolate(maxima, imf[maxima], len(imf))
            lower_envelope = self._interpolate(minima, imf[minima], len(imf))
            
            # 计算均值
            mean_envelope = (upper_envelope + lower_envelope) / 2
            
            # 更新IMF
            new_imf = imf - mean_envelope
            
            # 检查是否为IMF
            if self._is_imf(new_imf):
                return new_imf
            
            imf = new_imf
        
        return imf
    
    def hilbert_transform(self, imf):
        """
        Hilbert变换
        获取瞬时振幅和频率
        """
        analytic_signal = hilbert(imf)
        amplitude = np.abs(analytic_signal)
        phase = np.unwrap(np.angle(analytic_signal))
        frequency = np.diff(phase) / (2 * np.pi)
        
        return amplitude, frequency
    
    def generate_signal(self, signal, key_imfs=[2, 3]):
        """
        生成交易信号
        
        选择关键IMF分量重构信号
        """
        # EMD分解
        imfs, residue = self.emd_decomposition(signal)
        
        # 选择关键IMF
        reconstructed = np.zeros(len(signal))
        for i in key_imfs:
            if i < len(imfs):
                # 获取振幅和频率
                amp, freq = self.hilbert_transform(imfs[i])
                
                # 根据振幅和频率生成信号
                reconstructed += imfs[i]
        
        # 信号平滑
        reconstructed = pd.Series(reconstructed).rolling(5).mean()
        
        # 生成交易信号
        signals = np.where(reconstructed > 0, 1, np.where(reconstructed < 0, -1, 0))
        
        return signals

# 使用示例
hht = HHTTiming(n_imf=6)
signals = hht.generate_signal(data['close'].values, key_imfs=[2, 3, 4])
```

```python
import lightgbm as lgb
from sklearn.model_selection import TimeSeriesSplit
from sklearn.metrics import accuracy_score, precision_score, recall_score

class LightGBMTiming:
    """基于LightGBM的择时策略"""
    
    def __init__(self, params=None):
        self.params = params or {
            'objective': 'multiclass',
            'num_class': 3,  # 涨、跌、震荡
            'metric': 'multi_logloss',
            'boosting_type': 'gbdt',
            'num_leaves': 31,
            'learning_rate': 0.05,
            'feature_fraction': 0.8,
            'bagging_fraction': 0.8,
            'bagging_freq': 5,
            'verbose': -1,
            'num_threads': 4
        }
        self.model = None
        
    def create_features(self, data):
        """创建特征"""
        features = pd.DataFrame(index=data.index)
        
        # 价格特征
        features['returns'] = data['close'].pct_change()
        features['log_returns'] = np.log(data['close'] / data['close'].shift(1))
        features['high_low_ratio'] = data['high'] / data['low']
        features['close_open_ratio'] = data['close'] / data['open']
        
        # 移动平均
        for window in [5, 10, 20, 60]:
            features[f'ma_{window}'] = data['close'].rolling(window).mean() / data['close']
            features[f'ma_{window}_ratio'] = features[f'ma_{window}'] / features[f'ma_{window}'].shift(window)
        
        # 动量
        for window in [5, 10, 20]:
            features[f'momentum_{window}'] = data['close'] / data['close'].shift(window) - 1
        
        # 波动率
        for window in [5, 10, 20]:
            features[f'volatility_{window}'] = features['returns'].rolling(window).std()
        
        # 成交量
        features['volume_ratio'] = data['volume'] / data['volume'].rolling(20).mean()
        features['volume_ma_5'] = data['volume'].rolling(5).mean() / data['volume'].rolling(20).mean()
        
        # 技术指标
        features['rsi'] = self._calculate_rsi(data['close'])
        features['macd'], features['macd_signal'] = self._calculate_macd(data['close'])
        features['bollinger_upper'], features['bollinger_lower'] = self._calculate_bollinger(data['close'])
        
        return features.dropna()
    
    def create_labels(self, data, horizon=5, threshold=0.02):
        """创建标签"""
        future_returns = data['close'].shift(-horizon) / data['close'] - 1
        
        labels = np.zeros(len(data))
        labels[future_returns > threshold] = 2  # 上涨
        labels[future_returns < -threshold] = 0  # 下跌
        labels[(future_returns >= -threshold) & (future_returns <= threshold)] = 1  # 震荡
        
        return labels
    
    def train(self, features, labels, n_splits=5):
        """训练模型"""
        # 时间序列交叉验证
        tscv = TimeSeriesSplit(n_splits=n_splits)
        
        best_model = None
        best_score = 0
        
        for train_idx, val_idx in tscv.split(features):
            X_train, X_val = features.iloc[train_idx], features.iloc[val_idx]
            y_train, y_val = labels[train_idx], labels[val_idx]
            
            train_data = lgb.Dataset(X_train, label=y_train)
            val_data = lgb.Dataset(X_val, label=y_val, reference=train_data)
            
            model = lgb.train(
                self.params,
                train_data,
                num_boost_round=1000,
                valid_sets=[val_data],
                early_stopping_rounds=50,
                verbose_eval=False
            )
            
            # 评估
            y_pred = model.predict(X_val).argmax(axis=1)
            score = accuracy_score(y_val, y_pred)
            
            if score > best_score:
                best_score = score
                best_model = model
        
        self.model = best_model
        print(f"Best validation accuracy: {best_score:.4f}")
        
    def predict(self, features):
        """预测"""
        predictions = self.model.predict(features).argmax(axis=1)
        return predictions
    
    def get_feature_importance(self, features):
        """获取特征重要性"""
        importance = pd.DataFrame({
            'feature': features.columns,
            'importance': self.model.feature_importance()
        }).sort_values('importance', ascending=False)
        
        return importance
    
    # 技术指标计算函数
    def _calculate_rsi(self, prices, period=14):
        delta = prices.diff()
        gain = (delta.where(delta > 0, 0)).rolling(period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(period).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
    
    def _calculate_macd(self, prices, fast=12, slow=26, signal=9):
        ema_fast = prices.ewm(span=fast).mean()
        ema_slow = prices.ewm(span=slow).mean()
        macd = ema_fast - ema_slow
        signal_line = macd.ewm(span=signal).mean()
        return macd, signal_line
    
    def _calculate_bollinger(self, prices, period=20, std_dev=2):
        ma = prices.rolling(period).mean()
        std = prices.rolling(period).std()
        upper = ma + std_dev * std
        lower = ma - std_dev * std
        return upper, lower
```

### 2.4 强化学习择时

#### 2.4.1 DQN择时策略

```python
import torch
import torch.nn as nn
import torch.optim as optim
import numpy as np
from collections import deque
import random

class DQNTimingAgent:
    """基于DQN的择时智能体"""
    
    def __init__(self, state_size, action_size=3, hidden_size=128):
        self.state_size = state_size
        self.action_size = action_size
        self.memory = deque(maxlen=10000)
        
        # 超参数
        self.gamma = 0.95
        self.epsilon = 1.0
        self.epsilon_min = 0.01
        self.epsilon_decay = 0.995
        self.learning_rate = 0.001
        
        # 网络模型
        self.model = self._build_model(hidden_size)
        self.target_model = self._build_model(hidden_size)
        self.update_target_model()
        
        self.optimizer = optim.Adam(self.model.parameters(), lr=self.learning_rate)
        self.criterion = nn.MSELoss()
    
    def _build_model(self, hidden_size):
        """构建Q网络"""
        return nn.Sequential(
            nn.Linear(self.state_size, hidden_size),
            nn.ReLU(),
            nn.Dropout(0.2),
            nn.Linear(hidden_size, hidden_size),
            nn.ReLU(),
            nn.Dropout(0.2),
            nn.Linear(hidden_size, self.action_size)
        )
    
    def update_target_model(self):
        """更新目标网络"""
        self.target_model.load_state_dict(self.model.state_dict())
    
    def remember(self, state, action, reward, next_state, done):
        """存储经验"""
        self.memory.append((state, action, reward, next_state, done))
    
    def act(self, state):
        """选择动作"""
        if np.random.random() < self.epsilon:
            return random.randrange(self.action_size)
        
        with torch.no_grad():
            state = torch.FloatTensor(state).unsqueeze(0)
            q_values = self.model(state)
            return q_values.argmax().item()
    
    def replay(self, batch_size=32):
        """经验回放"""
        if len(self.memory) < batch_size:
            return
        
        minibatch = random.sample(self.memory, batch_size)
        
        for state, action, reward, next_state, done in minibatch:
            state = torch.FloatTensor(state).unsqueeze(0)
            next_state = torch.FloatTensor(next_state).unsqueeze(0)
            
            # 计算目标Q值
            with torch.no_grad():
                target = reward
                if not done:
                    target = reward + self.gamma * self.target_model(next_state).max().item()
            
            # 训练
            self.optimizer.zero_grad()
            output = self.model(state)
            target_tensor = output.clone()
            target_tensor[0][action] = target
            loss = self.criterion(output, target_tensor)
            loss.backward()
            self.optimizer.step()
        
        # 降低探索率
        if self.epsilon > self.epsilon_min:
            self.epsilon *= self.epsilon_decay
    
    def train(self, env, episodes=100):
        """训练智能体"""
        for episode in range(episodes):
            state = env.reset()
            total_reward = 0
            
            for t in range(env.max_steps):
                # 选择动作
                action = self.act(state)
                
                # 执行动作
                next_state, reward, done, _ = env.step(action)
                total_reward += reward
                
                # 存储经验
                self.remember(state, action, reward, next_state, done)
                
                # 经验回放
                self.replay(batch_size=32)
                
                state = next_state
                
                if done:
                    break
            
            # 更新目标网络
            if episode % 10 == 0:
                self.update_target_model()
            
            if episode % 10 == 0:
                print(f"Episode {episode}, Total Reward: {total_reward:.2f}, Epsilon: {self.epsilon:.4f}")
```

```python
    metrics['annual_return'] = (1 + returns.mean()) ** 252 - 1
    
    # 年化波动率
    metrics['annual_volatility'] = returns.std() * np.sqrt(252)
    
    # 夏普比率
    risk_free_rate = 0.03
    metrics['sharpe_ratio'] = (metrics['annual_return'] - risk_free_rate) / metrics['annual_volatility']
    
    # 最大回撤
    cum_returns = (1 + returns).cumprod()
    running_max = cum_returns.cummax()
    drawdown = (cum_returns - running_max) / running_max
    metrics['max_drawdown'] = drawdown.min()
    
    # 卡玛比率
    metrics['calmar_ratio'] = metrics['annual_return'] / abs(metrics['max_drawdown'])
    
    # 胜率
    metrics['win_rate'] = (returns > 0).sum() / len(returns)
    
    # 盈亏比
    win_returns = returns[returns > 0]
    loss_returns = returns[returns < 0]
    metrics['profit_loss_ratio'] = win_returns.mean() / abs(loss_returns.mean())
    
    # Sortino比率
    downside_returns = returns[returns < 0]
    downside_std = downside_returns.std() * np.sqrt(252)
    metrics['sortino_ratio'] = (metrics['annual_return'] - risk_free_rate) / downside_std
    
    return metrics
```

## 五、策略组合与优化

### 5.1 多策略组合

```python
class StrategyEnsemble:
    """策略集成"""
    
    def __init__(self, strategies, weights=None):
        self.strategies = strategies
        self.weights = weights or [1/len(strategies)] * len(strategies)
        
    def generate_signals(self, data):
        """生成组合信号"""
        all_signals = []
        
        for strategy in self.strategies:
            signals = strategy.generate_signal(data)
            all_signals.append(signals)
        
        # 加权平均
        combined_signals = np.zeros(len(all_signals[0]))
        for i, signals in enumerate(all_signals):
            combined_signals += self.weights[i] * signals
        
        # 信号阈值化
        combined_signals = np.where(combined_signals > 0.5, 1,
                                   np.where(combined_signals < -0.5, -1, 0))
        
        return combined_signals
```

### 5.2 动态权重调整

```python
class DynamicWeighting:
    """动态权重调整"""
    
    def __init__(self, lookback=63):
        self.lookback = lookback
        
    def calculate_weights(self, strategy_returns):
        """根据历史表现计算权重"""
        # 计算夏普比率
        sharpe_ratios = strategy_returns.mean() / strategy_returns.std() * np.sqrt(252)
        
        # 归一化权重
        weights = sharpe_ratios / sharpe_ratios.sum()
        
        # 确保权重为正且和为1
        weights = np.maximum(weights, 0)
        weights = weights / weights.sum()
        
        return weights
```

### 6.1 风险控制

**仓位管理**：
- 单策略仓位不超过总资金的30%

- 不同时间周期策略组合

**实时监控**：
- 每日监控策略表现
- 定期检查策略有效性
- 及时调整失效策略

### 6.2 交易成本

```python
def calculate_transaction_cost(signals, prices, volume, 
                               commission_rate=0.0003, 
                               slippage_rate=0.0001):
    """计算交易成本"""
    # 佣金
    commission = np.abs(signals.diff()) * prices * commission_rate
    
    # 滑点
    slippage = np.abs(signals.diff()) * prices * slippage_rate
    
    # 冲击成本
    avg_volume = volume.rolling(20).mean()
    trade_size = np.abs(signals.diff()) * prices
    impact = 0.1 * (trade_size / avg_volume) ** 0.5 * prices
    
    total_cost = commission + slippage + impact
    
    return total_cost
```
