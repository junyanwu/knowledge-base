## 二、策略思路描述

### 2.1 VaR风险控制系统

**核心原理**：
- 在给定置信水平下，计算投资组合最大可能损失
- 实时监控VaR，超过阈值时降低仓位

**计算方法对比**：

| 方法 | 优点 | 缺点 | 适用场景 |
|------|------|------|---------|
| 历史模拟 | 无分布假设 | 依赖历史 | 尾部风险关注 |
| 参数法 | 计算简单 | 低估尾部 | 正态分布假设 |
| 蒙特卡洛 | 灵活 | 计算量大 | 复杂组合 |

### 2.2 凯利公式仓位管理

**核心公式**：
```
f* = p - (1-p)/b

其中：
f* = 最优仓位比例
p = 胜率
b = 盈亏比
```

**实际应用**：
- 使用半凯利或四分之一凯利降低风险
- 根据策略历史表现估算参数
- 动态调整仓位

### 2.3 风险平价组合

**核心思想**：
- 每个资产贡献相等的风险
- 不依赖收益预测
- 实现真正的风险分散

### 2.4 动态止损策略

**固定比例止损**：买入价 × (1 - 止损比例)
**ATR动态止损**：价格 - n × ATR
**移动止损**：跟踪最高价，锁定盈利

## 三、实现伪代码

### 3.1 VaR计算模块

```python
import numpy as np
from scipy import stats

class VaRCalculator:
    """VaR计算模块"""
    
    def __init__(self, confidence_level=0.95):
        self.confidence_level = confidence_level
    
    def historical_var(self, returns):
        """历史模拟法VaR"""
        return -np.percentile(returns, (1 - self.confidence_level) * 100)
    
    def parametric_var(self, returns):
        """参数法VaR"""
        mu = returns.mean()
        sigma = returns.std()
        z = stats.norm.ppf(1 - self.confidence_level)
        return -(mu + z * sigma)
    
    def cvar(self, returns):
        """条件VaR (Expected Shortfall)"""
        var = self.historical_var(returns)
        return -returns[returns < -var].mean()
    
    def monte_carlo_var(self, returns, n_simulations=10000):
        """蒙特卡洛模拟VaR"""
        mu = returns.mean()
        sigma = returns.std()
        simulated = np.random.normal(mu, sigma, n_simulations)
        return self.historical_var(simulated)
```

### 3.2 仓位管理模块

```python
class PositionManager:
    """仓位管理模块"""
    
    def __init__(self, total_capital, max_position_pct=0.10, 
                 max_sector_pct=0.30, max_drawdown=0.15):
        self.total_capital = total_capital
        self.max_position_pct = max_position_pct
        self.max_sector_pct = max_sector_pct
        self.max_drawdown = max_drawdown
    
    def kelly_position(self, win_rate, win_loss_ratio, fraction=0.5):
        """凯利公式仓位（使用半凯利）"""
        kelly = win_rate - (1 - win_rate) / win_loss_ratio
        # 限制仓位范围
        kelly = max(0, min(kelly, 0.25))  # 最大25%
        return kelly * fraction  # 使用半凯利
    
    def volatility_adjusted_position(self, target_vol, current_vol, base_position):
        """波动率调整仓位"""
        return base_position * (target_vol / current_vol)
    
    def check_position_limit(self, position, sector_positions, sector):
        """检查仓位限制"""
        # 单资产限制
        if position > self.total_capital * self.max_position_pct:
            position = self.total_capital * self.max_position_pct
        
        # 行业限制
        sector_total = sector_positions.get(sector, 0) + position
        if sector_total > self.total_capital * self.max_sector_pct:
            position = self.total_capital * self.max_sector_pct - sector_positions.get(sector, 0)
        
        return position
```

### 3.3 止损模块

```python
class StopLossManager:
    """止损管理模块"""
    
    def __init__(self, stop_loss_pct=0.05, trailing_stop_pct=0.15):
        self.stop_loss_pct = stop_loss_pct
        self.trailing_stop_pct = trailing_stop_pct
    
    def fixed_stop_loss(self, entry_price):
        """固定止损"""
        return entry_price * (1 - self.stop_loss_pct)
    
    def atr_stop_loss(self, price, atr, multiplier=2):
        """ATR动态止损"""
        return price - multiplier * atr
    
    def trailing_stop(self, prices, entry_price):
        """移动止损"""
        peak = prices.expanding().max()
        stop_price = peak * (1 - self.trailing_stop_pct)
        # 止损只能上移，不能下移
        stop_price = stop_price.expanding().max()
        return stop_price
    
    def check_stop_loss(self, current_price, entry_price, stop_price):
        """检查是否触发止损"""
        return current_price <= stop_price
```

### 3.4 风险平价组合

```python
from scipy.optimize import minimize

class RiskParityPortfolio:
    """风险平价组合"""
    
    def __init__(self, n_assets):
        self.n_assets = n_assets
    
    def portfolio_volatility(self, weights, cov_matrix):
        """组合波动率"""
        return np.sqrt(weights @ cov_matrix @ weights)
    
    def risk_contribution(self, weights, cov_matrix):
        """各资产风险贡献"""
        port_vol = self.portfolio_volatility(weights, cov_matrix)
        marginal_contrib = cov_matrix @ weights
        risk_contrib = weights * marginal_contrib / port_vol
        return risk_contrib
    
    def objective(self, weights, cov_matrix):
        """优化目标：风险贡献相等"""
        rc = self.risk_contribution(weights, cov_matrix)
        target_risk = 1 / self.n_assets
        return np.sum((rc - target_risk) ** 2)
    
    def optimize(self, cov_matrix):
        """优化权重"""
        init_weights = np.ones(self.n_assets) / self.n_assets
        bounds = [(0.01, 1) for _ in range(self.n_assets)]
        constraints = {'type': 'eq', 'fun': lambda w: np.sum(w) - 1}
        
        result = minimize(
            self.objective, init_weights,
            args=(cov_matrix,),
            method='SLSQP',
            bounds=bounds,
            constraints=constraints
        )
        
        return result.x
```

### 3.5 完整风控系统

```python
class RiskManagementSystem:
    """完整风险管理系统"""
    
    def __init__(self, capital, risk_limits):
        self.capital = capital
        self.limits = risk_limits
        self.var_calculator = VaRCalculator()
        self.position_manager = PositionManager(capital)
        self.stop_loss_manager = StopLossManager()
    
    def daily_risk_check(self, portfolio, returns):
        """日度风险检查"""
        risks = {
            'var_95': self.var_calculator.historical_var(returns),
            'cvar_95': self.var_calculator.cvar(returns),
            'volatility': returns.std() * np.sqrt(252),
            'max_drawdown': self.calculate_max_drawdown(portfolio),
            'position_check': self.check_positions(portfolio),
            'stop_loss_check': self.check_stop_losses(portfolio)
        }
        
        alerts = []
        if risks['var_95'] > self.limits.max_var:
            alerts.append(f"VaR超限: {risks['var_95']:.2%} > {self.limits.max_var:.2%}")
        if risks['max_drawdown'] > self.limits.max_drawdown:
            alerts.append(f"回撤超限: {risks['max_drawdown']:.2%} > {self.limits.max_drawdown:.2%}")
        
        return risks, alerts
    
    def calculate_max_drawdown(self, portfolio_values):
        """计算最大回撤"""
        peak = np.maximum.accumulate(portfolio_values)
        drawdown = (peak - portfolio_values) / peak
        return np.max(drawdown)
    
    def check_positions(self, portfolio):
        """检查仓位限制"""
        total_value = sum(portfolio.positions.values())
        position_check = {}
        
        for asset, value in portfolio.positions.items():
            pct = value / self.capital
            position_check[asset] = {
                'weight': pct,
                'within_limit': pct <= self.limits.max_position_pct
            }
        
        return position_check
    
    def check_stop_losses(self, portfolio):
        """检查止损"""
        stop_loss_check = {}
        
        for asset, position in portfolio.positions.items():
            stop_price = self.stop_loss_manager.fixed_stop_loss(position.entry_price)
            triggered = position.current_price <= stop_price
            stop_loss_check[asset] = {
                'stop_price': stop_price,
                'current_price': position.current_price,
                'triggered': triggered
            }
        
        return stop_loss_check
    
    def generate_risk_report(self, portfolio, returns):
        """生成风险报告"""
        risks, alerts = self.daily_risk_check(portfolio, returns)
        
        report = {
            'date': pd.Timestamp.now(),
            'portfolio_value': portfolio.total_value,
            'risks': risks,
            'alerts': alerts,
            'positions': portfolio.positions,
            'risk_status': 'WARNING' if alerts else 'NORMAL'
        }
        
        return report
```
