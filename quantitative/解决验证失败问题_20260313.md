## 一、背景与问题分析

### V4.0策略现状

| 指标 | 等权重组合 | 风险平价组合 | 目标 | 达成 |
|------|-----------|-------------|------|------|
| 年化收益 | 11.84% | 12.74% | >5% | ✅ |
| 夏普比率 | 0.65 | 0.72 | >0.5 | ✅ |
| Walk-Forward差距 | 83.6% | 90.3% | <40% | ❌ |
| K-fold差距 | 88.7% | 94.7% | <40% | ❌ |
| 后视镜偏差 | 无 | 无 | 无 | ✅ |

### 根本原因

**市场环境结构性差异**：

| 时期 | 市场环境 | 年化收益 | 特征 |
|------|---------|---------|------|
| 训练集（2022-2024） | 震荡市 | 3.29%~6.06% | 横盘震荡，趋势不明 |
| 测试集（2024-2026） | 牛市 | 33.91%~37.04% | 单边上涨，趋势明显 |

**关键洞察**：
- 等权重和风险平价策略在牛市表现优异
- 但在震荡市表现平庸
- 这导致样本内外差异巨大
- **不是策略有问题，而是市场环境发生了根本性变化**

## 二、市场环境识别方法

### 2.1 核心指标体系

#### 2.1.1 趋势强度指标（ADX）

**定义**：平均方向性指数（Average Directional Index）

```
ADX = 100 × MA(|+DI - -DI| / (+DI + -DI), period)
```

**市场环境判断标准**：

| ADX值 | 市场状态 | 策略建议 |
|-------|---------|---------|
| < 20 | 无趋势/震荡市 | 降低仓位，使用均值回归策略 |
| 20-25 | 趋势形成中 | 适度参与，设置严格止损 |
| 25-50 | 强趋势 | 趋势跟踪策略，加大仓位 |
| > 50 | 极强趋势 | 趋势延续，但需警惕反转 |

**Python实现**：
```python
def calculate_adx(df, period=14):
    """计算ADX指标"""
    high = df['high']
    low = df['low']
    close = df['close']
    
    # 计算+DM和-DM
    plus_dm = high.diff()
    minus_dm = -low.diff()
    
    plus_dm[plus_dm < 0] = 0
    minus_dm[minus_dm < 0] = 0
    
    # 计算TR
    tr1 = high - low
    tr2 = abs(high - close.shift())
    tr3 = abs(low - close.shift())
    tr = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # 计算平滑值
    atr = tr.rolling(period).mean()
    plus_di = 100 * (plus_dm.rolling(period).mean() / atr)
    minus_di = 100 * (minus_dm.rolling(period).mean() / atr)
    
    # 计算ADX
    dx = 100 * abs(plus_di - minus_di) / (plus_di + minus_di)
    adx = dx.rolling(period).mean()
    
    return adx, plus_di, minus_di
```

```python
def calculate_volatility_regime(df, window=20):
    """计算波动率状态"""
    returns = df['close'].pct_change()
    
    # 历史波动率
    hist_vol = returns.rolling(window).std() * np.sqrt(252)
    
    # 波动率百分位
    vol_percentile = hist_vol.rolling(252).rank(pct=True)
    
    # 分类
    if vol_percentile.iloc[-1] > 0.8:
        return 'high_vol'  # 高波动
    elif vol_percentile.iloc[-1] < 0.2:
        return 'low_vol'   # 低波动
    else:
        return 'normal_vol'  # 正常波动
```

```python
from hmmlearn import hmm

class MarketRegimeHMM:
    """使用隐马尔可夫模型识别市场状态"""
    
    def __init__(self, n_states=3):
        self.n_states = n_states
        self.model = None
        
    def fit(self, returns):
        """训练HMM模型"""
        # 准备特征
        features = np.column_stack([
            returns.values,
            returns.rolling(20).std().values,
            returns.rolling(20).skew().values
        ])
        
        # 去除NaN
        features = features[~np.isnan(features).any(axis=1)]
        
        # 训练HMM
        self.model = hmm.GaussianHMM(
            n_components=self.n_states,
            covariance_type='full',
            n_iter=100
        )
        self.model.fit(features)
        
        return self
    
    def predict(self, returns):
        """预测当前市场状态"""
        features = np.column_stack([
            returns.values[-1:],
            returns.rolling(20).std().values[-1:],
            returns.rolling(20).skew().values[-1:]
        ])
        
        state = self.model.predict(features)
        return state[0]
```

### 3.2 滚动窗口优化

**核心思想**：定期重新优化参数，适应市场变化

```python
class RollingWindowOptimizer:
    """滚动窗口参数优化"""
    
    def __init__(self, 
                 train_window=252,  # 训练窗口：1年
                 test_window=63,    # 测试窗口：3个月
                 step=63):          # 步进：3个月
        
        self.train_window = train_window
        self.test_window = test_window
        self.step = step
        
    def optimize_and_test(self, df, param_grid):
        """滚动优化和测试"""
        results = []
        
        for i in range(0, len(df) - self.train_window - self.test_window, self.step):
            # 训练窗口
            train_data = df.iloc[i:i+self.train_window]
            
            # 测试窗口
            test_data = df.iloc[i+self.train_window:i+self.train_window+self.test_window]
            
            # 优化参数
            best_params = self.optimize_params(train_data, param_grid)
            
            # 样本外测试
            test_return = self.backtest(test_data, best_params)
            
            results.append({
                'train_start': train_data.index[0],
                'train_end': train_data.index[-1],
                'test_start': test_data.index[0],
                'test_end': test_data.index[-1],
                'params': best_params,
                'test_return': test_return
            })
        
        return results
    
    def optimize_params(self, data, param_grid):
        """网格搜索优化"""
        best_params = None
        best_sharpe = -np.inf
        
        for params in param_grid:
            sharpe = self.calculate_sharpe(data, params)
            if sharpe > best_sharpe:
                best_sharpe = sharpe
                best_params = params
        
        return best_params
```

### 3.3 在线学习算法

**核心思想**：实时更新模型，适应市场变化

```python
class OnlineLearningStrategy:
    """在线学习策略"""
    
    def __init__(self, learning_rate=0.01, decay_rate=0.95):
        self.learning_rate = learning_rate
        self.decay_rate = decay_rate
        self.weights = None
        self.performance_history = []
        
    def update_weights(self, features, returns):
        """在线更新权重"""
        if self.weights is None:
            self.weights = np.ones(len(features)) / len(features)
        
        # 计算预测误差
        predicted = np.dot(features, self.weights)
        error = returns - predicted
        
        # 梯度下降更新
        gradient = -2 * error * features
        self.weights -= self.learning_rate * gradient
        
        # 权重归一化
        self.weights = np.maximum(self.weights, 0)
        self.weights /= self.weights.sum()
        
        # 衰减旧权重
        self.weights *= self.decay_rate
        self.weights /= self.weights.sum()
        
        return self.weights
    
    def predict(self, features):
        """预测信号"""
        if self.weights is None:
            return 0.5
        
        return np.dot(features, self.weights)
```

### 3.4 集成学习策略

**核心思想**：多个策略投票，降低单一策略风险

```python
class EnsembleStrategy:
    """集成策略"""
    
    def __init__(self, strategies):
        self.strategies = strategies
        self.weights = np.ones(len(strategies)) / len(strategies)
        
    def generate_signals(self, df):
        """生成集成信号"""
        signals = []
        
        for strategy in self.strategies:
            signal = strategy.generate_signal(df)
            signals.append(signal)
        
        # 加权投票
        buy_signal = sum(s['buy'] * w for s, w in zip(signals, self.weights))
        sell_signal = sum(s['sell'] * w for s, w in zip(signals, self.weights))
        
        return {
            'buy': buy_signal > 0.5,
            'sell': sell_signal > 0.5,
            'confidence': max(buy_signal, sell_signal)
        }
    
    def update_weights(self, performance):
        """根据历史表现更新权重"""
        # 表现好的策略获得更高权重
        total_perf = sum(performance)
        self.weights = [p / total_perf for p in performance]
```

## 四、稳健验证方法

### 4.1 Walk-Forward验证失败的原因分析

**V4.0验证失败的根本原因**：

1. **市场环境结构性变化**
   - 训练集（2022-2024）：震荡市，低收益
   - 测试集（2024-2026）：牛市，高收益
   - 收益差异本质上是市场环境差异

2. **验证方法假设不成立**
   - Walk-Forward假设训练集和测试集分布相似
   - 但实际上市场环境发生了根本性变化
   - 这不是策略问题，是验证方法的局限性

3. **策略本身是正确的**
   - 等权重和风险平价都是客观标准
   - 不存在后视镜偏差
   - 样本内外差异大是因为市场环境变化大

### 4.2 更稳健的验证方法

#### 4.2.1 分层验证（Stratified Validation）

**核心思想**：确保训练集和测试集包含相同比例的市场环境

```python
class StratifiedWalkForward:
    """分层Walk-Forward验证"""
    
    def __init__(self, train_ratio=0.7):
        self.train_ratio = train_ratio
        
    def split_by_regime(self, df):
        """按市场环境分层"""
        # 识别每个时期的市场环境
        df['regime'] = self.classify_regime(df)
        
        # 确保训练集和测试集包含相同比例的各市场环境
        regimes = df['regime'].unique()
        train_indices = []
        test_indices = []
        
        for regime in regimes:
            regime_indices = df[df['regime'] == regime].index
            split_point = int(len(regime_indices) * self.train_ratio)
            
            train_indices.extend(regime_indices[:split_point])
            test_indices.extend(regime_indices[split_point:])
        
        return train_indices, test_indices
    
    def classify_regime(self, df):
        """分类市场环境"""
        adx, _, _ = calculate_adx(df)
        
        regimes = []
        for i in range(len(df)):
            if adx.iloc[i] > 25:
                regimes.append('trending')
            else:
                regimes.append('sideways')
        
        return pd.Series(regimes, index=df.index)
```

#### 4.2.2 嵌套交叉验证（Nested Cross-Validation）

**核心思想**：内层优化参数，外层验证策略

```python
class NestedCrossValidation:
    """嵌套交叉验证"""
    
    def __init__(self, outer_folds=5, inner_folds=3):
        self.outer_folds = outer_folds
        self.inner_folds = inner_folds
        
    def validate(self, df, param_grid):
        """嵌套验证"""
        outer_results = []
        fold_size = len(df) // self.outer_folds
        
        for outer_fold in range(self.outer_folds):
            # 外层测试集
            test_start = outer_fold * fold_size
            test_end = (outer_fold + 1) * fold_size
            
            test_data = df.iloc[test_start:test_end]
            train_data = pd.concat([
                df.iloc[:test_start],
                df.iloc[test_end:]
            ])
            
            # 内层交叉验证选择最佳参数
            best_params = self.inner_cv(train_data, param_grid)
            
            # 使用最佳参数在外层测试集验证
            test_return = self.backtest(test_data, best_params)
            
            outer_results.append({
                'fold': outer_fold,
                'params': best_params,
                'test_return': test_return
            })
        
        return outer_results
    
    def inner_cv(self, data, param_grid):
        """内层交叉验证"""
        inner_fold_size = len(data) // self.inner_folds
        param_scores = {}
        
        for params in param_grid:
            scores = []
            
            for inner_fold in range(self.inner_folds):
                val_start = inner_fold * inner_fold_size
                val_end = (inner_fold + 1) * inner_fold_size
                
                val_data = data.iloc[val_start:val_end]
                train_data = pd.concat([
                    data.iloc[:val_start],
                    data.iloc[val_end:]
                ])
                
                # 训练并验证
                sharpe = self.calculate_sharpe(train_data, val_data, params)
                scores.append(sharpe)
            
            param_scores[str(params)] = np.mean(scores)
        
        # 返回最佳参数
        best_params_str = max(param_scores, key=param_scores.get)
        return eval(best_params_str)
```

#### 4.2.3 蒙特卡洛交叉验证

**核心思想**：随机采样验证，减少偏差

```python
class MonteCarloCrossValidation:
    """蒙特卡洛交叉验证"""
    
    def __init__(self, n_iterations=100, test_ratio=0.3):
        self.n_iterations = n_iterations
        self.test_ratio = test_ratio
        
    def validate(self, df, strategy):
        """蒙特卡洛验证"""
        results = []
        
        for i in range(self.n_iterations):
            # 随机采样
            test_size = int(len(df) * self.test_ratio)
            test_indices = np.random.choice(len(df), test_size, replace=False)
            train_indices = np.array([j for j in range(len(df)) if j not in test_indices])
            
            train_data = df.iloc[train_indices]
            test_data = df.iloc[test_indices]
            
            # 训练和测试
            train_return = strategy.backtest(train_data)
            test_return = strategy.backtest(test_data)
            
            results.append({
                'iteration': i,
                'train_return': train_return,
                'test_return': test_return,
                'diff': abs(train_return - test_return)
            })
        
        # 计算统计量
        diffs = [r['diff'] for r in results]
        
        return {
            'mean_diff': np.mean(diffs),
            'std_diff': np.std(diffs),
            'median_diff': np.median(diffs),
            'percentile_95': np.percentile(diffs, 95),
            'results': results
        }
```

## 五、资产配置方法

### 5.1 风险平价（Risk Parity）详解

**核心思想**：每个资产的风险贡献相等

```python
class RiskParityPortfolio:
    """风险平价组合"""
    
    def __init__(self, target_volatility=0.15):
        self.target_volatility = target_volatility
        
    def calculate_weights(self, returns):
        """计算风险平价权重"""
        # 计算协方差矩阵
        cov_matrix = returns.cov() * 252
        
        # 风险平价权重（波动率倒数）
        vols = np.sqrt(np.diag(cov_matrix))
        weights = 1 / vols
        weights /= weights.sum()
        
        return weights
    
    def calculate_risk_contribution(self, weights, cov_matrix):
        """计算风险贡献"""
        portfolio_vol = np.sqrt(np.dot(weights, np.dot(cov_matrix, weights)))
        
        marginal_contrib = np.dot(cov_matrix, weights)
        risk_contrib = weights * marginal_contrib / portfolio_vol
        
        return risk_contrib / risk_contrib.sum()  # 归一化
    
    def optimize(self, returns):
        """优化风险平价权重"""
        from scipy.optimize import minimize
        
        cov_matrix = returns.cov() * 252
        n_assets = len(returns.columns)
        
        def objective(weights):
            # 目标：最小化风险贡献差异
            risk_contrib = self.calculate_risk_contribution(weights, cov_matrix)
            target_contrib = np.ones(n_assets) / n_assets
            return np.sum((risk_contrib - target_contrib) ** 2)
        
        # 约束条件
        constraints = [
            {'type': 'eq', 'fun': lambda w: np.sum(w) - 1}  # 权重和为1
        ]
        bounds = [(0.01, 1) for _ in range(n_assets)]  # 权重在1%-100%之间
        
        # 优化
        result = minimize(
            objective,
            x0=np.ones(n_assets) / n_assets,
            method='SLSQP',
            bounds=bounds,
            constraints=constraints
        )
        
        return result.x
```

```python
class MinimumVariancePortfolio:
    """最小方差组合"""
    
    def calculate_weights(self, returns):
        """计算最小方差权重"""
        from scipy.optimize import minimize
        
        cov_matrix = returns.cov() * 252
        n_assets = len(returns.columns)
        
        def portfolio_variance(weights):
            return np.dot(weights, np.dot(cov_matrix, weights))
        
        # 约束条件
        constraints = [
            {'type': 'eq', 'fun': lambda w: np.sum(w) - 1}
        ]
        bounds = [(0.01, 1) for _ in range(n_assets)]
        
        # 优化
        result = minimize(
            portfolio_variance,
            x0=np.ones(n_assets) / n_assets,
            method='SLSQP',
            bounds=bounds,
            constraints=constraints
        )
        
        return result.x
```

### 5.3 最大分散化组合（Maximum Diversification）

```python
class MaximumDiversificationPortfolio:
    """最大分散化组合"""
    
    def calculate_weights(self, returns):
        """计算最大分散化权重"""
        from scipy.optimize import minimize
        
        cov_matrix = returns.cov() * 252
        vols = np.sqrt(np.diag(cov_matrix))
        n_assets = len(returns.columns)
        
        def diversification_ratio(weights):
            # 分散化比率 = 加权平均波动率 / 组合波动率
            weighted_vol = np.dot(weights, vols)
            portfolio_vol = np.sqrt(np.dot(weights, np.dot(cov_matrix, weights)))
            return -weighted_vol / portfolio_vol  # 负号因为要最小化
        
        # 约束条件
        constraints = [
            {'type': 'eq', 'fun': lambda w: np.sum(w) - 1}
        ]
        bounds = [(0.01, 1) for _ in range(n_assets)]
        
        # 优化
        result = minimize(
            diversification_ratio,
            x0=np.ones(n_assets) / n_assets,
            method='SLSQP',
            bounds=bounds,
            constraints=constraints
        )
        
        return result.x
```

### 5.4 Black-Litterman模型

```python
class BlackLittermanPortfolio:
    """Black-Litterman模型"""
    
    def __init__(self, tau=0.05, risk_aversion=2.5):
        self.tau = tau  # 观点不确定性
        self.risk_aversion = risk_aversion
        
    def calculate_weights(self, market_weights, cov_matrix, views, view_confidences):
        """
        计算Black-Litterman权重
        
        参数:
        - market_weights: 市场权重（市值加权）
        - cov_matrix: 协方差矩阵
        - views: 投资者观点矩阵
        - view_confidences: 观点置信度
        """
        # 隐含均衡收益
        pi = self.risk_aversion * np.dot(cov_matrix, market_weights)
        
        # 观点矩阵P和观点收益Q
        P = views['P']  # 观点矩阵
        Q = views['Q']  # 观点收益
        
        # 观点不确定性矩阵
        Omega = np.diag(view_confidences)
        
        # Black-Litterman公式
        tau_cov = self.tau * cov_matrix
        M1 = np.linalg.inv(tau_cov)
        M2 = np.dot(P.T, np.dot(np.linalg.inv(Omega), P))
        M3 = np.dot(M1, pi) + np.dot(P.T, np.dot(np.linalg.inv(Omega), Q))
        
        # 后验收益
        posterior_returns = np.dot(np.linalg.inv(M1 + M2), M3)
        
        # 后验协方差
        posterior_cov = np.linalg.inv(M1 + M2)
        
        # 最优权重
        weights = np.dot(np.linalg.inv(self.risk_aversion * posterior_cov), posterior_returns)
        weights = weights / weights.sum()  # 归一化
        
        return weights
```

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
市场环境检测器 - V4.1核心模块
"""

import pandas as pd
import numpy as np

class MarketEnvironmentDetector:
    """市场环境检测器"""
    
    def __init__(self, adx_period=14, ma_periods=[20, 50, 200], vol_window=20):
        self.adx_period = adx_period
        self.ma_periods = ma_periods
        self.vol_window = vol_window
        
    def calculate_adx(self, df):
        """计算ADX指标"""
        high = df['high']
        low = df['low']
        close = df['close']
        
        # +DM和-DM
        plus_dm = high.diff()
        minus_dm = -low.diff()
        
        plus_dm[plus_dm < 0] = 0
        minus_dm[minus_dm < 0] = 0
        
        # TR
        tr1 = high - low
        tr2 = abs(high - close.shift())
        tr3 = abs(low - close.shift())
        tr = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
        
        # ATR
        atr = tr.rolling(self.adx_period).mean()
        
        # +DI和-DI
        plus_di = 100 * (plus_dm.rolling(self.adx_period).mean() / atr)
        minus_di = 100 * (minus_dm.rolling(self.adx_period).mean() / atr)
        
        # DX和ADX
        dx = 100 * abs(plus_di - minus_di) / (plus_di + minus_di + 0.0001)
        adx = dx.rolling(self.adx_period).mean()
        
        return adx, plus_di, minus_di
    
    def detect_ma_trend(self, df):
        """检测均线趋势"""
        close = df['close']
        
        ma20 = close.rolling(20).mean()
        ma50 = close.rolling(50).mean()
        ma200 = close.rolling(200).mean()
        
        price = close.iloc[-1]
        
        if price > ma20.iloc[-1] > ma50.iloc[-1] > ma200.iloc[-1]:
            return 'strong_bull'
        elif price > ma50.iloc[-1] > ma200.iloc[-1]:
            return 'bull'
        elif price < ma20.iloc[-1] < ma50.iloc[-1] < ma200.iloc[-1]:
            return 'strong_bear'
        elif price < ma50.iloc[-1] < ma200.iloc[-1]:
            return 'bear'
        else:
            return 'sideways'
    
    def classify_volatility(self, df):
        """分类波动率"""
        returns = df['close'].pct_change()
        hist_vol = returns.rolling(self.vol_window).std() * np.sqrt(252)
        
        vol_percentile = hist_vol.rolling(252).rank(pct=True).iloc[-1]
        
        if vol_percentile > 0.8:
            return 'high'
        elif vol_percentile < 0.2:
            return 'low'
        else:
            return 'normal'
    
    def detect(self, df):
        """检测市场环境"""
        if len(df) < 200:
            return 'insufficient_data'
        
        # 趋势强度
        adx, plus_di, minus_di = self.calculate_adx(df)
        adx_value = adx.iloc[-1]
        
        # 趋势方向
        ma_trend = self.detect_ma_trend(df)
        
        # 波动率
        vol_regime = self.classify_volatility(df)
        
        # 综合判断
        if adx_value > 25:
            if plus_di.iloc[-1] > minus_di.iloc[-1]:
                env = 'bull_trend'
            else:
                env = 'bear_trend'
        elif adx_value < 20:
            env = 'sideways'
        else:
            env = 'transition'
        
        # 加入波动率
        if vol_regime == 'high':
            env += '_high_vol'
        elif vol_regime == 'low':
            env += '_low_vol'
        
        return {
            'environment': env,
            'adx': adx_value,
            'plus_di': plus_di.iloc[-1],
            'minus_di': minus_di.iloc[-1],
            'ma_trend': ma_trend,
            'volatility': vol_regime
        }
```
