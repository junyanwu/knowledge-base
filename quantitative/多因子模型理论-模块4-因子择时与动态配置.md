# 多因子模型理论 - 模块4：因子择时与动态配置

## 1. 因子择时基础

### 1.1 因子择时原理

**核心思想：** 因子收益具有时变性，可根据市场状态动态调整因子权重

**择时信号来源：**
- 宏观经济指标
- 市场情绪指标
- 因子动量
- 技术指标

### 1.2 因子周期性

```python
import pandas as pd
import numpy as np

def analyze_factor_cycle(factor_returns, lookback=60):
    """
    分析因子周期性
    
    参数:
    factor_returns: 因子收益序列
    lookback: 回溯期
    
    返回:
    周期指标
    """
    from scipy.fft import fft
    
    # 傅里叶变换
    n = len(factor_returns)
    freqs = np.fft.fftfreq(n)
    fft_vals = fft(factor_returns)
    
    # 功率谱
    power = np.abs(fft_vals)**2
    
    # 主周期
    peak_freq = freqs[np.argmax(power[1:n//2]) + 1]
    period = 1 / peak_freq if peak_freq != 0 else np.inf
    
    return {
        'period': period,
        'power_spectrum': power,
        'frequencies': freqs
    }
```

## 2. 因子动量

### 2.1 时间序列动量

```python
def factor_momentum_ts(factor_returns, lookback=12):
    """
    因子时间序列动量
    
    参数:
    factor_returns: 因子收益 (n_periods, n_factors)
    lookback: 回溯期
    
    返回:
    动量信号
    """
    # 累计收益
    cumulative_returns = factor_returns.rolling(lookback).sum()
    
    # 动量信号
    momentum_signal = (cumulative_returns > 0).astype(int)
    momentum_signal[momentum_signal == 0] = -1
    
    return momentum_signal
```

### 2.2 截面动量

```python
def factor_momentum_cross_section(factor_returns, lookback=12, n_top=3):
    """
    因子截面动量
    
    参数:
    factor_returns: 因子收益 (n_periods, n_factors)
    lookback: 回溯期
    n_top: 选择表现最好的n个因子
    
    返回:
    因子权重
    """
    # 累计收益
    cumulative_returns = factor_returns.rolling(lookback).sum()
    
    # 排名
    ranks = cumulative_returns.rank(axis=1, ascending=False)
    
    # 选择top因子
    weights = (ranks <= n_top).astype(float)
    weights = weights.div(weights.sum(axis=1), axis=0)
    
    return weights
```

## 3. 市场状态识别

### 3.1 市场状态分类

```python
from sklearn.cluster import KMeans

def identify_market_regime(market_data, n_regimes=3):
    """
    识别市场状态
    
    参数:
    market_data: 市场数据（波动率、收益率、成交量等）
    n_regimes: 状态数量
    
    返回:
    市场状态标签
    """
    # 特征
    features = market_data[['volatility', 'momentum', 'volume']].values
    
    # K-means聚类
    kmeans = KMeans(n_clusters=n_regimes, random_state=42)
    regimes = kmeans.fit_predict(features)
    
    return regimes, kmeans
```

### 3.2 马尔可夫切换模型

```python
from statsmodels.tsa.regime_switching.markov_regression import MarkovRegression

def markov_switching_model(returns, k_regimes=2):
    """
    马尔可夫切换模型
    
    参数:
    returns: 收益率序列
    k_regimes: 状态数量
    
    返回:
    模型结果
    """
    model = MarkovRegression(
        returns,
        k_regimes=k_regimes,
        switching_variance=True
    )
    results = model.fit()
    
    # 状态概率
    smoothed_probs = results.smoothed_marginal_probabilities
    
    return results, smoothed_probs
```

## 4. 动态因子配置

### 4.1 基于市场状态的配置

```python
def regime_based_allocation(factor_returns, market_regimes, n_regimes=3):
    """
    基于市场状态的因子配置
    
    参数:
    factor_returns: 因子收益 (n_periods, n_factors)
    market_regimes: 市场状态标签
    n_regimes: 状态数量
    
    返回:
    动态权重
    """
    n_factors = factor_returns.shape[1]
    weights = pd.DataFrame(index=factor_returns.index, columns=factor_returns.columns)
    
    # 各状态下的最优因子
    for regime in range(n_regimes):
        regime_mask = market_regimes == regime
        regime_returns = factor_returns[regime_mask]
        
        # 计算夏普比率
        sharpe = regime_returns.mean() / regime_returns.std()
        
        # 选择表现好的因子
        top_factors = sharpe.nlargest(n_factors // 2).index
        regime_weights = pd.Series(0, index=factor_returns.columns)
        regime_weights[top_factors] = 1 / len(top_factors)
        
        weights[regime_mask] = regime_weights
    
    return weights
```

### 4.2 风险预算动态调整

```python
def dynamic_risk_budget(factor_returns, target_vol=0.15):
    """
    动态风险预算
    
    参数:
    factor_returns: 因子收益
    target_vol: 目标波动率
    
    返回:
    动态权重
    """
    # 滚动波动率
    rolling_vol = factor_returns.rolling(60).std() * np.sqrt(252)
    
    # 风险预算权重
    risk_budget = 1 / rolling_vol
    weights = risk_budget.div(risk_budget.sum(axis=1), axis=0)
    
    # 波动率调整
    portfolio_vol = (weights * factor_returns).sum(axis=1).rolling(60).std() * np.sqrt(252)
    vol_adj = target_vol / portfolio_vol
    vol_adj = vol_adj.clip(0.5, 2.0)  # 限制在0.5-2倍
    
    # 调整权重
    adj_weights = weights.mul(vol_adj, axis=0)
    
    return adj_weights
```

## 5. 因子择时评估

### 5.1 择时能力检验

```python
def timing_ability_test(factor_returns, timing_signal):
    """
    检验择时能力
    
    参数:
    factor_returns: 因子收益
    timing_signal: 择时信号（-1或1）
    
    返回:
    择时能力指标
    """
    # 择时收益
    timing_returns = factor_returns * timing_signal.shift(1)
    
    # 累计收益
    cumulative_timing = (1 + timing_returns).cumprod()
    cumulative_buy_hold = (1 + factor_returns).cumprod()
    
    # 择时能力
    from scipy.stats import pearsonr
    correlation, p_value = pearsonr(timing_signal[:-1], factor_returns[1:])
    
    # HM检验
    def henriksson_merton_test(returns, signal):
        # 预测正确率
        correct = ((signal > 0) & (returns > 0)) | ((signal < 0) & (returns < 0))
        hit_rate = correct.mean()
        
        # 检验
        from scipy.stats import binom_test
        p_value = binom_test(correct.sum(), len(correct), 0.5)
        
        return hit_rate, p_value
    
    hit_rate, hm_pvalue = henriksson_merton_test(factor_returns, timing_signal)
    
    return {
        'timing_return': timing_returns.mean(),
        'correlation': correlation,
        'correlation_pvalue': p_value,
        'hit_rate': hit_rate,
        'hm_pvalue': hm_pvalue
    }
```

## 6. 实战案例

### 6.1 完整动态因子策略

```python
class DynamicFactorStrategy:
    """动态因子策略"""
    
    def __init__(self, factor_returns, lookback=60):
        self.factor_returns = factor_returns
        self.lookback = lookback
    
    def calc_timing_signals(self):
        """计算择时信号"""
        # 因子动量
        momentum = self.factor_returns.rolling(self.lookback).sum()
        signal_momentum = (momentum > 0).astype(int)
        signal_momentum[signal_momentum == 0] = -1
        
        # 市场状态
        volatility = self.factor_returns.rolling(self.lookback).std()
        market_state = self.identify_state(volatility)
        
        return signal_momentum, market_state
    
    def identify_state(self, volatility):
        """识别市场状态"""
        from sklearn.cluster import KMeans
        kmeans = KMeans(n_clusters=3, random_state=42)
        return kmeans.fit_predict(volatility.fillna(0))
    
    def allocate(self, signals, market_state):
        """动态配置"""
        # 根据状态调整信号
        state_weights = {
            0: 0.5,  # 低波动
            1: 1.0,  # 中波动
            2: 1.5   # 高波动
        }
        
        weights = signals.mul([state_weights[s] for s in market_state], axis=0)
        weights = weights.div(weights.abs().sum(axis=1), axis=0)
        
        return weights
    
    def backtest(self):
        """回测"""
        signals, market_state = self.calc_timing_signals()
        weights = self.allocate(signals, market_state)
        
        # 计算收益
        portfolio_returns = (weights.shift(1) * self.factor_returns).sum(axis=1)
        
        # 绩效
        sharpe = portfolio_returns.mean() / portfolio_returns.std() * np.sqrt(252)
        cum_return = (1 + portfolio_returns).cumprod()
        max_dd = (cum_return / cum_return.cummax() - 1).min()
        
        return {
            'returns': portfolio_returns,
            'sharpe': sharpe,
            'max_drawdown': max_dd
        }
```
