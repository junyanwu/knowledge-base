## 1. 多策略整合框架

### 1.1 策略体系概览

本研究生成的震荡市交易策略体系：

| 策略类型 | 核心逻辑 | 适用场景 | 预期收益 | 风险特征 |
|---------|---------|---------|---------|---------|
| 网格交易 | 价格区间波动 | 强震荡市 | 8-15% | 低风险 |
| 均值回归 | 价格回归均值 | 弱震荡市 | 10-15% | 中等风险 |
| 波动率突破 | 波动率扩张 | 波动率收缩后 | 14-18% | 中等风险 |
| 波动率收缩 | 波动率收缩突破 | 低波动后 | 15-19% | 中等风险 |

### 1.2 策略权重分配

**基于市场状态的动态权重**:

```python
import numpy as np
import pandas as pd
from typing import Dict, List

class MultiStrategyPortfolio:
    """多策略组合系统"""
    
    def __init__(self,
                 strategies: Dict[str, object],
                 initial_capital: float = 100000):
        """
        Args:
            strategies: 策略字典 {策略名称: 策略对象}
            initial_capital: 初始资金
        """
        self.strategies = strategies
        self.initial_capital = initial_capital
        
        # 策略权重（初始等权重）
        self.weights = {name: 1.0 / len(strategies) for name in strategies.keys()}
        
        # 策略资金分配
        self.capital_allocation = {
            name: initial_capital * weight 
            for name, weight in self.weights.items()
        }
    
    def update_weights_by_market_regime(self, 
                                       regime: str,
                                       sideways_score: float) -> Dict[str, float]:
        """
        根据市场状态更新策略权重
        
        Args:
            regime: 市场状态（强震荡/弱震荡/趋势市）
            sideways_score: 震荡评分（0-1）
            
        Returns:
            更新后的权重字典
        """
        # 根据震荡强度分配权重
        if sideways_score >= 0.8:  # 极强震荡
            self.weights = {
                'grid_trading': 0.5,
                'mean_reversion': 0.3,
                'volatility_breakout': 0.1,
                'volatility_contraction': 0.1
            }
        elif sideways_score >= 0.6:  # 强震荡
            self.weights = {
                'grid_trading': 0.4,
                'mean_reversion': 0.35,
                'volatility_breakout': 0.15,
                'volatility_contraction': 0.1
            }
        elif sideways_score >= 0.4:  # 弱震荡
            self.weights = {
                'grid_trading': 0.2,
                'mean_reversion': 0.35,
                'volatility_breakout': 0.25,
                'volatility_contraction': 0.2
            }
        elif sideways_score >= 0.2:  # 趋势萌芽
            self.weights = {
                'grid_trading': 0.1,
                'mean_reversion': 0.2,
                'volatility_breakout': 0.35,
                'volatility_contraction': 0.35
            }
        else:  # 强趋势
            self.weights = {
                'grid_trading': 0.0,
                'mean_reversion': 0.0,
                'volatility_breakout': 0.5,
                'volatility_contraction': 0.5
            }
        
        # 更新资金分配
        self.capital_allocation = {
            name: self.initial_capital * weight 
            for name, weight in self.weights.items()
        }
        
        return self.weights
    
    def generate_combined_signals(self, df: pd.DataFrame) -> pd.DataFrame:
        """
        生成组合信号
        
        Returns:
            包含各策略信号和组合信号的DataFrame
        """
        result = df.copy()
        
        # 收集各策略信号
        signals = {}
        for name, strategy in self.strategies.items():
            strategy_signals = strategy.generate_signals(df)
            signals[name] = strategy_signals['signal']
        
        # 转换为DataFrame
        signals_df = pd.DataFrame(signals, index=df.index)
        
        # 加权组合信号
        result['combined_signal'] = sum(
            signals_df[name] * self.weights.get(name, 0) 
            for name in signals_df.columns
        )
        
        # 标准化信号
        result['combined_signal'] = result['combined_signal'] / sum(self.weights.values())
        
        # 离散化信号
        result['final_signal'] = 0
        result.loc[result['combined_signal'] > 0.3, 'final_signal'] = 1
        result.loc[result['combined_signal'] < -0.3, 'final_signal'] = -1
        
        # 保存各策略信号
        for name in signals_df.columns:
            result[f'{name}_signal'] = signals_df[name]
        
        return result
    
    def backtest_portfolio(self, df: pd.DataFrame,
                          commission_rate: float = 0.0003) -> Dict:
        """
        回测组合策略
        
        Returns:
            组合回测结果
        """
        signals = self.generate_combined_signals(df)
        
        # 初始化组合状态
        cash = self.initial_capital
        positions = {name: 0 for name in self.strategies.keys()}
        total_position = 0
        
        trades = []
        nav_curve = []
        
        for i in range(1, len(signals)):
            date = signals.index[i]
            close = signals['close'].iloc[i]
            final_signal = signals['final_signal'].iloc[i]
            
            # 计算当前净值
            current_nav = cash
            for name, pos in positions.items():
                current_nav += pos * close
            
            # 根据信号调整仓位
            if final_signal > 0 and total_position == 0:
                # 买入信号
                total_investment = current_nav * 0.9  # 保留10%现金
                for name, weight in self.weights.items():
                    if weight > 0:
                        investment = total_investment * weight
                        shares = investment / close
                        commission = shares * close * commission_rate
                        cash -= (shares * close + commission)
                        positions[name] = shares
                        total_position += shares
                
                trades.append({
                    'date': date,
                    'action': 'buy',
                    'price': close,
                    'total_shares': total_position
                })
            
            elif final_signal < 0 and total_position > 0:
                # 卖出信号
                for name, pos in positions.items():
                    if pos > 0:
                        commission = pos * close * commission_rate
                        cash += (pos * close - commission)
                        positions[name] = 0
                
                trades.append({
                    'date': date,
                    'action': 'sell',
                    'price': close,
                    'total_shares': total_position
                })
                
                total_position = 0
            
            # 记录净值
            nav = cash + total_position * close
            nav_curve.append({
                'date': date,
                'nav': nav,
                'cash': cash,
                'position_value': total_position * close
            })
        
        # 计算绩效
        nav_df = pd.DataFrame(nav_curve).set_index('date')
        returns = nav_df['nav'].pct_change().dropna()
        
        annual_return = (nav_df['nav'].iloc[-1] / self.initial_capital - 1) * 252 / len(nav_df)
        annual_volatility = returns.std() * np.sqrt(252)
        sharpe_ratio = annual_return / annual_volatility if annual_volatility > 0 else 0
        
        cummax = nav_df['nav'].cummax()
        drawdown = (nav_df['nav'] - cummax) / cummax
        max_drawdown = drawdown.min()
        
        sell_trades = [t for t in trades if t['action'] == 'sell']
        win_trades = len([t for t in sell_trades if t.get('profit', 0) > 0])
        win_rate = win_trades / len(sell_trades) if sell_trades else 0
        
        return {
            'nav_curve': nav_df,
            'trades': trades,
            'performance': {
                'initial_capital': self.initial_capital,
                'final_nav': nav_df['nav'].iloc[-1],
                'total_return': nav_df['nav'].iloc[-1] / self.initial_capital - 1,
                'annual_return': annual_return,
                'annual_volatility': annual_volatility,
                'sharpe_ratio': sharpe_ratio,
                'max_drawdown': max_drawdown,
                'total_trades': len(sell_trades),
                'win_rate': win_rate,
                'strategy_weights': self.weights
            }
        }
```

```python
class SignalFusion:
    """信号融合系统"""
    
    def __init__(self,
                 voting_threshold: float = 0.6,
                 confidence_weights: bool = True):
        """
        Args:
            voting_threshold: 投票阈值
            confidence_weights: 是否使用置信度加权
        """
        self.voting_threshold = voting_threshold
        self.confidence_weights = confidence_weights
    
    def fuse_signals(self, 
                    signals: pd.DataFrame,
                    confidences: pd.DataFrame = None) -> pd.Series:
        """
        融合多策略信号
        
        Args:
            signals: 各策略信号 DataFrame
            confidences: 各策略置信度 DataFrame（可选）
            
        Returns:
            融合后的信号序列
        """
        if self.confidence_weights and confidences is not None:
            # 置信度加权
            weighted_signals = signals * confidences
            fused_signal = weighted_signals.sum(axis=1) / confidences.sum(axis=1)
        else:
            # 简单投票
            buy_votes = (signals > 0).sum(axis=1)
            sell_votes = (signals < 0).sum(axis=1)
            total_votes = len(signals.columns)
            
            fused_signal = pd.Series(0, index=signals.index)
            fused_signal[buy_votes / total_votes >= self.voting_threshold] = 1
            fused_signal[sell_votes / total_votes >= self.voting_threshold] = -1
        
        return fused_signal
    
    def calculate_signal_confidence(self, 
                                   signals: pd.DataFrame) -> pd.DataFrame:
        """
        计算信号置信度
        
        Returns:
            各策略信号置信度
        """
        confidences = pd.DataFrame(index=signals.index, columns=signals.columns)
        
        for col in signals.columns:
            signal = signals[col]
            
            # 基于信号强度和一致性计算置信度
            # 信号强度：信号的绝对值
            strength = signal.abs()
            
            # 信号一致性：信号与移动平均的一致性
            consistency = (signal * signal.rolling(5).mean()).abs()
            
            # 综合置信度
            confidences[col] = (strength + consistency) / 2
        
        return confidences
```

```python
class MarketRegimeDetector:
    """市场状态识别系统"""
    
    def __init__(self):
        from sideways_detector import SidewaysMarketDetector
        self.sideways_detector = SidewaysMarketDetector()
    
    def detect_regime(self, df: pd.DataFrame) -> pd.DataFrame:
        """
        识别市场状态
        
        Returns:
            包含市场状态识别的DataFrame
        """
        # 震荡识别
        sideways = self.sideways_detector.detect_sideways(df)
        
        result = df.copy()
        result['sideways_score'] = sideways['sideways_score']
        result['sideways_state'] = sideways['sideways_state']
        result['adx'] = sideways['adx']
        result['atr_percentile'] = sideways['atr_percentile']
        result['bandwidth_percentile'] = sideways['bandwidth_percentile']
        
        # 波动率状态
        returns = result['close'].pct_change()
        result['volatility'] = returns.rolling(20).std() * np.sqrt(252)
        result['volatility_percentile'] = result['volatility'].rolling(252).rank(pct=True)
        
        # 市场状态分类
        result['market_regime'] = self._classify_regime(result)
        
        # 策略建议
        result['strategy_recommendation'] = self._recommend_strategy(result)
        
        return result
    
    def _classify_regime(self, df: pd.DataFrame) -> pd.Series:
        """分类市场状态"""
        regime = pd.Series('unknown', index=df.index)
        
        # 极强震荡
        regime.loc[df['sideways_score'] >= 0.8] = 'extreme_sideways'
        
        # 强震荡
        regime.loc[(df['sideways_score'] >= 0.6) & (df['sideways_score'] < 0.8)] = 'strong_sideways'
        
        # 弱震荡
        regime.loc[(df['sideways_score'] >= 0.4) & (df['sideways_score'] < 0.6)] = 'weak_sideways'
        
        # 趋势萌芽
        regime.loc[(df['sideways_score'] >= 0.2) & (df['sideways_score'] < 0.4)] = 'emerging_trend'
        
        # 强趋势
        regime.loc[df['sideways_score'] < 0.2] = 'strong_trend'
        
        return regime
    
    def _recommend_strategy(self, df: pd.DataFrame) -> pd.Series:
        """推荐策略"""
        recommendation = pd.Series('hold', index=df.index)
        
        # 极强震荡：网格交易为主
        recommendation.loc[df['market_regime'] == 'extreme_sideways'] = 'grid_trading'
        
        # 强震荡：网格+均值回归
        recommendation.loc[df['market_regime'] == 'strong_sideways'] = 'grid_mean_reversion'
        
        # 弱震荡：均值回归+波动率
        recommendation.loc[df['market_regime'] == 'weak_sideways'] = 'mean_reversion_volatility'
        
        # 趋势萌芽：波动率突破
        recommendation.loc[df['market_regime'] == 'emerging_trend'] = 'volatility_breakout'
        
        # 强趋势：趋势跟踪
        recommendation.loc[df['market_regime'] == 'strong_trend'] = 'trend_following'
        
        return recommendation
```

```python
class DynamicStrategySwitcher:
    """动态策略切换系统"""
    
    def __init__(self,
                 min_switch_interval: int = 20,
                 transition_buffer: int = 5):
        """
        Args:
            min_switch_interval: 最小切换间隔（天）
            transition_buffer: 过渡期缓冲（天）
        """
        self.min_switch_interval = min_switch_interval
        self.transition_buffer = transition_buffer
        self.last_switch_date = None
        self.current_strategy = None
    
    def should_switch_strategy(self,
                              current_regime: str,
                              current_date: pd.Timestamp,
                              current_nav: float,
                              initial_capital: float) -> Dict:
        """
        判断是否应该切换策略
        
        Returns:
            切换决策和原因
        """
        # 检查最小切换间隔
        if self.last_switch_date is not None:
            days_since_last_switch = (current_date - self.last_switch_date).days
            if days_since_last_switch < self.min_switch_interval:
                return {
                    'should_switch': False,
                    'reason': f'距离上次切换仅{days_since_last_switch}天，未达最小间隔'
                }
        
        # 检查回撤
        drawdown = (current_nav - initial_capital) / initial_capital
        if drawdown < -0.15:  # 回撤超过15%
            return {
                'should_switch': True,
                'reason': f'回撤{drawdown:.2%}超过阈值，切换到保守策略',
                'new_strategy': 'defensive'
            }
        
        # 根据市场状态切换
        strategy_map = {
            'extreme_sideways': 'grid_trading',
            'strong_sideways': 'grid_mean_reversion',
            'weak_sideways': 'mean_reversion_volatility',
            'emerging_trend': 'volatility_breakout',
            'strong_trend': 'trend_following'
        }
        
        recommended_strategy = strategy_map.get(current_regime, 'hold')
        
        if recommended_strategy != self.current_strategy:
            return {
                'should_switch': True,
                'reason': f'市场状态变为{current_regime}，推荐策略{recommended_strategy}',
                'new_strategy': recommended_strategy
            }
        
        return {
            'should_switch': False,
            'reason': '当前策略与市场状态匹配'
        }
    
    def execute_switch(self,
                      new_strategy: str,
                      current_date: pd.Timestamp):
        """执行策略切换"""
        self.current_strategy = new_strategy
        self.last_switch_date = current_date
```

### 2.3 过渡期处理

**过渡期风险管理**:

```python
class TransitionManager:
    """策略过渡期管理"""
    
    def __init__(self,
                 transition_period: int = 5,
                 max_position_reduction: float = 0.5):
        """
        Args:
            transition_period: 过渡期天数
            max_position_reduction: 最大仓位削减比例
        """
        self.transition_period = transition_period
        self.max_position_reduction = max_position_reduction
        self.in_transition = False
        self.transition_start_date = None
        self.transition_day = 0
    
    def start_transition(self, start_date: pd.Timestamp):
        """开始过渡期"""
        self.in_transition = True
        self.transition_start_date = start_date
        self.transition_day = 0
    
    def get_transition_position_adjustment(self) -> float:
        """
        获取过渡期仓位调整
        
        Returns:
            仓位调整比例（0-1）
        """
        if not self.in_transition:
            return 1.0
        
        self.transition_day += 1
        
        # 线性减少仓位
        reduction = self.max_position_reduction * (self.transition_day / self.transition_period)
        position_factor = 1.0 - reduction
        
        # 过渡期结束
        if self.transition_day >= self.transition_period:
            self.in_transition = False
            self.transition_day = 0
        
        return position_factor
```

```python
def backtest_combined_strategy(df: pd.DataFrame,
                               initial_capital: float = 100000,
                               commission_rate: float = 0.0003) -> Dict:
    """
    回测组合策略
    
    Returns:
        组合回测结果
    """
    from grid_trading import GridTradingBacktester
    from mean_reversion import BollingerBandsStrategy
    from volatility_strategy import VolatilityBreakoutStrategy
    
    # 初始化策略
    strategies = {
        'grid_trading': GridTradingBacktester(initial_capital * 0.3),
        'mean_reversion': BollingerBandsStrategy(lookback=20, num_std=2.0),
        'volatility_breakout': VolatilityBreakoutStrategy(lookback=20, vol_threshold=2.0)
    }
    
    # 初始化组合
    portfolio = MultiStrategyPortfolio(strategies, initial_capital)
    
    # 市场状态识别
    regime_detector = MarketRegimeDetector()
    regime = regime_detector.detect_regime(df)
    
    # 回测
    result = portfolio.backtest_portfolio(df, commission_rate)
    
    # 添加市场状态信息
    result['regime'] = regime
    
    return result
```

```python
def parameter_sensitivity_analysis(df: pd.DataFrame,
                                  param_ranges: Dict) -> pd.DataFrame:
    """
    参数敏感性分析
    
    Args:
        df: 价格数据
        param_ranges: 参数范围字典
        
    Returns:
        参数敏感性分析结果
    """
    results = []
    
    # 网格交易参数敏感性
    for grid_count in param_ranges['grid_count']:
        for vol_mult in param_ranges['volatility_multiplier']:
            result = backtest_combined_strategy(df)
            results.append({
                'grid_count': grid_count,
                'vol_mult': vol_mult,
                'annual_return': result['performance']['annual_return'],
                'sharpe_ratio': result['performance']['sharpe_ratio']
            })
    
    results_df = pd.DataFrame(results)
    
    # 计算敏感性系数
    sensitivity = {}
    for param in ['grid_count', 'vol_mult']:
        param_std = results_df.groupby(param)['annual_return'].std().mean()
        sensitivity[param] = param_std
    
    return results_df, sensitivity
```
