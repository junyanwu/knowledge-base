### 1.1 框架设计

```python
import numpy as np
import pandas as pd
from scipy.optimize import minimize
from typing import Dict, List, Tuple, Optional, Union
from dataclasses import dataclass, field
from enum import Enum
import warnings
warnings.filterwarnings('ignore')

class RebalanceFrequency(Enum):
    """再平衡频率"""
    DAILY = 'D'
    WEEKLY = 'W'
    MONTHLY = 'M'
    QUARTERLY = 'Q'
    ANNUALLY = 'A'

class OptimizationMethod(Enum):
    """优化方法"""
    MEAN_VARIANCE = 'mean_variance'
    RISK_PARITY = 'risk_parity'
    BLACK_LITTERMAN = 'black_litterman'
    MIN_VARIANCE = 'min_variance'
    MAX_SHARPE = 'max_sharpe'
    EQUAL_WEIGHT = 'equal_weight'

@dataclass
class BacktestConfig:
    """回测配置"""
    start_date: str = '2015-01-01'
    end_date: str = '2025-12-31'
    initial_capital: float = 1000000.0
    rebalance_freq: RebalanceFrequency = RebalanceFrequency.MONTHLY
    lookback_window: int = 252
    transaction_cost: float = 0.001  # 千分之一
    slippage: float = 0.0005  # 万分之五
    min_weight: float = 0.0
    max_weight: float = 1.0
    risk_free_rate: float = 0.02

@dataclass
class PerformanceMetrics:
    """绩效指标"""
    total_return: float = 0.0
    annual_return: float = 0.0
    annual_volatility: float = 0.0
    sharpe_ratio: float = 0.0
    sortino_ratio: float = 0.0
    max_drawdown: float = 0.0
    calmar_ratio: float = 0.0
    win_rate: float = 0.0
    profit_loss_ratio: float = 0.0
    information_ratio: float = 0.0
    tracking_error: float = 0.0
    turnover: float = 0.0

class PortfolioBacktestFramework:
    """组合回测框架"""
    
    def __init__(self, prices: pd.DataFrame, config: BacktestConfig):
        """
        初始化
        
        参数:
        prices: 价格数据
        config: 回测配置
        """
        self.prices = prices
        self.config = config
        self.returns = prices.pct_change().dropna()
        self.assets = prices.columns.tolist()
        self.n_assets = len(self.assets)
        
        # 回测状态
        self.portfolio_values = []
        self.weights_history = []
        self.trades_history = []
        self.daily_returns = []
        
    def run_backtest(self, 
                     optimization_method: OptimizationMethod,
                     views: Optional[Dict] = None,
                     risk_budget: Optional[np.ndarray] = None) -> PerformanceMetrics:
        """
        运行回测
        
        参数:
        optimization_method: 优化方法
        views: Black-Litterman观点
        risk_budget: 风险预算
        
        返回:
        绩效指标
        """
        # 初始化
        self.portfolio_values = [self.config.initial_capital]
        self.weights_history = []
        self.trades_history = []
        self.daily_returns = []
        
        current_weights = None
        last_rebalance = -self.config.lookback_window
        
        # 回测循环
        for i in range(self.config.lookback_window, len(self.returns)):
            current_date = self.returns.index[i]
            
            # 检查是否需要再平衡
            need_rebalance = self._check_rebalance(i, last_rebalance)
            
            if need_rebalance:
                # 获取历史数据
                hist_returns = self.returns.iloc[i-self.config.lookback_window:i]
                
                # 优化权重
                target_weights = self._optimize_weights(
                    hist_returns, optimization_method, views, risk_budget
                )
                
                # 计算交易成本
                if current_weights is not None:
                    turnover = np.abs(target_weights - current_weights).sum()
                    trade_cost = turnover * self.config.transaction_cost
                    trade_cost += turnover * self.config.slippage
                else:
                    turnover = 0.0
                    trade_cost = 0.0
                
                current_weights = target_weights
                last_rebalance = i
                
                # 记录
                self.weights_history.append({
                    'date': current_date,
                    'weights': target_weights,
                    'turnover': turnover,
                    'trade_cost': trade_cost
                })
            
            # 计算当日收益
            if current_weights is not None:
                daily_return = (current_weights * self.returns.iloc[i]).sum()
                self.daily_returns.append(daily_return)
                
                # 更新组合价值
                new_value = self.portfolio_values[-1] * (1 + daily_return)
                if len(self.weights_history) > 0 and 'trade_cost' in self.weights_history[-1]:
                    new_value *= (1 - self.weights_history[-1]['trade_cost'])
                self.portfolio_values.append(new_value)
        
        # 计算绩效指标
        metrics = self._calculate_metrics()
        
        return metrics
    
    def _check_rebalance(self, current_idx: int, last_rebalance: int) -> bool:
        """检查是否需要再平衡"""
        freq_days = {
            RebalanceFrequency.DAILY: 1,
            RebalanceFrequency.WEEKLY: 5,
            RebalanceFrequency.MONTHLY: 20,
            RebalanceFrequency.QUARTERLY: 60,
            RebalanceFrequency.ANNUALLY: 252
        }
        
        days_since_last = current_idx - last_rebalance
        return days_since_last >= freq_days.get(self.config.rebalance_freq, 20)
    
    def _optimize_weights(self, 
                         hist_returns: pd.DataFrame,
                         method: OptimizationMethod,
                         views: Optional[Dict],
                         risk_budget: Optional[np.ndarray]) -> np.ndarray:
        """优化权重"""
        # 计算统计量
        expected_returns = hist_returns.mean() * 252
        cov_matrix = hist_returns.cov() * 252
        
        if method == OptimizationMethod.EQUAL_WEIGHT:
            return np.ones(self.n_assets) / self.n_assets
        
        elif method == OptimizationMethod.MIN_VARIANCE:
            return self._min_variance(cov_matrix)
        
        elif method == OptimizationMethod.MAX_SHARPE:
            return self._max_sharpe(expected_returns, cov_matrix)
        
        elif method == OptimizationMethod.MEAN_VARIANCE:
            return self._mean_variance(expected_returns, cov_matrix)
        
        elif method == OptimizationMethod.RISK_PARITY:
            return self._risk_parity(cov_matrix, risk_budget)
        
        elif method == OptimizationMethod.BLACK_LITTERMAN:
            return self._black_litterman(expected_returns, cov_matrix, views)
        
        return np.ones(self.n_assets) / self.n_assets
    
    def _min_variance(self, cov_matrix: np.ndarray) -> np.ndarray:
        """最小方差组合"""
        def objective(w):
            return w @ cov_matrix @ w
        
        constraints = {'type': 'eq', 'fun': lambda w: np.sum(w) - 1}
        bounds = [(self.config.min_weight, self.config.max_weight) 
                  for _ in range(self.n_assets)]
        
        result = minimize(
            objective,
            x0=np.ones(self.n_assets) / self.n_assets,
            method='SLSQP',
            bounds=bounds,
            constraints=constraints
        )
        
        return result.x
    
    def _max_sharpe(self, expected_returns: np.ndarray, 
                    cov_matrix: np.ndarray) -> np.ndarray:
        """最大夏普比率组合"""
        def neg_sharpe(w):
            ret = w @ expected_returns
            vol = np.sqrt(w @ cov_matrix @ w)
            return -(ret - self.config.risk_free_rate) / vol
        
        constraints = {'type': 'eq', 'fun': lambda w: np.sum(w) - 1}
        bounds = [(self.config.min_weight, self.config.max_weight) 
                  for _ in range(self.n_assets)]
        
        result = minimize(
            neg_sharpe,
            x0=np.ones(self.n_assets) / self.n_assets,
            method='SLSQP',
            bounds=bounds,
            constraints=constraints
        )
        
        return result.x
    
    def _mean_variance(self, expected_returns: np.ndarray,
                       cov_matrix: np.ndarray,
                       risk_aversion: float = 2.5) -> np.ndarray:
        """均值方差优化"""
        def neg_utility(w):
            ret = w @ expected_returns
            var = w @ cov_matrix @ w
            return -(ret - 0.5 * risk_aversion * var)
        
        constraints = {'type': 'eq', 'fun': lambda w: np.sum(w) - 1}
        bounds = [(self.config.min_weight, self.config.max_weight) 
                  for _ in range(self.n_assets)]
        
        result = minimize(
            neg_utility,
            x0=np.ones(self.n_assets) / self.n_assets,
            method='SLSQP',
            bounds=bounds,
            constraints=constraints
        )
        
        return result.x
    
    def _risk_parity(self, cov_matrix: np.ndarray,
                     risk_budget: Optional[np.ndarray] = None) -> np.ndarray:
        """风险平价"""
        if risk_budget is None:
            risk_budget = np.ones(self.n_assets) / self.n_assets
        
        def risk_parity_objective(w):
            portfolio_vol = np.sqrt(w @ cov_matrix @ w)
            marginal_risk = cov_matrix @ w / portfolio_vol
            risk_contrib = w * marginal_risk
            risk_contrib_pct = risk_contrib / portfolio_vol
            return np.sum((risk_contrib_pct - risk_budget) ** 2)
        
        constraints = {'type': 'eq', 'fun': lambda w: np.sum(w) - 1}
        bounds = [(0.001, 1) for _ in range(self.n_assets)]
        
        result = minimize(
            risk_parity_objective,
            x0=np.ones(self.n_assets) / self.n_assets,
            method='SLSQP',
            bounds=bounds,
            constraints=constraints
        )
        
        return result.x
    
    def _black_litterman(self, expected_returns: np.ndarray,
                         cov_matrix: np.ndarray,
                         views: Optional[Dict]) -> np.ndarray:
        """Black-Litterman模型"""
        # 市场权重（等权）
        market_weights = np.ones(self.n_assets) / self.n_assets
        
        # 风险厌恶系数
        market_return = market_weights @ expected_returns
        market_vol = np.sqrt(market_weights @ cov_matrix @ market_weights)
        delta = (market_return - self.config.risk_free_rate) / (market_vol ** 2)
        
        # 市场均衡收益
        pi = delta * cov_matrix @ market_weights + self.config.risk_free_rate
        
        if views is not None and 'P' in views and 'Q' in views:
            P = views['P']
            Q = views['Q']
            omega = views.get('omega', np.eye(len(Q)) * 0.01)
            tau = views.get('tau', 0.025)
            
            # Black-Litterman公式
            term1 = np.linalg.inv(tau * cov_matrix) + P.T @ np.linalg.inv(omega) @ P
            term2 = np.linalg.inv(tau * cov_matrix) @ pi + P.T @ np.linalg.inv(omega) @ Q
            
            posterior_returns = np.linalg.inv(term1) @ term2
            posterior_cov = np.linalg.inv(term1)
        else:
            posterior_returns = pi
            posterior_cov = cov_matrix
        
        # 使用后验收益优化
        return self._max_sharpe(posterior_returns, posterior_cov)
    
    def _calculate_metrics(self) -> PerformanceMetrics:
        """计算绩效指标"""
        if len(self.daily_returns) == 0:
            return PerformanceMetrics()
        
        returns = pd.Series(self.daily_returns)
        
        # 年化收益
        total_return = self.portfolio_values[-1] / self.portfolio_values[0] - 1
        n_years = len(self.daily_returns) / 252
        annual_return = (1 + total_return) ** (1 / n_years) - 1
        
        # 年化波动
        annual_vol = returns.std() * np.sqrt(252)
        
        # 夏普比率
        sharpe = (annual_return - self.config.risk_free_rate) / annual_vol if annual_vol > 0 else 0
        
        # 下行波动
        downside_returns = returns[returns < 0]
        downside_vol = downside_returns.std() * np.sqrt(252) if len(downside_returns) > 0 else 0
        sortino = (annual_return - self.config.risk_free_rate) / downside_vol if downside_vol > 0 else 0
        
        # 最大回撤
        portfolio_series = pd.Series(self.portfolio_values)
        cumulative_max = portfolio_series.cummax()
        drawdown = (portfolio_series - cumulative_max) / cumulative_max
        max_drawdown = drawdown.min()
        
        # Calmar比率
        calmar = annual_return / abs(max_drawdown) if max_drawdown != 0 else 0
        
        # 胜率
        win_rate = (returns > 0).sum() / len(returns)
        
        # 盈亏比
        avg_win = returns[returns > 0].mean() if (returns > 0).any() else 0
        avg_loss = abs(returns[returns < 0].mean()) if (returns < 0).any() else 1
        profit_loss_ratio = avg_win / avg_loss if avg_loss > 0 else 0
        
        # 换手率
        if len(self.weights_history) > 1:
            turnovers = [t['turnover'] for t in self.weights_history if 'turnover' in t]
            avg_turnover = np.mean(turnovers) * 252 / len(turnovers) if turnovers else 0
        else:
            avg_turnover = 0
        
        return PerformanceMetrics(
            total_return=total_return,
            annual_return=annual_return,
            annual_volatility=annual_vol,
            sharpe_ratio=sharpe,
            sortino_ratio=sortino,
            max_drawdown=max_drawdown,
            calmar_ratio=calmar,
            win_rate=win_rate,
            profit_loss_ratio=profit_loss_ratio,
            turnover=avg_turnover
        )
    
    def get_portfolio_series(self) -> pd.Series:
        """获取组合净值序列"""
        dates = [self.returns.index[0] - pd.Timedelta(days=1)] + \
                self.returns.index[self.config.lookback_window:].tolist()
        return pd.Series(self.portfolio_values, index=dates[:len(self.portfolio_values)])

def prepare_backtest_data(assets: List[str], 
                          start_date: str = '2015-01-01',
                          end_date: str = '2025-12-31') -> pd.DataFrame:
    """准备回测数据（模拟）"""
    np.random.seed(42)
    dates = pd.date_range(start_date, end_date, freq='B')
    
    # 模拟不同资产特征
    asset_params = {
        '股票ETF': {'return': 0.08, 'vol': 0.20, 'beta': 1.0},
        '债券ETF': {'return': 0.04, 'vol': 0.05, 'beta': 0.1},
        '商品ETF': {'return': 0.05, 'vol': 0.18, 'beta': 0.3},
        'REITsETF': {'return': 0.06, 'vol': 0.15, 'beta': 0.7},
        '海外ETF': {'return': 0.07, 'vol': 0.22, 'beta': 0.8}
    }
    
    prices = {}
    initial_prices = {'股票ETF': 3.0, '债券ETF': 100.0, '商品ETF': 5.0, 
                       'REITsETF': 2.5, '海外ETF': 1.5}
    
    # 生成市场因子
    market = np.random.normal(0.0003, 0.01, len(dates))
    
    for asset in assets:
        params = asset_params.get(asset, {'return': 0.05, 'vol': 0.15, 'beta': 0.5})
        # 包含市场因子和特质收益
        asset_returns = params['beta'] * market + \
                       (1 - params['beta']) * np.random.normal(
                           params['return']/252, params['vol']/np.sqrt(252), len(dates)
                       )
        prices[asset] = initial_prices.get(asset, 1.0) * (1 + asset_returns).cumprod()
    
    return pd.DataFrame(prices, index=dates)
```

### 1.2 绩效评估指标体系

```python
class PerformanceAnalyzer:
    """绩效分析器"""
    
    def __init__(self, returns: pd.Series, benchmark_returns: Optional[pd.Series] = None):
        """
        初始化
        
        参数:
        returns: 策略收益率
        benchmark_returns: 基准收益率
        """
        self.returns = returns
        self.benchmark_returns = benchmark_returns
    
    def calculate_all_metrics(self, risk_free_rate: float = 0.02) -> Dict:
        """计算所有指标"""
        metrics = {}
        
        # 收益指标
        metrics['total_return'] = (1 + self.returns).prod() - 1
        metrics['annual_return'] = self.returns.mean() * 252
        metrics['monthly_return'] = self.returns.mean() * 21
        
        # 风险指标
        metrics['annual_volatility'] = self.returns.std() * np.sqrt(252)
        metrics['downside_volatility'] = self.returns[self.returns < 0].std() * np.sqrt(252)
        
        # 风险调整收益
        metrics['sharpe_ratio'] = (metrics['annual_return'] - risk_free_rate) / metrics['annual_volatility']
        metrics['sortino_ratio'] = (metrics['annual_return'] - risk_free_rate) / metrics['downside_volatility']
        
        # 回撤指标
        cumulative = (1 + self.returns).cumprod()
        running_max = cumulative.cummax()
        drawdown = (cumulative - running_max) / running_max
        metrics['max_drawdown'] = drawdown.min()
        metrics['avg_drawdown'] = drawdown.mean()
        metrics['drawdown_duration'] = self._calculate_drawdown_duration(drawdown)
        
        # Calmar比率
        metrics['calmar_ratio'] = metrics['annual_return'] / abs(metrics['max_drawdown'])
        
        # 交易指标
        metrics['win_rate'] = (self.returns > 0).sum() / len(self.returns)
        metrics['avg_win'] = self.returns[self.returns > 0].mean() if (self.returns > 0).any() else 0
        metrics['avg_loss'] = self.returns[self.returns < 0].mean() if (self.returns < 0).any() else 0
        metrics['profit_loss_ratio'] = abs(metrics['avg_win'] / metrics['avg_loss']) if metrics['avg_loss'] != 0 else 0
        
        # 尾部风险
        metrics['var_95'] = np.percentile(self.returns, 5)
        metrics['var_99'] = np.percentile(self.returns, 1)
        metrics['cvar_95'] = self.returns[self.returns <= metrics['var_95']].mean()
        metrics['cvar_99'] = self.returns[self.returns <= metrics['var_99']].mean()
        
        # 基准相关指标
        if self.benchmark_returns is not None:
            excess_returns = self.returns - self.benchmark_returns
            metrics['alpha'] = excess_returns.mean() * 252
            metrics['tracking_error'] = excess_returns.std() * np.sqrt(252)
            metrics['information_ratio'] = metrics['alpha'] / metrics['tracking_error'] if metrics['tracking_error'] > 0 else 0
            
            # Beta
            cov_matrix = np.cov(self.returns, self.benchmark_returns)
            metrics['beta'] = cov_matrix[0, 1] / cov_matrix[1, 1]
            
            # Treynor比率
            metrics['treynor_ratio'] = (metrics['annual_return'] - risk_free_rate) / metrics['beta'] if metrics['beta'] != 0 else 0
        
        return metrics
    
    def _calculate_drawdown_duration(self, drawdown: pd.Series) -> int:
        """计算回撤持续时间"""
        in_drawdown = drawdown < 0
        if not in_drawdown.any():
            return 0
        
        # 计算最长连续回撤天数
        drawdown_periods = in_drawdown.astype(int).groupby(
            (~in_drawdown).cumsum()
        ).sum()
        return drawdown_periods.max()
    
    def rolling_performance(self, window: int = 252) -> pd.DataFrame:
        """滚动绩效"""
        rolling_return = self.returns.rolling(window).mean() * 252
        rolling_vol = self.returns.rolling(window).std() * np.sqrt(252)
        rolling_sharpe = (rolling_return - 0.02) / rolling_vol
        
        return pd.DataFrame({
            'rolling_return': rolling_return,
            'rolling_volatility': rolling_vol,
            'rolling_sharpe': rolling_sharpe
        })
    
    def monthly_returns(self) -> pd.DataFrame:
        """月度收益"""
        monthly = self.returns.resample('M').apply(lambda x: (1 + x).prod() - 1)
        monthly_table = monthly.groupby([monthly.index.year, monthly.index.month]).mean()
        monthly_table.index.names = ['Year', 'Month']
        return monthly_table.unstack()
    
    def yearly_performance(self) -> pd.DataFrame:
        """年度绩效"""
        yearly = self.returns.resample('Y')
        
        results = []
        for year, group in yearly:
            year_return = (1 + group).prod() - 1
            year_vol = group.std() * np.sqrt(252)
            year_sharpe = year_return / year_vol if year_vol > 0 else 0
            
            cumulative = (1 + group).cumprod()
            running_max = cumulative.cummax()
            drawdown = (cumulative - running_max) / running_max
            max_dd = drawdown.min()
            
            results.append({
                'year': year.year,
                'return': year_return,
                'volatility': year_vol,
                'sharpe': year_sharpe,
                'max_drawdown': max_dd
            })
        
        return pd.DataFrame(results).set_index('year')

def example_performance_analysis():
    """绩效分析示例"""
    np.random.seed(42)
    dates = pd.date_range('2015-01-01', '2025-12-31', freq='B')
    returns = pd.Series(np.random.normal(0.0005, 0.015, len(dates)), index=dates)
    benchmark = pd.Series(np.random.normal(0.0004, 0.012, len(dates)), index=dates)
    
    analyzer = PerformanceAnalyzer(returns, benchmark)
    metrics = analyzer.calculate_all_metrics()
    
    print("绩效指标:")
    for key, value in metrics.items():
        if isinstance(value, float):
            print(f"  {key}: {value:.4f}")
    
    return analyzer, metrics
```

```python
class ModelComparisonFramework:
    """模型对比框架"""
    
    def __init__(self, prices: pd.DataFrame, config: BacktestConfig):
        """
        初始化
        
        参数:
        prices: 价格数据
        config: 回测配置
        """

        self.config = config
        self.framework = PortfolioBacktestFramework(prices, config)
        self.results = {}
    
    def run_all_models(self, views: Optional[Dict] = None) -> Dict:
        """运行所有模型"""
        models = [
            OptimizationMethod.EQUAL_WEIGHT,
            OptimizationMethod.MIN_VARIANCE,
```

```python
class MarketEnvironmentAnalysis:
    """不同市场环境分析"""
    
    def __init__(self, prices: pd.DataFrame, config: BacktestConfig):
        self.prices = prices
        self.config = config
        self.returns = prices.pct_change().dropna()
    
    def identify_market_regimes(self, window: int = 60) -> pd.Series:
        """识别市场状态"""
        # 使用滚动收益和波动率
        rolling_return = self.returns.mean(axis=1).rolling(window).mean() * 252
        rolling_vol = self.returns.mean(axis=1).rolling(window).std() * np.sqrt(252)
        
        # 定义市场状态
        regimes = pd.Series('sideways', index=self.returns.index)
        
        # 牛市：高收益、低波动
        regimes[(rolling_return > 0.10) & (rolling_vol < 0.15)] = 'bull'
        
        # 熊市：低收益、高波动
        regimes[(rolling_return < -0.05) | (rolling_vol > 0.25)] = 'bear'
        
        # 危机：极端波动
        regimes[rolling_vol > 0.30] = 'crisis'
        
        return regimes
    
    def analyze_by_regime(self, method: OptimizationMethod) -> Dict:
        """按市场状态分析"""
        # 识别状态
        regimes = self.identify_market_regimes()
        
        # 运行回测
        framework = PortfolioBacktestFramework(self.prices, self.config)
        metrics = framework.run_backtest(method)
        
        # 获取收益序列
        portfolio_returns = pd.Series(framework.daily_returns, 
                                      index=self.returns.index[self.config.lookback_window:])
        
        # 按状态分组分析
        results = {}
        for regime in ['bull', 'bear', 'sideways', 'crisis']:
            regime_mask = regimes.iloc[self.config.lookback_window:] == regime
            if regime_mask.sum() > 0:
                regime_returns = portfolio_returns[regime_mask]
                results[regime] = {
                    'count': regime_mask.sum(),
                    'return': regime_returns.mean() * 252,
                    'volatility': regime_returns.std() * np.sqrt(252),
                    'sharpe': regime_returns.mean() / regime_returns.std() * np.sqrt(252) if regime_returns.std() > 0 else 0
                }
        
        return results
    
    def compare_across_regimes(self) -> pd.DataFrame:
        """跨市场状态对比"""
        models = [
            OptimizationMethod.EQUAL_WEIGHT,
            OptimizationMethod.MIN_VARIANCE,
            OptimizationMethod.RISK_PARITY,
            OptimizationMethod.BLACK_LITTERMAN
        ]
        
        results = []
        for model in models:
            regime_results = self.analyze_by_regime(model)
            for regime, metrics in regime_results.items():
                results.append({
                    '模型': model.value,
                    '市场状态': regime,
                    '天数': metrics['count'],
                    '年化收益': f"{metrics['return']:.2%}",
                    '年化波动': f"{metrics['volatility']:.2%}",
                    '夏普比率': f"{metrics['sharpe']:.4f}"
                })
        
        return pd.DataFrame(results)
```

```python
class BootstrapValidator:
    """Bootstrap检验"""
    
    def __init__(self, prices: pd.DataFrame, config: BacktestConfig, n_bootstrap: int = 100):
        self.prices = prices
        self.config = config
        self.n_bootstrap = n_bootstrap
    
    def bootstrap_backtest(self, method: OptimizationMethod) -> Dict:
        """Bootstrap回测"""
        results = []
        
        for i in range(self.n_bootstrap):
            # 重采样
            bootstrap_returns = self.prices.pct_change().sample(
                n=len(self.prices), replace=True
            )
            bootstrap_prices = (1 + bootstrap_returns).cumprod()
            bootstrap_prices.columns = self.prices.columns
            
            # 回测
            framework = PortfolioBacktestFramework(bootstrap_prices, self.config)
            metrics = framework.run_backtest(method)
            
            results.append({
                'annual_return': metrics.annual_return,
                'sharpe_ratio': metrics.sharpe_ratio,
                'max_drawdown': metrics.max_drawdown
            })
        
        # 计算统计量
        returns = [r['annual_return'] for r in results]
        sharpes = [r['sharpe_ratio'] for r in results]
        drawdowns = [r['max_drawdown'] for r in results]
        
        return {
            'return_mean': np.mean(returns),
            'return_std': np.std(returns),
            'return_ci': (np.percentile(returns, 2.5), np.percentile(returns, 97.5)),
            'sharpe_mean': np.mean(sharpes),
            'sharpe_std': np.std(sharpes),
            'sharpe_ci': (np.percentile(sharpes, 2.5), np.percentile(sharpes, 97.5)),
            'drawdown_mean': np.mean(drawdowns),
            'drawdown_std': np.std(drawdowns)
        }
    
    def compare_models_bootstrap(self, 
                                  methods: List[OptimizationMethod] = None) -> pd.DataFrame:
        """Bootstrap模型对比"""
        if methods is None:
            methods = [
                OptimizationMethod.EQUAL_WEIGHT,
                OptimizationMethod.MIN_VARIANCE,
                OptimizationMethod.RISK_PARITY,
                OptimizationMethod.BLACK_LITTERMAN
            ]
        
        results = []
        for method in methods:
            print(f"Bootstrap检验 {method.value}...")
            bootstrap_result = self.bootstrap_backtest(method)
            
            results.append({
                '模型': method.value,
                '年化收益均值': f"{bootstrap_result['return_mean']:.2%}",
                '年化收益标准差': f"{bootstrap_result['return_std']:.2%}",
                '年化收益95%CI': f"({bootstrap_result['return_ci'][0]:.2%}, {bootstrap_result['return_ci'][1]:.2%})",
                '夏普均值': f"{bootstrap_result['sharpe_mean']:.4f}",

                '最大回撤均值': f"{bootstrap_result['drawdown_mean']:.2%}"
            })
        
        return pd.DataFrame(results)
```

```python
class WalkForwardValidator:
    """Walk-Forward检验"""
    
    def __init__(self, prices: pd.DataFrame, 
                 train_window: int = 252,
                 test_window: int = 63):
        self.prices = prices
        self.train_window = train_window
        self.test_window = test_window
        self.returns = prices.pct_change().dropna()
    
    def walk_forward_test(self, method: OptimizationMethod) -> Dict:
        """Walk-Forward测试"""
        n_periods = len(self.returns)
        results = []
        
        for i in range(self.train_window, n_periods - self.test_window, self.test_window):
            # 训练期
            train_prices = self.prices.iloc[i-self.train_window:i]
            
            # 测试期
            test_returns = self.returns.iloc[i:i+self.test_window]
            
            # 训练模型
            config = BacktestConfig(lookback_window=self.train_window)
            framework = PortfolioBacktestFramework(train_prices, config)
            
            # 获取权重
            hist_returns = train_prices.pct_change().dropna()
            weights = framework._optimize_weights(hist_returns, method, None, None)
            
            # 测试
            portfolio_returns = (weights * test_returns).sum(axis=1)
            
            results.append({
                'period': i,
                'return': portfolio_returns.mean() * 252,
                'volatility': portfolio_returns.std() * np.sqrt(252),
                'sharpe': portfolio_returns.mean() / portfolio_returns.std() * np.sqrt(252)
            })
        
        # 汇总
        returns = [r['return'] for r in results]
        volatilities = [r['volatility'] for r in results]
        sharpes = [r['sharpe'] for r in results]
        
        return {
            'n_periods': len(results),
            'avg_return': np.mean(returns),
            'avg_volatility': np.mean(volatilities),
            'avg_sharpe': np.mean(sharpes),
            'return_consistency': np.mean([r > 0 for r in returns]),
            'sharpe_consistency': np.mean([s > 0 for s in sharpes])
        }
    
    def compare_walk_forward(self, 
                             methods: List[OptimizationMethod] = None) -> pd.DataFrame:
        """Walk-Forward模型对比"""

                OptimizationMethod.EQUAL_WEIGHT,

                OptimizationMethod.BLACK_LITTERMAN
            ]
        
        results = []
        for method in methods:

            wf_result = self.walk_forward_test(method)
            
            results.append({
                '模型': method.value,
                '测试期数': wf_result['n_periods'],
                '平均年化收益': f"{wf_result['avg_return']:.2%}",

                '平均夏普比率': f"{wf_result['avg_sharpe']:.4f}",
                '正收益比例': f"{wf_result['return_consistency']:.2%}",
                '正夏普比例': f"{wf_result['sharpe_consistency']:.2%}"
            })
        
        return pd.DataFrame(results)
```

```python
class ModelSelector:
    """模型选择器"""
    
    def __init__(self, comparison_results: Dict[str, PerformanceMetrics]):
        self.results = comparison_results
    
    def select_by_sharpe(self) -> str:
        """按夏普比率选择"""
        return max(self.results.items(), key=lambda x: x[1].sharpe_ratio)[0]
    
    def select_by_calmar(self) -> str:
        """按卡玛比率选择"""
        return max(self.results.items(), key=lambda x: x[1].calmar_ratio)[0]
    
    def select_by_drawdown(self, max_dd_limit: float = -0.15) -> str:
        """按最大回撤选择"""
        valid_models = {k: v for k, v in self.results.items() 
                        if v.max_drawdown > max_dd_limit}
        if valid_models:
            return max(valid_models.items(), key=lambda x: x[1].sharpe_ratio)[0]
        return min(self.results.items(), key=lambda x: x[1].max_drawdown)[0]
    
    def select_multi_criteria(self, 
                              weights: Dict[str, float] = None) -> str:
        """多准则选择"""
        if weights is None:
            weights = {
                'sharpe': 0.35,
                'calmar': 0.25,
                'max_drawdown': 0.25,
                'turnover': 0.15
            }
        
        scores = {}
        for model_name, metrics in self.results.items():
            # 归一化指标
            sharpe_score = metrics.sharpe_ratio / max(m.sharpe_ratio for m in self.results.values())
            calmar_score = metrics.calmar_ratio / max(m.calmar_ratio for m in self.results.values()) if max(m.calmar_ratio for m in self.results.values()) != 0 else 0
            dd_score = abs(metrics.max_drawdown) / max(abs(m.max_drawdown) for m in self.results.values())
            turnover_score = 1 - (metrics.turnover / max(m.turnover for m in self.results.values()) if max(m.turnover for m in self.results.values()) > 0 else 0)
            
            # 综合得分
            scores[model_name] = (
                weights['sharpe'] * sharpe_score +
                weights['calmar'] * calmar_score +
                weights['max_drawdown'] * (1 - dd_score) +
                weights['turnover'] * turnover_score
            )
        
        return max(scores.items(), key=lambda x: x[1])[0]
    
    def get_recommendation(self) -> Dict:
        """获取推荐"""
        return {
            '最佳夏普': self.select_by_sharpe(),
            '最佳卡玛': self.select_by_calmar(),
            '最低回撤': self.select_by_drawdown(),
            '多准则最优': self.select_multi_criteria(),
            '适用场景': {
                '牛市': self.select_by_sharpe(),
                '熊市': self.select_by_drawdown(-0.10),
                '震荡市': self.select_multi_criteria({'sharpe': 0.25, 'calmar': 0.35, 'max_drawdown': 0.30, 'turnover': 0.10})
            }
        }

    # 使用之前的对比结果

    prices = prepare_backtest_data(assets)
```

## 5. 总结与实践要点

### 5.1 核心要点

1. **回测框架设计**
   - 标准化回测流程
   - 多模型对比机制
   - 完善的绩效评估

2. **模型对比**
   - 均值方差：收益导向，可能过度集中
   - 风险平价：风险分散，稳定性好
   - Black-Litterman：融入观点，灵活性高

3. **鲁棒性检验**
   - 参数敏感性：回溯期、再平衡频率、交易成本
   - Bootstrap：统计显著性检验
   - Walk-Forward：样本外验证

4. **模型选择**
   - 多准则综合评价
   - 区分不同投资者类型
   - 考虑市场环境适应性

### 5.2 实践要点

1. **数据质量**
   - 确保数据准确、完整
   - 处理缺失值和异常值
   - 考虑存续偏差

2. **模型设定**
   - 合理的回溯期（1-2年）
   - 适度的再平衡频率（月度）
   - 考虑交易成本

3. **风险控制**
   - 设置权重上下限
   - 控制最大回撤
   - 动态调整杠杆

4. **持续监控**
   - 定期评估模型有效性
   - 跟踪市场环境变化
   - 及时调整配置

### 5.3 模型适用场景

| 模型 | 适用场景 | 优势 | 劣势 |
|------|----------|------|------|
| 均值方差 | 牛市、趋势明确 | 收益最大化 | 集中度高 |
| 最小方差 | 熊市、风险厌恶 | 风险最低 | 收益较低 |
| 风险平价 | 全天候、长期投资 | 风险分散 | 需要杠杆 |
| Black-Litterman | 有观点、灵活配置 | 观点融合 | 参数敏感 |
