# 多因子模型理论 - 模块5：机器学习因子挖掘

## 1. 机器学习因子挖掘概述

### 1.1 传统方法 vs 机器学习

| 维度 | 传统方法 | 机器学习 |
|------|---------|---------|
| 因子发现 | 经济学逻辑 | 数据驱动 |
| 非线性 | 难以捕捉 | 自然处理 |
| 交互效应 | 需手动构造 | 自动学习 |
| 过拟合 | 相对可控 | 风险较高 |
| 解释性 | 强 | 弱 |

### 1.2 机器学习因子挖掘流程

```
数据准备 → 特征工程 → 模型训练 → 因子提取 → 因子检验 → 组合构建
```

## 2. 特征工程

### 2.1 基础特征

**价格特征：**
```python
def price_features(df):
    """
    价格相关特征
    
    参数:
    df: 包含open, high, low, close, volume的DataFrame
    
    返回:
    特征DataFrame
    """
    features = pd.DataFrame()
    
    # 收益率
    features['return_1d'] = df['close'].pct_change(1)
    features['return_5d'] = df['close'].pct_change(5)
    features['return_20d'] = df['close'].pct_change(20)
    
    # 波动率
    features['volatility_5d'] = df['return_1d'].rolling(5).std()
    features['volatility_20d'] = df['return_1d'].rolling(20).std()
    
    # 价格位置
    features['price_pos_20d'] = (df['close'] - df['low'].rolling(20).min()) / \
                                 (df['high'].rolling(20).max() - df['low'].rolling(20).min())
    
    # 动量
    features['momentum_5d'] = df['close'] / df['close'].shift(5) - 1
    features['momentum_20d'] = df['close'] / df['close'].shift(20) - 1
    
    return features
```

**技术指标：**
```python
def technical_indicators(df):
    """
    技术指标特征
    """
    features = pd.DataFrame()
    
    # 移动平均
    features['ma5'] = df['close'].rolling(5).mean() / df['close'] - 1
    features['ma10'] = df['close'].rolling(10).mean() / df['close'] - 1
    features['ma20'] = df['close'].rolling(20).mean() / df['close'] - 1
    
    # RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(14).mean()
    features['rsi_14'] = 100 - (100 / (1 + gain / loss))
    
    # MACD
    ema12 = df['close'].ewm(span=12, adjust=False).mean()
    ema26 = df['close'].ewm(span=26, adjust=False).mean()
    features['macd'] = ema12 - ema26
    features['macd_signal'] = features['macd'].ewm(span=9, adjust=False).mean()
    
    # 布林带
    features['boll_upper'] = df['close'].rolling(20).mean() + 2 * df['close'].rolling(20).std()
    features['boll_lower'] = df['close'].rolling(20).mean() - 2 * df['close'].rolling(20).std()
    features['boll_width'] = (features['boll_upper'] - features['boll_lower']) / df['close']
    
    return features
```

### 2.2 衍生特征

**交互特征：**
```python
def interaction_features(df, base_features):
    """
    交互特征
    
    参数:
    df: 数据
    base_features: 基础特征列表
    
    返回:
    交互特征
    """
    features = pd.DataFrame()
    
    # 特征交互
    for i, f1 in enumerate(base_features):
        for f2 in base_features[i+1:]:
            features[f'{f1}_x_{f2}'] = df[f1] * df[f2]
            features[f'{f1}_div_{f2}'] = df[f1] / (df[f2] + 1e-6)
    
    return features
```

**滞后特征：**
```python
def lag_features(df, features, lags=[1, 5, 10, 20]):
    """
    滞后特征
    
    参数:
    df: 数据
    features: 特征列表
    lags: 滞后期数
    
    返回:
    滞后特征
    """
    lagged = pd.DataFrame()
    
    for feature in features:
        for lag in lags:
            lagged[f'{feature}_lag{lag}'] = df[feature].shift(lag)
    
    return lagged
```

### 3.1 随机森林

**特征重要性：**
```python
from sklearn.ensemble import RandomForestRegressor
import numpy as np

def random_forest_factor_importance(X, y, n_estimators=100):
    """
    随机森林因子重要性
    
    参数:
    X: 特征矩阵
    y: 目标变量（收益率）
    n_estimators: 树的数量
    
    返回:
    特征重要性
    """
    # 训练模型
    rf = RandomForestRegressor(n_estimators=n_estimators, random_state=42, n_jobs=-1)
    rf.fit(X, y)
    
    # 特征重要性
    importance = pd.DataFrame({
        'feature': X.columns,
        'importance': rf.feature_importances_
    }).sort_values('importance', ascending=False)
    
    return importance, rf
```

**因子提取：**
```python
def extract_tree_factors(rf_model, X):
    """
    从树模型提取因子
    
    参数:
    rf_model: 训练好的随机森林模型
    X: 特征矩阵
    
    返回:
    树模型因子
    """
    # 叶节点编码
    leaf_nodes = rf_model.apply(X)
    
    # 每棵树的叶节点作为一个因子
    factors = pd.DataFrame(leaf_nodes, columns=[f'tree_{i}' for i in range(leaf_nodes.shape[1])])
    
    return factors
```

**训练与预测：**

import xgboost as xgb

def xgboost_factor_model(X_train, y_train, X_test, params=None):
    """
    XGBoost因子模型
    
    参数:
    X_train: 训练特征
    y_train: 训练目标
    X_test: 测试特征
    params: 模型参数
    
    返回:
    预测结果、特征重要性
    """
    if params is None:
        params = {
            'objective': 'reg:squarederror',
            'max_depth': 5,
            'learning_rate': 0.01,
            'n_estimators': 500,
            'subsample': 0.8,
            'colsample_bytree': 0.8,
            'random_state': 42
        }
    
    # 训练
    model = xgb.XGBRegressor(**params)
    model.fit(X_train, y_train, eval_set=[(X_test, y_test)], early_stopping_rounds=50, verbose=False)
    
    # 预测
    predictions = model.predict(X_test)
    
    # 特征重要性
    importance = pd.DataFrame({
        'feature': X_train.columns,
        'importance': model.feature_importances_
    }).sort_values('importance', ascending=False)
    
    return predictions, importance, model
```

**SHAP值解释：**

import shap

def shap_feature_importance(model, X):
    """
    SHAP特征重要性
    
    参数:
    model: 训练好的模型
    X: 特征矩阵
    
    返回:
    SHAP值
    """
    # 计算SHAP值
    explainer = shap.TreeExplainer(model)
    shap_values = explainer.shap_values(X)
    
    # 特征重要性
    shap_importance = pd.DataFrame({
        'feature': X.columns,
        'importance': np.abs(shap_values).mean(axis=0)
    }).sort_values('importance', ascending=False)
    
    return shap_values, shap_importance
```

import lightgbm as lgb

def lightgbm_factor_model(X_train, y_train, X_valid, y_valid, params=None):
    """
    LightGBM因子模型
    
    参数:
    X_train: 训练特征
    y_train: 训练目标
    X_valid: 验证特征

            'bagging_fraction': 0.8,

            'verbose': -1,

        }

## 4. 深度学习因子挖掘

### 4.1 MLP因子模型

```python
import torch
import torch.nn as nn

class MLPFactorModel(nn.Module):
    """多层感知机因子模型"""
    
    def __init__(self, input_dim, hidden_dims=[256, 128, 64]):
        super(MLPFactorModel, self).__init__()
        
        layers = []
        prev_dim = input_dim
        
        for hidden_dim in hidden_dims:
            layers.append(nn.Linear(prev_dim, hidden_dim))
            layers.append(nn.BatchNorm1d(hidden_dim))
            layers.append(nn.ReLU())
            layers.append(nn.Dropout(0.2))
            prev_dim = hidden_dim
        
        layers.append(nn.Linear(prev_dim, 1))
        
        self.network = nn.Sequential(*layers)
    
    def forward(self, x):
        return self.network(x)

def train_mlp(model, train_loader, valid_loader, epochs=100, lr=0.001):
    """
    训练MLP模型
    """
    optimizer = torch.optim.Adam(model.parameters(), lr=lr)
    criterion = nn.MSELoss()
    
    for epoch in range(epochs):
        model.train()
        for X, y in train_loader:
            optimizer.zero_grad()
            pred = model(X)
            loss = criterion(pred.squeeze(), y)
            loss.backward()
            optimizer.step()
        
        # 验证
        model.eval()
        valid_loss = 0
        with torch.no_grad():
            for X, y in valid_loader:
                pred = model(X)
                valid_loss += criterion(pred.squeeze(), y).item()
        
        if epoch % 10 == 0:
            print(f'Epoch {epoch}: Valid Loss = {valid_loss/len(valid_loader):.6f}')
    
    return model
```

### 4.2 LSTM因子模型

```python
class LSTMFactorModel(nn.Module):
    """LSTM因子模型"""
    
    def __init__(self, input_dim, hidden_dim=64, num_layers=2, dropout=0.2):
        super(LSTMFactorModel, self).__init__()
        
        self.hidden_dim = hidden_dim
        self.num_layers = num_layers
        
        self.lstm = nn.LSTM(input_dim, hidden_dim, num_layers, 
                           batch_first=True, dropout=dropout)
        self.fc = nn.Linear(hidden_dim, 1)
    
    def forward(self, x):
        # x: (batch, seq_len, input_dim)
        lstm_out, _ = self.lstm(x)
        out = self.fc(lstm_out[:, -1, :])
        return out

def prepare_lstm_data(features, target, seq_len=20):
    """
    准备LSTM数据
    
    参数:
    features: 特征 (n_samples, n_features)
    target: 目标 (n_samples,)
    seq_len: 序列长度
    
    返回:
    X: (n_samples, seq_len, n_features)
    y: (n_samples,)
    """
    X, y = [], []
    for i in range(seq_len, len(features)):
        X.append(features[i-seq_len:i])
        y.append(target[i])
    return np.array(X), np.array(y)
```

### 4.3 Transformer因子模型

```python
class TransformerFactorModel(nn.Module):
    """Transformer因子模型"""
    
    def __init__(self, input_dim, d_model=64, nhead=4, num_layers=2, dropout=0.2):
        super(TransformerFactorModel, self).__init__()
        
        self.embedding = nn.Linear(input_dim, d_model)
        self.pos_encoding = PositionalEncoding(d_model, dropout)
        
        encoder_layer = nn.TransformerEncoderLayer(d_model=d_model, nhead=nhead, dropout=dropout)
        self.transformer = nn.TransformerEncoder(encoder_layer, num_layers=num_layers)
        
        self.fc = nn.Linear(d_model, 1)
    
    def forward(self, x):
        # x: (batch, seq_len, input_dim)
        x = self.embedding(x)
        x = self.pos_encoding(x)
        x = x.permute(1, 0, 2)  # (seq_len, batch, d_model)
        x = self.transformer(x)
        x = x.permute(1, 0, 2)  # (batch, seq_len, d_model)
        out = self.fc(x[:, -1, :])
        return out

class PositionalEncoding(nn.Module):
    """位置编码"""
    
    def __init__(self, d_model, dropout=0.1, max_len=500):
        super(PositionalEncoding, self).__init__()
        self.dropout = nn.Dropout(p=dropout)
        
        pe = torch.zeros(max_len, d_model)
        position = torch.arange(0, max_len, dtype=torch.float).unsqueeze(1)
        div_term = torch.exp(torch.arange(0, d_model, 2).float() * (-np.log(10000.0) / d_model))
        pe[:, 0::2] = torch.sin(position * div_term)
        pe[:, 1::2] = torch.cos(position * div_term)
        pe = pe.unsqueeze(0)
        self.register_buffer('pe', pe)
    
    def forward(self, x):
        x = x + self.pe[:, :x.size(1), :]
        return self.dropout(x)
```

## 5. 因子检验与组合

### 5.1 机器学习因子检验

**IC检验：**
```python
def ml_factor_ic(predictions, returns, dates):
    """
    机器学习因子IC检验
    
    参数:
    predictions: 模型预测值
    returns: 真实收益率
    dates: 日期
    
    返回:
    IC时间序列、IC统计
    """
    from scipy.stats import spearmanr
    
    # 按日期计算IC
    ic_series = []
    for date in dates.unique():
        mask = dates == date
        pred_date = predictions[mask]
        ret_date = returns[mask]
        
        if len(pred_date) > 10:  # 至少10个样本
            ic, _ = spearmanr(pred_date, ret_date)
            ic_series.append({'date': date, 'ic': ic})
    
    ic_df = pd.DataFrame(ic_series)
    
    # IC统计
    ic_mean = ic_df['ic'].mean()
    ic_std = ic_df['ic'].std()
    icir = ic_mean / ic_std
    ic_positive_ratio = (ic_df['ic'] > 0).mean()
    
    print(f'IC均值: {ic_mean:.4f}')
    print(f'IC标准差: {ic_std:.4f}')
    print(f'ICIR: {icir:.4f}')
    print(f'IC胜率: {ic_positive_ratio:.2%}')
    
    return ic_df, {'ic_mean': ic_mean, 'ic_std': ic_std, 'icir': icir}
```

### 5.2 因子组合

**Stacking集成：**
```python
from sklearn.model_selection import KFold

def stacking_ensemble(X, y, models, meta_model):
    """
    Stacking集成
    
    参数:
    X: 特征
    y: 目标
    models: 基模型列表
    meta_model: 元模型
    
    返回:
    集成模型预测
    """
    n_samples = len(X)
    n_models = len(models)
    
    # K折交叉验证
    kf = KFold(n_splits=5, shuffle=True, random_state=42)
    
    # 元特征
    meta_features = np.zeros((n_samples, n_models))
    
    for i, model in enumerate(models):
        for train_idx, valid_idx in kf.split(X):
            X_train, X_valid = X[train_idx], X[valid_idx]
            y_train = y[train_idx]
            
            model.fit(X_train, y_train)
            meta_features[valid_idx, i] = model.predict(X_valid)
    
    # 训练元模型
    meta_model.fit(meta_features, y)
    
    return meta_model, models
```

## 6. 实战案例

### 6.1 完整流程

```python
class MLFactorPipeline:
    """机器学习因子挖掘流水线"""
    
    def __init__(self, data):
        self.data = data
        self.features = None
        self.model = None
    
    def feature_engineering(self):
        """特征工程"""
        # 价格特征
        price_feats = price_features(self.data)
        
        # 技术指标
        tech_feats = technical_indicators(self.data)
        
        # 财务特征（如果有）
        # financial_feats = financial_features(self.data)
        
        # 合并
        self.features = pd.concat([price_feats, tech_feats], axis=1)
        
        # 标准化
        self.features = (self.features - self.features.mean()) / self.features.std()
        
        return self
    
    def train_model(self, target, model_type='xgboost'):
        """训练模型"""
        # 准备数据
        X = self.features.dropna()
        y = target.loc[X.index]
        
        # 时间序列分割
        train_size = int(len(X) * 0.7)
        X_train, X_test = X[:train_size], X[train_size:]
        y_train, y_test = y[:train_size], y[train_size:]
        
        if model_type == 'xgboost':
            self.model, self.predictions, self.importance = xgboost_factor_model(
                X_train, y_train, X_test
            )
        
        return self
    
    def evaluate(self, y_true):
        """评估"""
        from scipy.stats import spearmanr
        from sklearn.metrics import mean_squared_error, r2_score
        
        # IC
        ic, _ = spearmanr(self.predictions, y_true)
        
        # MSE
        mse = mean_squared_error(y_true, self.predictions)
        
        # R²
        r2 = r2_score(y_true, self.predictions)
        
        print(f'IC: {ic:.4f}')
        print(f'MSE: {mse:.6f}')
        print(f'R²: {r2:.4f}')
        
        return {'ic': ic, 'mse': mse, 'r2': r2}
```

## 7. 总结

机器学习因子挖掘要点：

1. **特征工程**：价格特征、技术指标、衍生特征
2. **树模型**：随机森林、XGBoost、LightGBM
3. **深度学习**：MLP、LSTM、Transformer
4. **因子检验**：IC、分组回测、因子组合
5. **实战流程**：特征工程 → 模型训练 → 因子提取 → 检验

**注意事项：**
- 避免未来函数
- 时间序列交叉验证
- 特征选择
- 过拟合控制
- 可解释性
