```python
def technical_indicators(df):
    features = pd.DataFrame()
    
    # 均线系统
    for period in [5, 10, 20, 60]:
        features[f'ma{period}'] = df['close'].rolling(period).mean() / df['close'] - 1
    
    # RSI
    for period in [6, 14, 24]:
        delta = df['close'].diff()
        gain = delta.where(delta > 0, 0).rolling(period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(period).mean()
        features[f'rsi_{period}'] = 100 - (100 / (1 + gain / loss))
    
    # MACD
    ema12 = df['close'].ewm(span=12, adjust=False).mean()
    ema26 = df['close'].ewm(span=26, adjust=False).mean()
    features['macd'] = ema12 - ema26
    
    return features
```

```python
import torch
import torch.nn as nn

class NodeTransformer(nn.Module):
    """股票关系网络Transformer"""
    
    def __init__(self, num_stocks, feature_dim, num_heads=8):
        super().__init__()
        self.stock_embedding = nn.Embedding(num_stocks, feature_dim)
        self.transformer = nn.TransformerEncoder(
            nn.TransformerEncoderLayer(
                d_model=feature_dim,
                nhead=num_heads,
                batch_first=True
            ),
            num_layers=6
        )
        self.predictor = nn.Linear(feature_dim, 1)
    
    def forward(self, x, stock_ids):
        # 股票嵌入
        stock_emb = self.stock_embedding(stock_ids)
        x = x + stock_emb
        
        # Transformer编码
        x = self.transformer(x)
        
        # 预测未来收益
        pred = self.predictor(x)
        return pred
```

### 4.3 模型集成策略

**优化器多样性集成**：
```python
class OptimizerEnsemble:
    """优化器集成模型"""
    
    def __init__(self, base_model_class):
        self.models = []
        # 使用不同优化器训练相同架构
        for optimizer in ['adam', 'sgd', 'rmsprop', 'adamw']:
            model = base_model_class()
            model.optimizer = optimizer
            self.models.append(model)
    
    def fit(self, X, y):
        for model in self.models:
            model.fit(X, y)
    
    def predict(self, X):
        predictions = [model.predict(X) for model in self.models]
        # 加权平均
        return np.average(predictions, axis=0, weights=self.weights)
```

**欠规范性启示**：
- 相同架构+相同损失函数，不同优化器学习不同函数
- 对决策层面（换手率、夏普比）有实质影响
- 建议多优化器集成，提升策略多样性

```python
class DynamicTrackingError:
    """动态跟踪误差组合管理"""
    
    def __init__(self, risk_aversion, base_tracking_limit):
        self.risk_aversion = risk_aversion
        self.base_limit = base_tracking_limit
    
    def adjust_tracking_limit(self, market_volatility, crisis_mode):
        """动态调整跟踪误差限制"""
        if crisis_mode:
            # 危机期约束成本最高，前瞻收益最丰富
```

```python
def out_of_sample_test(model, X_train, y_train, X_test, y_test):
    """样本外测试"""
    model.fit(X_train, y_train)
    
    # 样本内表现
    train_pred = model.predict(X_train)
    train_ic = np.corrcoef(train_pred, y_train)[0, 1]
    
    # 样本外表现
    test_pred = model.predict(X_test)
    test_ic = np.corrcoef(test_pred, y_test)[0, 1]
    
    # 过拟合检测
    overfitting_ratio = train_ic / test_ic
    if overfitting_ratio > 2.0:
        print(f"警告：可能过拟合，比值={overfitting_ratio:.2f}")
    
    return train_ic, test_ic
```

**Walk-Forward分析**：
```python
def walk_forward_analysis(data, model_class, n_splits=5):
    """Walk-Forward验证"""
    results = []
    tscv = TimeSeriesSplit(n_splits=n_splits)
    
    for train_idx, test_idx in tscv.split(data):
        X_train, X_test = data[train_idx], data[test_idx]
        y_train, y_test = labels[train_idx], labels[test_idx]
        
        model = model_class()
        model.fit(X_train, y_train)
        
        pred = model.predict(X_test)
        ic = np.corrcoef(pred, y_test)[0, 1]
        results.append(ic)
    
    return np.mean(results), np.std(results)
```

```python
class RiskMonitor:
    """实时风险监控"""
    
    def __init__(self, max_drawdown=0.15, max_position=0.1):
        self.max_drawdown = max_drawdown
        self.max_position = max_position
    
    def check_position(self, weights):
        """检查仓位限制"""
        if (weights > self.max_position).any():
            raise RiskAlert("单一资产超限")
    
    def check_drawdown(self, portfolio_value):
        """检查回撤限制"""
        peak = portfolio_value.expanding().max()
        drawdown = (portfolio_value - peak) / peak
        
        if drawdown.min() < -self.max_drawdown:
            self.trigger_circuit_breaker()
```

**事后风控**：
