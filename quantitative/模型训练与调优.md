## 1. 数据准备

### 1.1 数据划分

```python
from sklearn.model_selection import train_test_split

def prepare_data(features, target, test_size=0.2, val_size=0.2):
    """
    准备数据
    
    参数:
    features: 特征矩阵
    target: 目标变量
    test_size: 测试集比例
    val_size: 验证集比例
    
    返回:
    训练集、验证集、测试集
    """
    # 时间序列划分
    n = len(features)
    train_end = int(n * (1 - test_size - val_size))
    val_end = int(n * (1 - test_size))
    
    X_train = features.iloc[:train_end]
    y_train = target.iloc[:train_end]
    
    X_val = features.iloc[train_end:val_end]
    y_val = target.iloc[train_end:val_end]
    
    X_test = features.iloc[val_end:]
    y_test = target.iloc[val_end:]
    
    return X_train, X_val, X_test, y_train, y_val, y_test
```

### 1.2 时间序列交叉验证

```python
from sklearn.model_selection import TimeSeriesSplit

def time_series_cv(X, y, n_splits=5):
    """
    时间序列交叉验证
    
    参数:
    X: 特征矩阵
    y: 目标变量
    n_splits: 分割数
    
    返回:
    训练索引、验证索引
    """
    tscv = TimeSeriesSplit(n_splits=n_splits)
    
    for train_idx, val_idx in tscv.split(X):
        X_train, X_val = X.iloc[train_idx], X.iloc[val_idx]
        y_train, y_val = y.iloc[train_idx], y.iloc[val_idx]
        yield X_train, X_val, y_train, y_val
```

## 2. 模型训练

### 2.1 基础模型

**线性回归：**
```python
from sklearn.linear_model import Ridge, Lasso

def train_linear_model(X_train, y_train, alpha=1.0):
    """
    训练线性模型
    
    参数:
    X_train: 训练特征
    y_train: 训练目标
    alpha: 正则化参数
    
    返回:
    训练好的模型
    """
    # Ridge回归
    ridge = Ridge(alpha=alpha)
    ridge.fit(X_train, y_train)
    
    # Lasso回归
    lasso = Lasso(alpha=alpha)
    lasso.fit(X_train, y_train)
    
    return ridge, lasso
```

**树模型：**
```python
from sklearn.ensemble import RandomForestRegressor
import xgboost as xgb
import lightgbm as lgb

def train_tree_models(X_train, y_train):

    训练树模型

    X_train: 训练特征

    返回:
    训练好的模型

        random_state=42,
        n_jobs=-1
    )
    lgb_model.fit(X_train, y_train)
```

## 3. 超参数调优

### 3.1 网格搜索

```python
from sklearn.model_selection import GridSearchCV

def grid_search(X, y, model, param_grid, cv=5):
    """
    网格搜索
    
    参数:
    X: 特征
    y: 目标
    model: 模型
    param_grid: 参数网格
    cv: 交叉验证数
    
    返回:
    最优模型
    """
    grid_search = GridSearchCV(
        model, param_grid, cv=cv, scoring='neg_mean_squared_error', n_jobs=-1
    )
    grid_search.fit(X, y)
    
    print(f'最优参数: {grid_search.best_params_}')
    print(f'最优得分: {grid_search.best_score_}')
    
    return grid_search.best_estimator_

# 示例
param_grid = {
    'n_estimators': [50, 100, 200],
    'max_depth': [3, 5, 7],
    'learning_rate': [0.01, 0.1, 0.2]
}

best_model = grid_search(X_train, y_train, xgb.XGBRegressor(), param_grid)
```

## 5. 模型集成

### 5.1 Stacking

```python
from sklearn.ensemble import StackingRegressor

def stacking_ensemble(base_models, meta_model, X_train, y_train):
    """
    Stacking集成
    
    参数:
    base_models: 基模型列表
    meta_model: 元模型
    X_train: 训练特征
    y_train: 训练目标
    
    返回:
    Stacking模型
    """
    stacking = StackingRegressor(
        estimators=base_models,
        final_estimator=meta_model,
        cv=5
    )
    stacking.fit(X_train, y_train)
    
    return stacking

# 示例
base_models = [
    ('rf', RandomForestRegressor(n_estimators=100)),
    ('xgb', xgb.XGBRegressor(n_estimators=100)),
    ('lgb', lgb.LGBMRegressor(n_estimators=100))
]
meta_model = Ridge()

stacking_model = stacking_ensemble(base_models, meta_model, X_train, y_train)
```
