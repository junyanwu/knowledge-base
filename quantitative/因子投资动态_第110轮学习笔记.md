### 2.3 价值因子优化

#### 质量加权价值因子
```python
def quality_weighted_value(pe, pb, roe, debt_ratio):
    """质量加权价值因子"""
    # 价值得分
    value_score = - (pe.rank() + pb.rank()) / 2  # 低估值得分高
    
    # 质量得分
    quality_score = (roe.rank() - debt_ratio.rank()) / 2
    
    # 综合得分
    combined_score = 0.6 * value_score + 0.4 * quality_score
    
    return combined_score
```

#### 动态价值因子
- 根据经济周期调整价值因子权重
- 经济复苏期：提高价值因子权重
- 经济衰退期：降低价值因子权重

```python
def momentum_crash_risk(momentum_returns, lookback=60):
    """动量崩溃风险评估"""
    # 计算动量收益的下行波动率
    downside_vol = momentum_returns[momentum_returns < 0].std()
    
    # 计算动量收益的偏度
    skewness = momentum_returns.skew()
    
    # 动量崩溃风险指标
    crash_risk = downside_vol * (1 - skewness)
    
    return crash_risk
```

```python
def composite_quality_factor(df):
    """综合质量因子"""
    # 盈利能力
    profitability = (df['roe'].rank() + df['roa'].rank()) / 2
    
    # 盈利质量
    earning_quality = (df['cash_flow'] / df['net_profit']).rank()
    
    # 财务安全
    financial_safety = (1 - df['debt_ratio']).rank()
    
    # 盈利增长
    growth = (df['revenue_growth'].rank() + df['profit_growth'].rank()) / 2
    
    # 综合得分
    quality_score = (0.4 * profitability + 
                     0.2 * earning_quality + 
                     0.2 * financial_safety + 
                     0.2 * growth)
    
    return quality_score
```

```python
def volatility_factor(prices, window=60):
    """波动率因子"""
    returns = prices.pct_change()
    volatility = returns.rolling(window).std()
    return volatility
```

#### Beta因子
```python
def beta_factor(stock_returns, market_returns, window=60):
    """Beta因子"""
    covariance = stock_returns.rolling(window).cov(market_returns)
    market_variance = market_returns.rolling(window).var()
    beta = covariance / market_variance
    return beta
```

```python
class IndustryRotation:
    """行业轮动三标尺"""
    
    def __init__(self):
        self.momentum_window = 60
        self.lookback = 120
    
    def calculate_industry_momentum(self, industry_returns):
        """行业动量"""
        return industry_returns.rolling(self.momentum_window).mean()
    
    def calculate_industry_crowding(self, industry_positions):
        """行业拥挤度"""
        # 基于持仓集中度、换手率、波动率综合计算
        position_concentration = industry_positions / industry_positions.sum()
        crowding_score = position_concentration.rolling(20).std()
        return crowding_score
    
    def generate_signal(self, momentum, crowding):
        """生成轮动信号"""
        # 动量高且拥挤度低
        signal = (momentum.rank() - crowding.rank()).rank()
        return signal
```

### 6.3 多因子组合优化

#### 均值方差优化
```python
def mean_variance_optimization(factor_returns, target_vol=0.15):
    """均值方差优化因子权重"""
    from scipy.optimize import minimize
    
    mu = factor_returns.mean()
    cov = factor_returns.cov()
    n = len(mu)
    
    def objective(w):
        portfolio_return = w @ mu
        portfolio_vol = np.sqrt(w @ cov @ w)
        # 最大化夏普比率
        return -portfolio_return / portfolio_vol
    
    constraints = [{'type': 'eq', 'fun': lambda w: np.sum(w) - 1}]
    bounds = [(0, 0.3) for _ in range(n)]  # 单因子权重上限30%
    
    result = minimize(objective, np.ones(n)/n, 
                     method='SLSQP', 
                     bounds=bounds, 
                     constraints=constraints)
    
    return result.x
```

### 8.3 因子动态权重

#### 因子择时信号
```python
def factor_timing_signals(factor_returns, lookback=60):
    """基于历史表现动态调整因子权重"""
    signals = {}
    
    # 动量信号：过去N期因子收益
    signals['momentum'] = factor_returns.rolling(lookback).mean()
    
    # 波动率信号：因子波动率倒数
    signals['volatility'] = 1 / factor_returns.rolling(lookback).std()
    
    # 夏普信号：因子夏普比率
    signals['sharpe'] = (factor_returns.rolling(lookback).mean() / 
                         factor_returns.rolling(lookback).std())
    
    # 综合信号
    combined_signal = (signals['momentum'] + signals['volatility'] + signals['sharpe']) / 3
    
    return combined_signal
```

### 8.4 风险管理

#### 因子暴露监控
```python
class FactorExposureMonitor:
    """因子暴露监控"""
    
    def __init__(self, factor_models):
        self.models = factor_models
    
    def calculate_exposure(self, portfolio_weights, factor_data):
        """计算组合因子暴露"""
        exposures = {}
        for factor_name, factor_values in factor_data.items():
            exposure = (portfolio_weights * factor_values).sum()
            exposures[factor_name] = exposure
        return exposures
    
    def check_risk(self, exposures, limits):
        """检查因子暴露是否超限"""
        alerts = []
        for factor, exposure in exposures.items():
            if abs(exposure) > limits.get(factor, 1.0):
                alerts.append(f'{factor}暴露超限：{exposure:.2f}')
        return alerts
```
