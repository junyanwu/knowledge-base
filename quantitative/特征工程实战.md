## 1. 金融特征分类

### 1.1 价格特征

**基础价格特征：**
```python
def price_features(df):
    """
    价格相关特征
    
    参数:
    df: 包含open, high, low, close的DataFrame
    
    返回:
    特征DataFrame
    """
    features = pd.DataFrame(index=df.index)
    
    # 收益率
    features['return_1d'] = df['close'].pct_change(1)
    features['return_5d'] = df['close'].pct_change(5)
    features['return_10d'] = df['close'].pct_change(10)
    features['return_20d'] = df['close'].pct_change(20)
    
    # 价格位置
    features['price_pos_20d'] = (df['close'] - df['low'].rolling(20).min()) / \
                                  (df['high'].rolling(20).max() - df['low'].rolling(20).min())
    
    # 价格动量
    features['momentum_5d'] = df['close'] / df['close'].shift(5) - 1
    features['momentum_20d'] = df['close'] / df['close'].shift(20) - 1
    
    # 高低点距离
    features['high_low_ratio'] = df['high'] / df['low']
    features['close_to_high'] = df['close'] / df['high']
    features['close_to_low'] = df['close'] / df['low']
    
    return features
```

### 1.2 技术指标特征

**趋势指标：**
```python
def trend_features(df):
    """
    趋势指标特征
    """
    features = pd.DataFrame(index=df.index)
    
    # 移动平均
    features['ma5'] = df['close'].rolling(5).mean() / df['close'] - 1
    features['ma10'] = df['close'].rolling(10).mean() / df['close'] - 1
    features['ma20'] = df['close'].rolling(20).mean() / df['close'] - 1
    features['ma60'] = df['close'].rolling(60).mean() / df['close'] - 1
    
    # MA交叉
    features['ma5_cross_ma10'] = (df['close'].rolling(5).mean() > df['close'].rolling(10).mean()).astype(int)
    features['ma10_cross_ma20'] = (df['close'].rolling(10).mean() > df['close'].rolling(20).mean()).astype(int)
    
    # MACD
    ema12 = df['close'].ewm(span=12, adjust=False).mean()
    ema26 = df['close'].ewm(span=26, adjust=False).mean()
    features['macd'] = ema12 - ema26
    features['macd_signal'] = features['macd'].ewm(span=9, adjust=False).mean()
    features['macd_hist'] = features['macd'] - features['macd_signal']
    
    return features
```

def volatility_features(df):
    """
    波动率指标特征
    """
    features = pd.DataFrame(index=df.index)
    
    # 历史波动率
    features['volatility_5d'] = df['close'].pct_change().rolling(5).std() * np.sqrt(252)
    features['volatility_10d'] = df['close'].pct_change().rolling(10).std() * np.sqrt(252)
    features['volatility_20d'] = df['close'].pct_change().rolling(20).std() * np.sqrt(252)
    
    # ATR
    high = df['high']
    low = df['low']
    close = df['close']
    
    tr = pd.concat([
        high - low,
        abs(high - close.shift(1)),
        abs(low - close.shift(1))
    ], axis=1).max(axis=1)
    
    features['atr_14'] = tr.rolling(14).mean()
    features['atr_ratio'] = features['atr_14'] / close
    
    # 布林带
    features['boll_upper'] = df['close'].rolling(20).mean() + 2 * df['close'].rolling(20).std()
    features['boll_lower'] = df['close'].rolling(20).mean() - 2 * df['close'].rolling(20).std()
    features['boll_width'] = (features['boll_upper'] - features['boll_lower']) / df['close']
    features['boll_position'] = (df['close'] - features['boll_lower']) / (features['boll_upper'] - features['boll_lower'])
    
    return features
```

def momentum_features(df):
    """
    动量指标特征
    """
    features = pd.DataFrame(index=df.index)
    
    # RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(14).mean()
    features['rsi_14'] = 100 - (100 / (1 + gain / loss))
    
    # KDJ
    low_9 = df['low'].rolling(9).min()
    high_9 = df['high'].rolling(9).max()
    
    features['kdj_k'] = (df['close'] - low_9) / (high_9 - low_9) * 100
    features['kdj_d'] = features['kdj_k'].rolling(3).mean()
    features['kdj_j'] = 3 * features['kdj_k'] - 2 * features['kdj_d']
    
    # Williams %R
    features['williams_r'] = (high_9 - df['close']) / (high_9 - low_9) * 100
    
    # CCI
    tp = (df['high'] + df['low'] + df['close']) / 3
    features['cci'] = (tp - tp.rolling(20).mean()) / (0.015 * tp.rolling(20).std())
    
    return features
```

### 1.3 成交量特征

```python
def volume_features(df):
    """
    成交量特征
    """
    features = pd.DataFrame(index=df.index)
    
    # 成交量移动平均
    features['vol_ma5'] = df['volume'].rolling(5).mean()
    features['vol_ma10'] = df['volume'].rolling(10).mean()
    features['vol_ma20'] = df['volume'].rolling(20).mean()
    
    # 成交量比率
    features['vol_ratio'] = df['volume'] / features['vol_ma5']
    features['vol_ratio_20'] = df['volume'] / features['vol_ma20']
    
    # 成交量动量
    features['vol_momentum_5d'] = df['volume'] / df['volume'].shift(5)
    features['vol_momentum_20d'] = df['volume'] / df['volume'].shift(20)
    
    # OBV
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    features['obv'] = obv
    features['obv_ma20'] = obv.rolling(20).mean()
    
    # VWAP
    features['vwap'] = (df['close'] * df['volume']).rolling(20).sum() / df['volume'].rolling(20).sum()
    features['price_to_vwap'] = df['close'] / features['vwap']
    
    return features
```

## 2. 衍生特征

### 2.1 统计特征

```python
def statistical_features(df, window=20):
    """
    统计特征
    
    参数:
    df: 价格数据
    window: 窗口期
    
    返回:
    特征DataFrame
    """
    features = pd.DataFrame(index=df.index)
    
    returns = df['close'].pct_change()
    
    # 偏度
    features['skewness'] = returns.rolling(window).skew()
    
    # 峰度
    features['kurtosis'] = returns.rolling(window).kurt()
    
    # 分位数
    features['quantile_25'] = returns.rolling(window).quantile(0.25)
    features['quantile_75'] = returns.rolling(window).quantile(0.75)
    features['iqr'] = features['quantile_75'] - features['quantile_25']
    
    # 最大值/最小值
    features['max_return'] = returns.rolling(window).max()
    features['min_return'] = returns.rolling(window).min()
    features['range'] = features['max_return'] - features['min_return']
    
    return features
```

### 2.2 滞后特征

```python
def lag_features(df, features, lags=[1, 2, 3, 5, 10, 20]):
    """
    滞后特征
    
    参数:
    df: 数据
    features: 特征列表
    lags: 滞后期
    
    返回:
    滞后特征
    """
    lagged = pd.DataFrame(index=df.index)
    
    for feature in features:
        for lag in lags:
            lagged[f'{feature}_lag{lag}'] = df[feature].shift(lag)
    
    return lagged
```

### 2.3 交互特征

```python
def interaction_features(df, feature_pairs):
    """
    交互特征
    
    参数:
    df: 数据
    feature_pairs: 特征对列表
    
    返回:
    交互特征
    """
    features = pd.DataFrame(index=df.index)
    
    for f1, f2 in feature_pairs:
        # 乘积
        features[f'{f1}_x_{f2}'] = df[f1] * df[f2]
        
        # 比率
        features[f'{f1}_div_{f2}'] = df[f1] / (df[f2] + 1e-6)
        
        # 差分
        features[f'{f1}_minus_{f2}'] = df[f1] - df[f2]
    
    return features
```

## 3. 特征选择

### 3.1 相关性过滤

```python
def correlation_filter(features, target, threshold=0.05):
    """
    相关性过滤
    
    参数:
    features: 特征矩阵
    target: 目标变量
    threshold: 相关性阈值
    
    返回:
    过滤后的特征
    """
    # 计算与目标的相关性
    correlations = features.corrwith(target).abs()
    
    # 过滤低相关性特征
    selected_features = correlations[correlations > threshold].index
    
    return features[selected_features]
```

### 3.2 递归特征消除

```python
from sklearn.feature_selection import RFE
from sklearn.ensemble import RandomForestRegressor

def recursive_feature_elimination(X, y, n_features=50):
    """
    递归特征消除
    
    参数:
    X: 特征矩阵
    y: 目标变量
    n_features: 目标特征数
    
    返回:
    选中的特征
    """
    # 基模型
    estimator = RandomForestRegressor(n_estimators=100, random_state=42)
    
    # RFE
    rfe = RFE(estimator, n_features_to_select=n_features)
    rfe = rfe.fit(X, y)
    
    # 选中的特征
    selected_features = X.columns[rfe.support_]
    
    return selected_features
```

### 3.3 特征重要性

```python
def feature_importance_selection(X, y, n_features=50):
    """
    特征重要性选择
    
    参数:
    X: 特征矩阵
    y: 目标变量
    n_features: 目标特征数
    
    返回:
    选中的特征
    """
    # 训练模型
    model = RandomForestRegressor(n_estimators=100, random_state=42)
    model.fit(X, y)
    
    # 特征重要性
    importance = pd.DataFrame({
        'feature': X.columns,
        'importance': model.feature_importances_
    }).sort_values('importance', ascending=False)
    
    # 选择top特征
    selected_features = importance.head(n_features)['feature'].values
    
    return selected_features, importance
```

## 4. 特征工程最佳实践

### 4.1 完整流程

```python
class FeatureEngineering:
    """特征工程流水线"""
    
    def __init__(self, df):
        self.df = df
        self.features = pd.DataFrame(index=df.index)
    
    def generate_all_features(self):
        """生成所有特征"""
        # 价格特征
        price_feats = price_features(self.df)
        
        # 技术指标
        trend_feats = trend_features(self.df)
        vol_feats = volatility_features(self.df)
        mom_feats = momentum_features(self.df)
        
        # 成交量特征
        vol_feats = volume_features(self.df)
        
        # 统计特征
        stat_feats = statistical_features(self.df)
        
        # 合并
        self.features = pd.concat([
            price_feats, trend_feats, vol_feats, mom_feats, vol_feats, stat_feats
        ], axis=1)
        
        # 删除NaN
        self.features = self.features.dropna()
        
        return self
    
    def select_features(self, target, n_features=50):
        """特征选择"""
        # 相关性过滤
        self.features = correlation_filter(self.features, target, threshold=0.05)
        
        # 特征重要性
        selected, importance = feature_importance_selection(
            self.features, target.loc[self.features.index], n_features
        )
        
        self.features = self.features[selected]
        self.importance = importance
        
        return self
    
    def normalize(self):
        """标准化"""
        from sklearn.preprocessing import StandardScaler
        
        scaler = StandardScaler()
        self.features = pd.DataFrame(
            scaler.fit_transform(self.features),
            index=self.features.index,
            columns=self.features.columns
        )
        
        return self
```

### 4.2 避免未来函数

```python
def check_future_functions(features, max_lag=20):
    """
    检查未来函数
    
    参数:
    features: 特征DataFrame
    max_lag: 最大检查滞后期
    
    返回:
    有问题的特征列表
    """
    problematic_features = []
    
    for col in features.columns:
        # 检查是否有未来数据泄露
        if 'future' in col.lower() or 'next' in col.lower():
            problematic_features.append(col)
    
    return problematic_features
```
